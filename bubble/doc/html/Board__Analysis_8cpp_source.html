38 void Routing::activate_node(const Node& node) {
39  current_arc_.add_uid(node);
40  for (Node neigh : original_arc_.get_neighbour_uids(node)) {
41  if (node_active(qmap, neigh)) {
42  if (original_arc_.connection_exists(node, neigh)) {

45  if (original_arc_.connection_exists(neigh, node)) {
46  current_arc_.add_connection(neigh, node);
