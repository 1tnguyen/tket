42 static std::vector<std::pair<Node, Node>> as_nodepairs(
43  const std::vector<std::pair<unsigned, unsigned>> &edges) {
44  std::vector<std::pair<Node, Node>> nodepairs;
45  nodepairs.reserve(edges.size());
46  for (auto &[m, n] : edges) {

48  }
