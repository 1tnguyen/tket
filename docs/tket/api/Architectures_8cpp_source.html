38 unsigned Architecture::get_diameter() const {
39  unsigned N = n_uids();
40  if (N == 0) {
41  throw ArchitectureInvalidity("No nodes in architecture.");
42  }
43  unsigned max = 0;
44  const node_vector_t uids = get_all_uids_vec();
45  for (unsigned i = 0; i < N; i++) {
46  for (unsigned j = i + 1; j < N; j++) {
47  unsigned d = get_distance(uids[i], uids[j]);
48  if (d > max) max = d;
49  }
50  }
51  return max;
52 }

54 // Given a vector of lengths of lines, returns a vector of lines of these sizes
55 // comprised of architecture nodes
56 std::vector<node_vector_t> Architecture::get_lines(
57  std::vector<unsigned> required_lengths) const {
58  // check total length doesn't exceed number of nodes
59  if (std::accumulate(required_lengths.begin(), required_lengths.end(), 0u) >
60  n_uids()) {
61  throw ArchitectureInvalidity(
62  "Not enough nodes to satisfy required lengths.");
63  }
64  std::sort(
65  required_lengths.begin(), required_lengths.end(),
66  std::greater<unsigned>());
67 
68  UndirectedConnGraph curr_graph(get_undirected_connectivity());
69  std::vector<node_vector_t> found_lines;
70  for (unsigned length : required_lengths) {
71  std::vector<Vertex> longest(graph::longest_simple_path(curr_graph, length));
72  if (longest.size() >= length) {
73  longest.resize(length);
74  // convert Vertex to Node
75  node_vector_t to_add;
76  for (Vertex v : longest) {
77  to_add.push_back(curr_graph[v].uid);
78  }
79  found_lines.push_back(to_add);
80  for (Vertex v : longest) {
81  boost::clear_vertex(v, curr_graph);
82  }
83  }
84  }
85  return found_lines;
86 }

118 static bool lexicographical_comparison(
119  const std::vector<std::size_t>& dist1,
120  const std::vector<std::size_t>& dist2) {
121  return std::lexicographical_compare(
122  dist1.begin(), dist1.end(), dist2.begin(), dist2.end());

125 int tri_lexicographical_comparison(
126  const std::vector<std::size_t>& dist1,
127  const std::vector<std::size_t>& dist2) {
128  // add assertion that these are the same size distance vectors
129  auto start_dist1 = dist1.cbegin();
130  auto start_dist2 = dist2.cbegin();
131 
132  while (start_dist1 != dist1.cend()) {
133  if (start_dist2 == dist2.cend() || *start_dist2 < *start_dist1) {
134  return 0;
135  } else if (*start_dist1 < *start_dist2) {
136  return 1;
137  }
138  ++start_dist1;
139  ++start_dist2;
140  }
141  return -1;
142 }

144 MatrixXb Architecture::get_connectivity() const {
145  unsigned n = n_uids();
146  MatrixXb connectivity = MatrixXb(n, n);
147  for (unsigned i = 0; i != n; ++i) {
148  for (unsigned j = 0; j != n; ++j) {
149  connectivity(i, j) = connection_exists(Node(i), Node(j)) |
150  connection_exists(Node(j), Node(i));
151  }
152  }
153  return connectivity;
154 }

156 std::optional<Node> Architecture::find_worst_node(
157  const Architecture& original_arch) {
158  node_set_t ap = get_articulation_points();
159  node_set_t min_nodes = min_degree_uids();
160 
161  std::set<Node> bad_nodes;
162  std::set_difference(
163  min_nodes.cbegin(), min_nodes.cend(), ap.cbegin(), ap.cend(),
164  std::inserter(bad_nodes, bad_nodes.begin()));
165 
166  if (bad_nodes.empty()) {
167  return std::nullopt;
168  }
169 
170  std::vector<std::size_t> worst_distances, temp_distances;
171  Node worst_node = *bad_nodes.begin();
172  worst_distances = get_distances(worst_node);
173  for (Node temp_node : bad_nodes) {
174  temp_distances = get_distances(temp_node);
175 
176  int distance_comp =
177  tri_lexicographical_comparison(temp_distances, worst_distances);
178  if (distance_comp == 1) {
179  worst_node = temp_node;
180  worst_distances = temp_distances;
181  } else if (distance_comp == -1) {
182  std::vector<std::size_t> temp_distances_full =
183  original_arch.get_distances(temp_node);
184  std::vector<std::size_t> worst_distances_full =
185  original_arch.get_distances(worst_node);
186  if (lexicographical_comparison(
187  temp_distances_full, worst_distances_full)) {
188  worst_node = temp_node;
189  worst_distances = temp_distances;
190  }
191  }
192  }
193  return worst_node;
194 }

207  // Preserve the internal order of ids since Placement depends on this
208  auto uid_its = ar.get_all_uids();
209  std::vector<Node> nodes{uid_its.begin(), uid_its.end()};
210  j["nodes"] = nodes;
211 
212  nlohmann::json links;
213  for (const Architecture::Connection& con : ar.get_connections_vec()) {
214  nlohmann::json entry;
215  entry["link"] = con;
216  entry["weight"] = ar.get_connection_weight(con.first, con.second);
217  links.push_back(entry);
218  }
219  j["links"] = links;
220 }
