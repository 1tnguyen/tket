43 // this method annihilates all primitives next to each other (accounting for
44 // previous annihilations)
45 // also removes redundant non-classically controlled Z basis gates before a z
46 // basis measurement so that eg. -H-X-X-H- always annihilates to -----
47 static bool redundancy_removal(Circuit &circ) {
48  bool success = false;
49  bool found_redundancy = true;
50  IndexMap im = circ.index_map();
51  std::set<IVertex> old_affected_verts;
52  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
53  old_affected_verts.insert({im.at(v), v});
54  }
55  VertexList bin;
56  while (found_redundancy) {
57  std::set<IVertex> new_affected_verts;
58  for (const IVertex &p : old_affected_verts) {
59  remove_redundancy(circ, p.second, bin, new_affected_verts, im);
60  }
61  found_redundancy = new_affected_verts.size() != 0;
62  success |= found_redundancy;
63  old_affected_verts = new_affected_verts;
64  }
65  circ.remove_vertices(
66  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
67  return success;
68 }

70 // called by the previous method. This should generally not be called
71 // independently
72 static bool remove_redundancy(
73  Circuit &circ, const Vertex &vert, VertexList &bin,
74  std::set<IVertex> &new_affected_verts, IndexMap &im) {
75  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
76  const OpDesc desc = op->get_desc();
77  if (!desc.is_gate()) return false;
78  if (circ.n_out_edges(vert) == 0 || circ.n_in_edges(vert) == 0) {
79  return false; // either a boundary vert or we have already removed it
80  }
81 
82  auto remove_single_vertex = [&bin, &circ, &new_affected_verts,
83  &im](const Vertex &v_remove) {
84  bin.push_back(v_remove);
85  for (const Vertex &l : circ.get_predecessors(v_remove)) {
86  new_affected_verts.insert({im.at(l), l});
87  }
88  circ.remove_vertex(
89  v_remove, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
90  };
91  // remove 0 angle rotations from circuit
92  std::optional<double> a = op->is_identity();
93  if (a) {
94  remove_single_vertex(vert);
95  circ.add_phase(a.value());
96  return true;
97  } else if (desc.type() == OpType::noop) {
98  // remove "noop" gates from circuit
99  remove_single_vertex(vert);
100  return true;
101  }
102  VertexVec kids = circ.get_successors(vert);
103 
104  // if op is immediately followed by a z basis measurement on all qubits,
105  // remove
106  if (circ.n_out_edges_of_type(vert, EdgeType::Classical) == 0) {
107  bool z_followed_by_measures = true;
108  for (port_t port = 0; port < kids.size() && z_followed_by_measures;
109  port++) {
110  if (circ.get_OpType_from_Vertex(kids[port]) == OpType::Measure) {
111  z_followed_by_measures &= op->commutes_with_basis(Pauli::Z, port);
112  } else {
113  z_followed_by_measures = false;
114  }
115  }
116  if (z_followed_by_measures) {
117  remove_single_vertex(vert);
118  return true;
119  }
120  }
121 
122  // check that both the vertex and its successor have each other and only each
123  // other
124  if ((kids.size() == 1) && (circ.get_predecessors(kids[0]).size() == 1)) {
125  // check that the ports match up between vertices
126  Vertex b = kids[0];
127  EdgeVec ins = circ.get_in_edges(b);
128  for (const Edge &in : ins) {
129  if (circ.get_source_port(in) != circ.get_target_port(in)) return false;
130  }
131 
132  // check that the classical edges match up correctly
133  if (circ.n_in_edges_of_type(vert, EdgeType::Boolean) != 0) return false;
134 
135  const Op_ptr b_op = circ.get_Op_ptr_from_Vertex(b);
136  const OpDesc b_desc = b_op->get_desc();
137 
138  if (!b_desc.is_oneway()) {
139  // if A = B.dagger(), AB = I
140  // This method cannot detect matches between rotation gates.
141  // Rotation gates are covered by the rotation gate combiner, everything
142  // else in this.
143  if (*b_op->dagger() == *op) {
144  bin.push_back(vert);
145  bin.push_back(b);
146  VertexVec last_verts = circ.get_predecessors(vert);
147  for (const Vertex &l : last_verts) {
148  new_affected_verts.insert({im.at(l), l});
149  }
150  VertexList to_detach{vert, b};
151  // detached from circuit but not removed from graph
152  circ.remove_vertices(
153  to_detach, Circuit::GraphRewiring::Yes,
154  Circuit::VertexDeletion::No);
155  return true;
156  } else if (desc.is_rotation()) {
157  // combine two rotation gates together, then if the combined
158  // operation is the identity up to phase, remove from circuit
159  if (b_desc.type() == desc.type()) {
160  Expr expr1 = op->get_params()[0];
161  Expr expr2 = b_op->get_params()[0];
162  VertexVec last_verts = circ.get_predecessors(vert);
163  for (const Vertex &l : last_verts) {
164  new_affected_verts.insert({im.at(l), l});
165  }
166  circ.remove_vertex(
167  b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
168  bin.push_back(b);
169  std::vector<Expr> params_new = {expr1 + expr2};
170  Op_ptr op_new = get_op_ptr(desc.type(), params_new, ins.size());
171  std::optional<double> a = op_new->is_identity();
172  if (a) {
173  bin.push_back(vert);
174  circ.remove_vertex(
175  vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
176  circ.add_phase(a.value());
177  } else {
178  new_affected_verts.insert({im[vert], vert});
179  circ.dag[vert].op = op_new;
180  }
181  return true;
182  }
183  }
184  }
185  }
186  return false;
187 }

202 // moves single qubit operations past multiqubit operations they commute with,
203 // towards front of circuit (hardcoded)
204 static bool commute_singles_to_front(Circuit &circ) {
205  bool success = false;
206  // follow each qubit path from output to input
207  for (const Qubit &q : circ.all_qubits()) {
208  Vertex prev_v = circ.get_out(q);
209  Edge current_e = circ.get_nth_in_edge(prev_v, 0);
210  Vertex current_v = circ.source(current_e);
211  while (!is_initial_q_type(circ.get_OpType_from_Vertex(current_v))) {
212  const Op_ptr curr_op = circ.get_Op_ptr_from_Vertex(current_v);
213  // if current vertex is a multiqubit gate
214  if (circ.n_in_edges_of_type(current_v, EdgeType::Quantum) > 1 &&
215  curr_op->get_desc().is_gate()) {
216  // need gate check to access commuting_basis
217  const std::pair<port_t, port_t> ports = circ.get_ports(current_e);
218  check_prev_commutes:
219  const Op_ptr prev_op = circ.get_Op_ptr_from_Vertex(prev_v);
220  // if previous vertex is single qubit gate
221  if (prev_op->get_desc().is_gate() &&
222  circ.n_in_edges_of_type(prev_v, EdgeType::Quantum) == 1) {
223  const std::optional<Pauli> prev_colour =
224  prev_op->commuting_basis(ports.second);
225 
226  if (curr_op->commutes_with_basis(prev_colour, ports.first)) {
227  // subsequent op on qubit path is a single qubit gate
228  // and commutes with current multi qubit gate
229  success = true;
230  circ.remove_vertex(
231  prev_v, Circuit::GraphRewiring::Yes,
232  Circuit::VertexDeletion::No);
233  Edge rewire_edge = circ.get_nth_in_edge(current_v, ports.first);
234  circ.rewire(prev_v, {rewire_edge}, {EdgeType::Quantum});
235  current_e = circ.get_nth_out_edge(current_v, ports.first);
236  prev_v = circ.target(current_e);
237  // check if new previous gate can be commuted through too
238  goto check_prev_commutes;
239  }
240  }
241  }
242  // move to next vertex (towards input)
243  prev_v = current_v;
244  std::tie(current_v, current_e) = circ.get_prev_pair(current_v, current_e);
245  }
246  }
247 
248  return success;
249 }

251 // Any PhasedX or NPhasedX gate is replaced by an NPhasedX that is global
252 static bool replace_non_global_phasedx(Circuit &circ) {
253  bool success = false;
254  std::vector<unsigned> range_qbs(circ.n_qubits());
255  std::iota(range_qbs.begin(), range_qbs.end(), 0);
256  std::vector<Edge> frontier;
257  frontier.reserve(range_qbs.size());
258  for (const Qubit &q : circ.all_qubits()) {
259  Vertex v_in = circ.get_in(q);
260  EdgeVec all_out_e = circ.get_all_out_edges(v_in);
261  TKET_ASSERT(all_out_e.size() == 1);
262  frontier.push_back(all_out_e[0]);
263  }
264 
265  for (Vertex v : circ.vertices_in_order()) {
266  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
267  OpType optype = op->get_type();
268 
269  Subcircuit hole;
270  hole.verts = {v};
271  hole.q_in_hole = frontier;
272  // move frontier forward
273  if (!is_final_q_type(optype)) {
274  for (Edge e : circ.get_in_edges_of_type(v, EdgeType::Quantum)) {
275  auto it = std::find(frontier.begin(), frontier.end(), e);
276  TKET_ASSERT(it != frontier.end());
277  *it = circ.get_next_edge(v, e);
278  }
279  }
280  hole.q_out_hole = frontier;
281 
282  if (optype == OpType::PhasedX || optype == OpType::NPhasedX) {
283  if (op->n_qubits() < range_qbs.size()) {
284  // global substitution for v
285  const std::vector<Expr> params = op->get_params();
286  TKET_ASSERT(params.size() == 2);
287  Expr a = params[0], b = params[1];
288  Circuit sub(range_qbs.size());
289  sub.add_op<unsigned>(OpType::NPhasedX, {-0.5, b + 0.5}, range_qbs);
290  for (unsigned i = 0; i < range_qbs.size(); ++i) {
291  if (circ.source(frontier[i]) == v) {
292  sub.add_op<unsigned>(OpType::Rz, {a}, {i});
293  }
294  }
295  sub.add_op<unsigned>(OpType::NPhasedX, {0.5, b + 0.5}, range_qbs);
296 
297  // backup frontier
298  std::vector<Vertex> frontier_v;
299  std::vector<port_t> frontier_p;
300  for (const Edge &e : frontier) {
301  frontier_v.push_back(circ.target(e));
302  frontier_p.push_back(circ.get_target_port(e));
303  }
304  // perform substitution
305  circ.substitute(sub, hole, Circuit::VertexDeletion::Yes);
306  // restore frontier
307  for (unsigned i = 0; i < frontier.size(); ++i) {
308  frontier[i] = circ.get_nth_in_edge(frontier_v[i], frontier_p[i]);
309  }
310 
311  success = true;
312  }
313  }
314  }
315  return success;
316 }

318 static bool replace_two_qubit_interaction(
319  Circuit &circ, Transform::Interaction &i,
320  std::map<Qubit, Edge> &current_edges, VertexList &bin,
321  const double cx_fidelity = 1.) {
322  EdgeVec in_edges = {i.e0, i.e1};
323  EdgeVec out_edges = {current_edges[i.q0], current_edges[i.q1]};
324  Edge next0, next1;
325  bool q0_is_out = is_final_q_type(
326  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q0])));
327  bool q1_is_out = is_final_q_type(
328  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q1])));
329  if (!q0_is_out) {
330  next0 = circ.get_next_edge(
331  circ.target(current_edges[i.q0]), current_edges[i.q0]);
332  }
333  if (!q1_is_out) {
334  next1 = circ.get_next_edge(
335  circ.target(current_edges[i.q1]), current_edges[i.q1]);
336  }
337  Subcircuit sub = {in_edges, out_edges, i.vertices};
338  Circuit subc = circ.subcircuit(sub);
339  Eigen::Matrix4cd mat = get_matrix_from_2qb_circ(subc);
340  Circuit replacement = two_qubit_canonical(mat, cx_fidelity);
341  const int nb_cx_old = subc.count_gates(OpType::CX);
342  const int nb_cx_new = replacement.count_gates(OpType::CX);
343  if (nb_cx_new < nb_cx_old) {
344  bin.insert(bin.end(), sub.verts.begin(), sub.verts.end());
345  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
346  if (!q0_is_out) {
347  current_edges[i.q0] = circ.get_last_edge(circ.source(next0), next0);
348  }
349  if (!q1_is_out) {
350  current_edges[i.q1] = circ.get_last_edge(circ.source(next1), next1);
351  }
352  return true;
353  } else {
354  return false;
355  }
356 }

358 Transform Transform::commute_and_combine_HQS2() {
359  return Transform([](Circuit &circ) {
360  bool success = false;
361  VertexList bin;
362  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
363  EdgeVec outs = circ.get_all_out_edges(v);
364  if (circ.get_OpType_from_Vertex(v) == OpType::ZZMax && outs.size() == 2) {
365  Vertex next0 = boost::target(outs[0], circ.dag);
366  Vertex next1 = boost::target(outs[1], circ.dag);
367  if (next0 == next1 &&
368  circ.get_OpType_from_Vertex(next0) == OpType::ZZMax) {
369  success = true;
370  EdgeVec h_in = circ.get_in_edges(v);
371  EdgeVec h_out = circ.get_all_out_edges(next0);
372  if (circ.get_target_port(outs[0]) != 0) {
373  h_out = {h_out[1], h_out[0]};
374  }
375  bin.push_back(v);
376  bin.push_back(next0);
377  Subcircuit sub = {h_in, h_out};
378  circ.substitute(
379  CircPool::two_Rz1(), sub, Circuit::VertexDeletion::No);
380  circ.add_phase(0.5);
381  continue;
382  }
383  if (circ.get_OpType_from_Vertex(next0) == OpType::Rz) {
384  success = true;
385  circ.remove_vertex(
386  next0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
387  Edge in_0 = circ.get_nth_in_edge(v, 0);
388  circ.rewire(next0, {in_0}, {EdgeType::Quantum});
389  }
390  if (circ.get_OpType_from_Vertex(next1) == OpType::Rz) {
391  success = true;
392  circ.remove_vertex(
393  next1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
394  Edge in_1 = circ.get_nth_in_edge(v, 1);
395  circ.rewire(next1, {in_1}, {EdgeType::Quantum});
396  }
397  }
398  }
399  circ.remove_vertices(
400  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
401  return success;
402  });
403 }

406 Transform Transform::two_qubit_squash(double cx_fidelity) {
407  return Transform([cx_fidelity](Circuit &circ) {
408  bool success = false;
409  VertexList bin;
410  // Get map from vertex/port to qubit number
411  std::map<VertPort, Qubit> v_to_qb;
412  std::map<Qubit, Edge> current_edge_on_qb;
413  std::vector<Interaction> i_vec;
414  std::map<Qubit, int> current_interaction;
415  for (const Qubit &qb : circ.all_qubits()) {
416  for (const VertPort &vp : circ.unit_path(qb)) {
417  v_to_qb.insert({vp, qb});
418  }
419  Vertex input = circ.get_in(qb);
420  Edge e = circ.get_nth_out_edge(input, 0);
421  current_edge_on_qb[qb] = e;
422  current_interaction[qb] = -1;
423  }
424  SliceVec slices = circ.get_slices();
425  slices.insert(slices.begin(), circ.q_inputs());
426  slices.push_back(circ.q_outputs());
427  for (SliceVec::iterator s = slices.begin(); s != slices.end(); ++s) {
428  for (Slice::iterator v = s->begin(); v != s->end(); ++v) {
429  const Op_ptr o = circ.get_Op_ptr_from_Vertex(*v);
430  OpType type = o->get_type();
431  unsigned n_ins = circ.n_in_edges_of_type(*v, EdgeType::Quantum);
432  // Measures, resets, outputs, barriers, symbolic gates, and many-qubit
433  // gates close interactions
434  if (is_projective_type(type) || is_final_q_type(type) ||
435  type == OpType::Barrier || n_ins > 2 ||
436  !o->free_symbols().empty()) {
437  for (port_t port = 0; port < n_ins; port++) {
438  Qubit q = v_to_qb.at({*v, port});
439  int i = current_interaction[q];
440  if (i != -1) {
441  if (i_vec[i].count >= 2) {
442  // Replace subcircuit
443  success |= replace_two_qubit_interaction(
444  circ, i_vec[i], current_edge_on_qb, bin, cx_fidelity);
445  }
446  current_interaction[i_vec[i].q0] = -1;
447  current_interaction[i_vec[i].q1] = -1;
448  }
449  if (!is_final_q_type(type)) {
450  current_edge_on_qb[q] =
451  circ.get_next_edge(*v, current_edge_on_qb[q]);
452  }
453  }
454  continue;
455  }
456 
457  // Check for 2qb gate
458  if (circ.n_in_edges_of_type(*v, EdgeType::Quantum) == 2) {
459  Qubit q0 = v_to_qb.at({*v, 0});
460  Qubit q1 = v_to_qb.at({*v, 1});
461  int i0 = current_interaction[q0];
462  int i1 = current_interaction[q1];
463  // If they are already interacting, extend it
464  if (i0 != -1 && i0 == i1) {
465  i_vec[i0].count++;
466  i_vec[i0].vertices.insert(*v);
467  current_edge_on_qb[q0] =
468  circ.get_next_edge(*v, current_edge_on_qb[q0]);
469  current_edge_on_qb[q1] =
470  circ.get_next_edge(*v, current_edge_on_qb[q1]);
471  continue;
472  }
473  // End any other interactions on q0
474  if (i0 != -1) {
475  if (i_vec[i0].count >= 2) {
476  // Replace subcircuit
477  success |= replace_two_qubit_interaction(
478  circ, i_vec[i0], current_edge_on_qb, bin, cx_fidelity);
479  }
480  current_interaction[i_vec[i0].q0] = -1;
481  current_interaction[i_vec[i0].q1] = -1;
482  }
483  // End any other interactions on q1
484  if (i1 != -1) {
485  if (i_vec[i1].count >= 2) {
486  // Replace subcircuit
487 
488  success |= replace_two_qubit_interaction(
489  circ, i_vec[i1], current_edge_on_qb, bin, cx_fidelity);
490  }
491  current_interaction[i_vec[i1].q0] = -1;
492  current_interaction[i_vec[i1].q1] = -1;
493  }
494  // Add new interaction
495  Interaction new_i(q0, q1);
496  new_i.e0 = current_edge_on_qb[q0];
497  new_i.e1 = current_edge_on_qb[q1];
498  new_i.count = 1;
499  new_i.vertices = {*v};
500  current_interaction[q0] = i_vec.size();
501  current_interaction[q1] = i_vec.size();
502  i_vec.push_back(new_i);
503  current_edge_on_qb[q0] =
504  circ.get_next_edge(*v, current_edge_on_qb[q0]);
505  current_edge_on_qb[q1] =
506  circ.get_next_edge(*v, current_edge_on_qb[q1]);
507  continue;
508  }
509 
510  // Otherwise, we don't care about other vertices, so just update edges
511  // and add vertices if interactions exist
512  for (port_t i = 0; i < circ.n_in_edges(*v); i++) {
513  Qubit q = v_to_qb.at({*v, i});
514  current_edge_on_qb[q] = circ.get_next_edge(*v, current_edge_on_qb[q]);
515  int inter = current_interaction[q];
516  if (inter != -1) {
517  i_vec[inter].vertices.insert(*v);
518  }
519  }
520  }
521  }
522  circ.remove_vertices(
523  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
524  return success;
525  });
526 }

540 class Squasher {
541  public:
542  Squasher(Circuit &circ, OpType p, OpType q, bool smart_squash = true)
543  : circ_(circ),
544  p_(p),
545  q_(q),
546  success(false),
547  bin(),
548  smart_squash_(smart_squash) {
549  if (!(p == OpType::Rx || p == OpType::Ry || p == OpType::Rz) ||

591  static bool fixup_angles(Expr &angle_p1, Expr &angle_q, Expr &angle_p2) {
592  if (equiv_val(angle_q, 1., 2) && !equiv_0(angle_p2, 4)) {
593  // Prefer --P(p1-p2)--Q(...)--P(0)--
594  // Only occurs if angle_q is pi or 3pi and angle_p2 is non-zero
595  angle_p1 -= angle_p2;
596  angle_p2 = 0;
597  return true;
598  } else if (equiv_val(angle_p2, 1., 4)) {
599  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
600  // Only occurs if angle_p2 is pi
601  angle_p1 += 1;
602  angle_q *= -1;
603  angle_p2 = 0;
604  return true;
605  } else if (equiv_val(angle_p2, 3., 4)) {
606  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
607  // Only occurs if angle_p2 is 3pi
608  angle_p1 += 3;
609  angle_q *= -1;
610  angle_p2 = 0;
611  return true;
612  } else if (equiv_val(angle_p1, 1., 4) && !equiv_0(angle_p2, 4)) {
613  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
614  // Only occurs if angle_p1 is pi and angle_p2 is non-zero
615  angle_q *= -1;
616  angle_p2 += 1;
617  angle_p1 = 0;
618  return true;
619  } else if (equiv_val(angle_p1, 3., 4) && !equiv_0(angle_p2, 4)) {
620  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
621  // Only occurs if angle_p1 is 3pi and angle_p2 is non-zero
622  angle_q *= -1;
623  angle_p2 += 3;
624  angle_p1 = 0;
625  return true;
626  } else
627  return false;
628  }

649  std::tuple<Expr, Expr, Expr> pqp_from_chain(
650  const VertexList &rotation_chain, bool invert_pqp = false) const {
651  OpType p = (invert_pqp) ? q_ : p_;
652  OpType q = (invert_pqp) ? p_ : q_;
653 
654  // Construct list of merged rotations
655  std::list<Rotation> rots;
656  VertexList::const_iterator iter = rotation_chain.begin();
657  while (iter != rotation_chain.end()) {
658  // Merge next q rotations
659  rots.push_back(merge_rotations(q, rotation_chain, iter));
660  // Merge next p rotations
661  rots.push_back(merge_rotations(p, rotation_chain, iter));
662  }
663 
664  // Perform any cancellations
665  std::list<Rotation>::iterator r = rots.begin();
666  while (r != rots.end()) {
667  if (r->is_id()) {
668  r = rots.erase(r);
669  if (r != rots.begin() && r != rots.end()) {
670  std::prev(r)->apply(*r);
671  r = rots.erase(r);
672  r--;
673  }
674  } else
675  r++;
676  }
677 
678  // Extract any P rotations from the beginning and end of the list
679  Expr p1 = 0, p2 = 0;
680  std::list<Rotation>::iterator i1 = rots.begin();
681  if (i1 != rots.end()) {
682  std::optional<Expr> a = i1->angle(p);
683  if (a) {
684  p1 = a.value();
685  rots.pop_front();
686  }
687  }
688  std::list<Rotation>::reverse_iterator i2 = rots.rbegin();
689  if (i2 != rots.rend()) {
690  std::optional<Expr> a = i2->angle(p);
691  if (a) {
692  p2 = a.value();
693  rots.pop_back();
694  }
695  }
696 
697  // Finish up:
698  Rotation R = {};
699  for (auto rot : rots) {
700  R.apply(rot);
701  }
702  std::tuple<Expr, Expr, Expr> pqp = R.to_pqp(p, q);
703  std::get<0>(pqp) += p1;
704  std::get<2>(pqp) += p2;
705  return pqp;
706  }

708  void squash_rotations(const VertexList &rotation_chain) {
709  // TODO:: break chain up with classical control
710 
711  // smart squashing: choose pqp (default) or qpq depending on next gate
712  bool choose_qpq = false;
713  // smart squashing: flag if last rotation can be commuted through next gate
714  bool commute_through = false;
715 
716  const Op_ptr next_op = circ_.get_Op_ptr_from_Vertex(v);
717  const OpType next_op_type = next_op->get_type();
718  if (smart_squash_ && is_gate_type(next_op_type)) {
719  const port_t source_port = circ_.get_source_port(e);
720  const std::optional<Pauli> commutation_colour =
721  next_op->commuting_basis(source_port);
722 
723  Gate P(p_, {0}, 1);
724  Gate Q(q_, {0}, 1);
725  if (P.commutes_with_basis(commutation_colour, 0)) {
726  commute_through = true;
727  } else if (Q.commutes_with_basis(commutation_colour, 0)) {
728  choose_qpq = true;
729  commute_through = true;
730  }
731  }
732 
733  // swap p, q if required
734  OpType p = choose_qpq ? q_ : p_;
735  OpType q = choose_qpq ? p_ : q_;
736 
737  auto angles = pqp_from_chain(rotation_chain, choose_qpq);
738 
739  Expr angle_p1 = std::get<0>(angles);
740  Expr angle_q = std::get<1>(angles);
741  Expr angle_p2 = std::get<2>(angles);
742  (void)fixup_angles(angle_p1, angle_q, angle_p2);
743 
744  Circuit replacement(1);
745  if (!commute_through) {
746  replacement.add_op<unsigned>(p, angle_p1, {0});
747  }
748  replacement.add_op<unsigned>(q, angle_q, {0});
749  replacement.add_op<unsigned>(p, angle_p2, {0});
750  redundancy_removal(replacement);
751 
752  // check if replacement is any different from original chain
753  if (!is_same_chain(replacement, rotation_chain)) {
754  success = true;
755 
756  // replace with new rotations in circuit
757  Subcircuit sub = {
758  {e}, {circ_.get_nth_out_edge(rotation_chain.back(), 0)}};
759  port_t port = circ_.get_source_port(e);
760  circ_.substitute(replacement, sub, Circuit::VertexDeletion::No);
761  e = circ_.get_nth_out_edge(v, port);
762  bin.insert(bin.end(), rotation_chain.begin(), rotation_chain.end());
763  // add gate commuted through v
764  if (commute_through) {
765  Edge last_e = circ_.get_last_edge(v, e);
766  Subcircuit before_v{{last_e}, {last_e}};
767  Circuit leftover_p_gate(1);
768  leftover_p_gate.add_op<unsigned>(p, angle_p1, {0});
769  circ_.substitute(
770  leftover_p_gate, before_v, Circuit::VertexDeletion::No);
771  }
772  }
773  }

775  void squash_wire(VertexVec::const_iterator i) {
776  e = circ_.get_nth_in_edge(*i, 0);
777  v = circ_.source(e);
778  VertexList rotation_chain;
779  while (true) {
780  OpType v_type = circ_.get_OpType_from_Vertex(v);
781  if (v_type == p_ || v_type == q_) {
782  rotation_chain.push_front(v);
783  } else if (!rotation_chain.empty()) {
784  squash_rotations(rotation_chain);
785  rotation_chain.clear();
786  }
787  if (is_initial_q_type(v_type)) {
788  break;
789  }
790  e = circ_.get_last_edge(v, e);
791  v = circ_.source(e);
792  }
793  }

797  OpType q_;
798  bool success;
799  VertexList bin;

810 static bool standard_squash(
811  Circuit &circ, const OpTypeSet &singleqs,
812  const std::function<Circuit(const Expr &, const Expr &, const Expr &)>
813  &tk1_replacement) {
814  bool success = false;
815  for (OpType ot : singleqs) {
816  if (!is_single_qubit_type(ot))
817  throw NotValid(
818  "OpType given to standard_squash is not a single qubit "
819  "gate");
820  }
821  for (const Vertex in : circ.q_inputs()) {
822  Vertex v = in;
823  port_t port = 0;
824  VertexList single_chain;
825  Rotation combined;
826  std::optional<std::pair<std::list<VertPort>, unsigned>> condition;
827  while (true) {
828  Op_ptr v_op = circ.get_Op_ptr_from_Vertex(v);
829  OpType v_type = v_op->get_type();
830  bool squash_chain = false;
831  bool add_to_chain = false;
832  std::optional<std::pair<std::list<VertPort>, unsigned>> this_condition =
833  std::nullopt;
834  if (v_type == OpType::Conditional) {
835  const Conditional &cond_op = static_cast<const Conditional &>(*v_op);
836  EdgeVec ins = circ.get_in_edges(v);
837  this_condition = std::pair<std::list<VertPort>, unsigned>();
838  for (port_t p = 0; p < cond_op.get_width(); ++p) {
839  Edge in_p = ins.at(p);
840  VertPort vp = {circ.source(in_p), circ.get_source_port(in_p)};
841  this_condition->first.push_back(vp);
842  }
843  this_condition->second = cond_op.get_value();
844  v_op = cond_op.get_op();
845  v_type = v_op->get_type();
846  }
847  if (condition != this_condition) squash_chain = true;
848  unsigned q_ins = circ.n_in_edges_of_type(v, EdgeType::Quantum);
849  if ((q_ins != 1) | (singleqs.find(v_type) == singleqs.end()) |
850  is_projective_type(v_type))
851  squash_chain = true;
852  else
853  add_to_chain = true;
854  if (squash_chain && !single_chain.empty()) {
855  auto [a, b, c] = combined.to_pqp(OpType::Rz, OpType::Rx);
856  Circuit replacement = tk1_replacement(c, b, a);
857  if (replacement.n_gates() < single_chain.size()) {
858  BGL_FORALL_VERTICES(rv, replacement.dag, DAG) {
859  OpType v_type = replacement.get_OpType_from_Vertex(rv);
860  if (!is_boundary_q_type(v_type) &&
861  singleqs.find(v_type) == singleqs.end())
862  throw NotValid(
863  "tk1_replacement given to standard_squash "
864  "does not preserve gate set");
865  }
866  if (condition) {
867  circ.substitute_conditional(
868  replacement, single_chain.front(), Circuit::VertexDeletion::No);
869  } else {
870  circ.substitute(
871  replacement, single_chain.front(), Circuit::VertexDeletion::No);
872  }
873  circ.remove_vertices(
874  single_chain, Circuit::GraphRewiring::Yes,
875  Circuit::VertexDeletion::Yes);
876  success = true;
877  }
878  single_chain = VertexList();
879  combined = Rotation();
880  condition = std::nullopt;
881  }
882  if (is_final_q_type(v_type)) break;
883  if (add_to_chain) {
884  if (single_chain.empty()) condition = this_condition;
885  single_chain.push_back(v);
886  std::vector<Expr> angs = as_gate_ptr(v_op)->get_tk1_angles();
887  combined.apply(Rotation(OpType::Rz, angs.at(2)));
888  combined.apply(Rotation(OpType::Rx, angs.at(1)));
889  combined.apply(Rotation(OpType::Rz, angs.at(0)));
890  }
891  Edge e = circ.get_nth_out_edge(v, port);
892  v = circ.target(e);
893  port = circ.get_target_port(e);
894  }
895  }
896  return success;
897 }

908 // Given a 'SWAP_chain', finds edge in chain (or qubit wire) with best fidelity
909 // and rewires the associated single qubit vertex in to it
910 static bool find_edge_rewire_vertex(
911  Circuit &circ,
912  std::pair<std::vector<std::pair<Edge, double>>, Vertex> &entry) {
913  std::vector<std::pair<Edge, double>> candidates = entry.first;
914  std::pair<Edge, double> best_pair = candidates[0];

916  // (causes many issues...)
917  for (unsigned i = 1; i < candidates.size(); i++) { // find best edge
918  if (candidates[i].second > best_pair.second) {
919  best_pair = candidates[i];
920  do_rewire = true;
921  }
922  }
923  if (do_rewire) {
924  // rewire in to best edge
925  circ.remove_vertex(
926  entry.second, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
927  circ.rewire(entry.second, {best_pair.first}, {EdgeType::Quantum});
928  }

932 // Given a SWAP vertex, which has some predecessor SWAP vertex, find the
933 // 'SWAP_chain' this predecessor SWAP vertex is in and add to it
934 static void extend_SWAP_chain(
935  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
936  &swap_chains,
937  Edge entry_edge, Node entry_node, const Edge &match, const Circuit &circ,
938  const DeviceCharacterisation &characterisation) {
939  for (auto it = swap_chains.begin(); it != swap_chains.end(); ++it) {
940  if ((*it).first.back().first == match) {
941  // extend chain, adding a new pair of edge and double to the end
942  (*it).first.push_back(
943  {entry_edge,
944  1.0 - characterisation.get_error(
945  entry_node, circ.get_OpType_from_Vertex((*it).second))});
946  return;
947  }

951 // Finds sequences of adjacent SWAP gates, with a predecessor Single Qubit
952 // Vertex. The error rate of the required Single Qubit Vertex is stored for each
953 // of the PhysicalQubits the LogicalQubit passes through Once 'SWAP_chains' are
954 // found throughout the whole circuit, predecessor Single Qubit Vertices are
955 // rewired into the edge with best error rate
956 static bool find_rewire_sq(
957  Circuit &circ, const DeviceCharacterisation &characterisation) {
958  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
959  swap_chains;
960  for (auto it = circ.begin(); it != circ.end(); ++it) {
961  if (it->get_op_ptr()->get_type() == OpType::SWAP) {
962  // find SWAP, if either predecessor is a single qubit unitary
963  // find resulting swap chain...
964  Vertex swap_vert = it.get_vertex();
965  unit_vector_t qubits = it->get_args();
966  node_vector_t nodes = {qubits.begin(), qubits.end()};
967  VertexVec pred_verts = circ.get_predecessors(swap_vert);
968  EdgeVec pred_edges = circ.get_in_edges(swap_vert);
969  EdgeVec post_edges = circ.get_all_out_edges(swap_vert);
970  for (unsigned i = 0; i < pred_verts.size(); i++) {
971  OpType optype = circ.get_OpType_from_Vertex(
972  pred_verts[i]); // OPT IS PREDECESSOR OPTYPE
973  if (circ.detect_singleq_unitary_op(pred_verts[i])) {
974  // wire has single qubit unitary->add a new swap chain
975  std::vector<std::pair<Edge, double>> swap_chain = {
976  {pred_edges[i],
977  1.0 - characterisation.get_error(nodes[i], optype)},
978  {post_edges[1 - i],
979  1.0 - characterisation.get_error(nodes[1 - i], optype)}};
980  swap_chains.push_back(std::make_pair(swap_chain, pred_verts[i]));
981  } else if (optype == OpType::SWAP) { // wire has swap->assume this swap
982  // already in chain, find chain
983  Edge pre_edge = pred_edges[i];
984  extend_SWAP_chain(
985  swap_chains, post_edges[1 - i], nodes[1 - i], pre_edge, circ,
986  characterisation);
987  }
988  }
989  }
990  }
991  // having produced swap chains, now find best qubit for gate to act on and
992  // implement
993  bool success = false;
994  while (!swap_chains.empty()) {
995  if (find_edge_rewire_vertex(circ, (*swap_chains.begin())) == true) {
996  success = true;
997  }
998  swap_chains.erase(swap_chains.begin());
999  }
1000  return success;
1001 }
