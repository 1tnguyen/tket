32 const CanonicalRelabelling::Result& CanonicalRelabelling::operator()(
33  const VertexMapping& desired_mapping) {
34  m_result.too_many_vertices = false;
35  m_result.permutation_hash = 0;
36  m_result.new_to_old_vertices.clear();
37  m_result.old_to_new_vertices.clear();
38  m_result.identity = all_tokens_home(desired_mapping);
39  if (m_result.identity) {
40  return m_result;
41  }
42  check_mapping(desired_mapping, m_work_mapping);
43  if (desired_mapping.size() > 6) {
44  m_result.too_many_vertices = true;
45  return m_result;
46  }
47  // If not the identity, at least 2 vertices moved.
48  TKET_ASSERT(desired_mapping.size() >= 2);
49  TKET_ASSERT(desired_mapping.size() <= 6);
50 
51  m_desired_mapping = desired_mapping;
52  unsigned next_cyc_index = 0;
53 
54  while (!m_desired_mapping.empty()) {
55  // New cycle starts
56  auto& this_cycle = m_cycles[next_cyc_index];
57  ++next_cyc_index;
58  this_cycle.clear();
59  this_cycle.push_back(m_desired_mapping.cbegin()->first);
60  bool terminated_correctly = false;
61  for (unsigned infinite_loop_guard = 1 + m_desired_mapping.size();
62  infinite_loop_guard != 0; --infinite_loop_guard) {
63  const auto curr_v = this_cycle.back();
64  const auto target_v = m_desired_mapping.at(curr_v);
65  TKET_ASSERT(m_desired_mapping.erase(curr_v) == 1);
66  if (target_v == this_cycle[0]) {
67  terminated_correctly = true;
68  break;
69  }
70  this_cycle.push_back(target_v);
71  }
72  TKET_ASSERT(terminated_correctly);
73  }
74  // Sort by cycle length, LONGEST cycles first.
75  // But, also want a "stable-like" sort:
76  // make a consistent choice across all platforms,
77  // if cycle lengths are equal,
78  // based only upon the vertex numbers.
79  m_sorted_cycles_indices.resize(next_cyc_index);
80  std::iota(m_sorted_cycles_indices.begin(), m_sorted_cycles_indices.end(), 0);
81  const auto& cycles = m_cycles;
82 
83  std::sort(
84  m_sorted_cycles_indices.begin(), m_sorted_cycles_indices.end(),
85  [cycles](unsigned ii, unsigned jj) {
86  const auto& cyc1 = cycles[ii];
87  const auto& cyc2 = cycles[jj];
88  return (cyc1.size() > cyc2.size()) ||
89  // Using the raw vertex numbers is, of course, non-canonical,
90  // but necessary if we are to have stable results
91  // across ALL nonstable sorting algorithms
92  // on different platforms/compilers.
93  ((cyc1.size() == cyc2.size()) && cyc1[0] < cyc2[0]);
94  });
95 
96  // Now we can set up the mapping.
97  m_result.new_to_old_vertices.clear();
98  for (auto ii : m_sorted_cycles_indices) {
99  const auto& cyc = m_cycles[ii];
100  TKET_ASSERT(!cyc.empty());
101  TKET_ASSERT(cyc.size() <= 6);
102  for (size_t old_v : cyc) {
103  m_result.new_to_old_vertices.push_back(old_v);
104  }
105  }
106  TKET_ASSERT(m_result.new_to_old_vertices.size() <= 6);
107  m_result.old_to_new_vertices.clear();
108  for (unsigned ii = 0; ii < m_result.new_to_old_vertices.size(); ++ii) {
109  m_result.old_to_new_vertices[m_result.new_to_old_vertices[ii]] = ii;
110  }
111  // GCOVR_EXCL_START
112  TKET_ASSERT(
113  m_result.new_to_old_vertices.size() ==
114  m_result.old_to_new_vertices.size());
115  // GCOVR_EXCL_STOP
116  // And finally, the permutation hash.
117  m_result.permutation_hash = 0;
118  for (auto ii : m_sorted_cycles_indices) {
119  const auto& cyc = m_cycles[ii];
120  if (cyc.size() == 1) {
121  break;
122  }
123  m_result.permutation_hash *= 10;
124  m_result.permutation_hash += cyc.size();
125  }
126  return m_result;
127 }
