551 Circuit CU3_using_CX(Expr theta, Expr phi, Expr lambda) {
552  Circuit c(2);
553  c.add_op<unsigned>(OpType::U1, (lambda + phi) / 2, {0});
554  c.add_op<unsigned>(OpType::U1, (lambda - phi) / 2, {1});
555  c.add_op<unsigned>(OpType::CX, {0, 1});
