29 size_t CyclesCandidateManager::fill_initial_cycle_ids(const Cycles& cycles) {
30  m_cycle_with_vertex_hash.clear();
31  m_cycles_to_keep.clear();
32  size_t cycle_length = 0;
33  for (auto id_opt = cycles.front_id(); id_opt;
34  id_opt = cycles.next(id_opt.value())) {
35  const auto& cycle = cycles.at(id_opt.value());
36  const auto& vertices = cycle.vertices;
37 
38  if (cycle_length == 0) {
39  cycle_length = vertices.size();
40  TKET_ASSERT(cycle_length >= 2);
41  } else {
42  TKET_ASSERT(cycle_length == vertices.size());
43  }
44  TKET_ASSERT(cycle.decrease > 0);
45 
46  // We want 50*(decrease)/(num swaps) >= min_candidate_power_percentage.
47  // (We multiply by 50 because a swap can change L by 2, not 1).
48  if (50 * static_cast<unsigned>(cycle.decrease) <
49  (m_options.min_candidate_power_percentage * cycle_length)) {
50  continue;
51  }
52 
53  CycleData cycle_data;
54  cycle_data.id = id_opt.value();
55  cycle_data.first_vertex_index = 0;
56  for (size_t ii = 1; ii < vertices.size(); ++ii) {
57  if (vertices[ii] < vertices[cycle_data.first_vertex_index]) {
58  cycle_data.first_vertex_index = ii;
59  }
60  }
61  size_t hash = static_cast<size_t>(cycle.decrease);
62  for (size_t ii = 0; ii < cycle_length; ++ii) {
63  boost::hash_combine(
64  hash, vertices[(ii + cycle_data.first_vertex_index) % cycle_length]);
65  }
66  const auto prev_cycle_citer = m_cycle_with_vertex_hash.find(hash);
67  if (prev_cycle_citer == m_cycle_with_vertex_hash.cend()) {
68  m_cycle_with_vertex_hash[hash] = cycle_data;
69  } else {
70  // A previous cycle with this hash; but is it equal?
71  const auto& previous_cycle_data = prev_cycle_citer->second;
72  const auto& previous_cycle = cycles.at(previous_cycle_data.id);
73  if (previous_cycle.decrease == cycle.decrease) {
74  bool equal_vertices = true;
75  for (size_t ii = 0; ii < cycle_length; ++ii) {
76  if (previous_cycle.vertices
77  [(ii + previous_cycle_data.first_vertex_index) %
78  cycle_length] !=
79  cycle.vertices
80  [(ii + cycle_data.first_vertex_index) % cycle_length]) {
81  equal_vertices = false;
82  break;
83  }
84  }
85  if (equal_vertices) {
86  // This new cycle is just the previous cycle repeated,
87  // but starting from a different vertex
88  continue;
89  }
90  }
91  }
92  m_cycles_to_keep.push_back(cycle_data.id);
93  }
94  return cycle_length;
95 }

97 void CyclesCandidateManager::discard_lower_power_solutions(
98  const Cycles& cycles) {
99  int highest_decrease = 0;
100  for (auto id : m_cycles_to_keep) {
101  highest_decrease = std::max(highest_decrease, cycles.at(id).decrease);
102  }
103  TKET_ASSERT(highest_decrease > 0);
104 
105  for (size_t ii = 0; ii < m_cycles_to_keep.size();) {
106  if (cycles.at(m_cycles_to_keep[ii]).decrease < highest_decrease) {
107  // This cycle is not good enough.
108  // Erase this ID, by swapping with the back
109  m_cycles_to_keep[ii] = m_cycles_to_keep.back();
110  m_cycles_to_keep.pop_back();
111  continue;
112  }
113  // Keep this ID. Onto the next!
114  ++ii;
115  }
116 }

118 void CyclesCandidateManager::sort_candidates(const Cycles& cycles) {
119  // Greedy heuristic: we want the maximal number of disjoint cycles.
120  // So, choose those which touch few others first.
121  // Experimentation is needed with other algorithms!
122  m_touching_data.clear();
123  for (size_t ii = 0; ii < m_cycles_to_keep.size(); ++ii) {
124  // Automatically set to zero on first use.
125  m_touching_data[m_cycles_to_keep[ii]];
126 
127  for (size_t jj = ii + 1; jj < m_cycles_to_keep.size(); ++jj) {
128  bool touches = false;
129  // For short cycles, not much slower than using sets
130  // or sorted vectors.
131  for (auto v1 : cycles.at(m_cycles_to_keep[ii]).vertices) {
132  if (touches) {
133  break;
134  }
135  for (auto v2 : cycles.at(m_cycles_to_keep[jj]).vertices) {
136  if (v1 == v2) {
137  touches = true;
138  break;
139  }
140  }
141  }
142  if (touches) {
143  ++m_touching_data[m_cycles_to_keep[ii]];
144  ++m_touching_data[m_cycles_to_keep[jj]];
145  }
146  }
147  }
148  // Now, sort...
149  auto& touching_data = m_touching_data;
150  std::sort(
151  m_cycles_to_keep.begin(), m_cycles_to_keep.end(),
152  [&touching_data](Cycles::ID lhs, Cycles::ID rhs) {
153  const auto lhs_touch_number = touching_data.at(lhs);
154  const auto rhs_touch_number = touching_data.at(rhs);
155 
156  // Don't JUST sort on the touch number, because then the order
157  // of equal-touch-number elements would be implementation dependent
158  // (i.e., not a "stable" sort across all platforms/compilers).
159  return (lhs_touch_number < rhs_touch_number) ||
160  (lhs_touch_number == rhs_touch_number && lhs < rhs);
161  });
162 }

164 bool CyclesCandidateManager::should_add_swaps_for_candidate(
165  const Cycles& cycles, Cycles::ID id) {
166  const auto& cycle = cycles.at(id);
167  const auto& vertices = cycle.vertices;
168  for (auto v : vertices) {
169  if (m_vertices_used.count(v) != 0) {
170  return false;
171  }
172  }
173  for (auto v : vertices) {
174  m_vertices_used.insert(v);
175  }
176  return true;

179 void CyclesCandidateManager::append_partial_solution(
180  const CyclesGrowthManager& growth_manager, SwapList& swaps,
181  VertexMapping& vertex_mapping) {
182  const auto& cycles = growth_manager.get_cycles();
183  const size_t cycle_size = fill_initial_cycle_ids(cycles);
184 
185  if (m_cycles_to_keep.empty()) {
186  return;
187  }
188  const bool keep_lower_power_solutions =
189  (cycle_size == 2)
190  ? m_options.return_all_good_single_swaps
191  : m_options.return_lower_power_solutions_for_multiswap_candidates;
192 
193  if (!keep_lower_power_solutions) {
194  discard_lower_power_solutions(cycles);
195  }
196  sort_candidates(cycles);
197  m_vertices_used.clear();
198 
199  // It's the final function, so don't bother erasing
200  // elements in m_cycles_to_keep.
201  for (auto id : m_cycles_to_keep) {
202  if (!should_add_swaps_for_candidate(cycles, id)) {
203  continue;
204  }
205  const auto& vertices = cycles.at(id).vertices;
206  for (size_t ii = vertices.size() - 1; ii > 0; --ii) {
207  VertexSwapResult(vertices[ii], vertices[ii - 1], vertex_mapping, swaps);
208  }
209  }
210 }
