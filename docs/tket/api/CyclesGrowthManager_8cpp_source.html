49 bool CyclesGrowthManager::reset(
50  const VertexMapping& vertex_mapping, DistancesInterface& distances,
51  NeighboursInterface& neighbours) {
52  m_cycles.clear();
53  m_cycles_are_candidates = false;
54 
55  // OK, a bit inefficient, every really good swap (decreasing L by 2)
56  // will appear twice, but not a disaster.
57  // If no such swap exists, then this stored data will be necessary, because
58  // direction matters in longer cycles: v0->v1->v2->v0 is very different
59  // from v2->v1->v0->v2.
60  // It's simplest just to treat swaps as special cases of cycles, on 2
61  // vertices.
62  for (const auto& entry : vertex_mapping) {
63  const auto source = entry.first;
64  const auto target = entry.second;
65  const auto source_distance_to_target = distances(source, target);
66  if (source_distance_to_target == 0) {
67  continue;
68  }
69  const auto& adj_vertices = neighbours(source);
70  for (auto adj_v : adj_vertices) {
71  const auto other_v_distance_to_target = distances(adj_v, target);
72  if (other_v_distance_to_target < source_distance_to_target) {
73  const auto new_id = m_cycles.emplace_back();
74  auto& cycle = m_cycles.at(new_id);
75  cycle.decrease = 1;
76  cycle.vertices.resize(2);
77  cycle.vertices[0] = source;
78  cycle.vertices[1] = adj_v;
79  if (m_cycles.size() >= m_options.max_number_of_cycles) {
80  return true;
81  }
82  }
83  }
84  }
85  return !m_cycles.empty();
86 }

88 bool CyclesGrowthManager::attempt_to_close_cycles(
89  const VertexMapping& vertex_mapping, DistancesInterface& distances) {
90  TKET_ASSERT(!m_cycles_are_candidates);
91  for (auto id_opt = m_cycles.front_id(); id_opt;) {
92  const auto id = id_opt.value();
93  id_opt = m_cycles.next(id);
94  auto& cycle = m_cycles.at(id);
95  const int decrease = get_move_decrease(
96  vertex_mapping, cycle.vertices.back(), cycle.vertices[0], distances);
97  const int new_decrease = cycle.decrease + decrease;
98  if (new_decrease > 0) {
99  cycle.decrease = new_decrease;
100  if (!m_cycles_are_candidates) {
101  // It's the first good one, so delete all previous.
102  for (auto prev_id_opt = m_cycles.previous(id); prev_id_opt;) {
103  const auto id_to_be_deleted = prev_id_opt.value();
104  prev_id_opt = m_cycles.previous(id_to_be_deleted);
105  m_cycles.erase(id_to_be_deleted);
106  }
107  }
108  m_cycles_are_candidates = true;
109  } else {
110  // Not a good closed cycle; do we delete it?
111  if (m_cycles_are_candidates) {
112  m_cycles.erase(id);
113  }
114  }
115  }
116  return m_cycles_are_candidates;
117 }

119 CyclesGrowthManager::GrowthResult CyclesGrowthManager::attempt_to_grow(
120  const VertexMapping& vertex_mapping, DistancesInterface& distances,
121  NeighboursInterface& neighbours) {
122  GrowthResult result;
123 
124  TKET_ASSERT(!m_cycles.empty());
125 
126  if (m_cycles.front().vertices.size() >= m_options.max_cycle_size) {
127  m_cycles.clear();
128  result.hit_cycle_length_limit = true;
129  result.empty = true;
130  return result;
131  }
132  for (auto id_opt = m_cycles.front_id(); id_opt;) {
133  const auto id = id_opt.value();
134  id_opt = m_cycles.next(id);
135 
136  // Add an arrow onto the back.
137  const auto back_vertex = m_cycles.at(id).vertices.back();
138  const auto& adj_vertices = neighbours(back_vertex);
139  for (auto adj_v : adj_vertices) {
140  int new_decr;
141  {
142  // Important not to reuse this once cycles are added,
143  // as it may be invalidated
144  const auto& cycle = m_cycles.at(id);
145  if (cycle.contains(adj_v)) {
146  continue;
147  }
148  new_decr =
149  cycle.decrease +
150  get_move_decrease(vertex_mapping, back_vertex, adj_v, distances);
151 
152  // If there are N moves, each move can only decrease L by at most one,
153  // so it's unfair to demand a huge L-decrease, because shorter cycles
154  // would be killed immediately.
155  // With N vertices there are N-1 moves, but we are about to add
156  // the new vertex adj_v to this partial cycle (unless we discard it),
157  // taking it back up to N.
158  const int num_moves = cycle.vertices.size();
159  int min_decrease = num_moves;
160  min_decrease =
161  std::min(min_decrease, m_options.min_decrease_for_partial_path);
162 
163  // We want 100*(L-decr)/(num.moves) >=
164  // min_power_percentage_for_partial_path. But we need the ceiling
165  // because of interger division.
166  min_decrease = std::max(
167  min_decrease,
168  (99 + m_options.min_power_percentage_for_partial_path * num_moves) /
169  100);
170 
171  if (new_decr < min_decrease) {
172  continue;
173  }
174  }
175  // A new cycle to be added. Add it before the current position,
176  // so we won't pass through it again in the main loop.
177  const auto new_id = m_cycles.insert_before(id);
178  auto& new_cycle = m_cycles.at(new_id);
179  new_cycle.decrease = new_decr;
180  new_cycle.vertices = m_cycles.at(id).vertices;
181  new_cycle.vertices.push_back(adj_v);
182  if (m_cycles.size() >= m_options.max_number_of_cycles) {
183  // Break out of the INNER loop, i.e. neighbours for this
184  // cycle endpoint. However, this cycle is about to be deleted,
185  // creating space, so continue with further cycles.
186  break;
187  }
188  }
189  m_cycles.erase(id);
190  }
191  result.empty = m_cycles.empty();
192  return result;
193 }
