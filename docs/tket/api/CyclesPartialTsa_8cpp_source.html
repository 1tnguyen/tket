26 void CyclesPartialTsa::append_partial_solution(
27  SwapList& swaps, VertexMapping& vertex_mapping,
28  DistancesInterface& distances, NeighboursInterface& neighbours,
29  RiverFlowPathFinder& path_finder) {
30  // We'll add the calculated swaps to the path finder at the end.
31  // THIS is the right place to do it, not the caller, because
32  // (as far as the caller knows) it's possible that PartialTSA objects
33  // reduce/reorder swaps, and so it would be invalid just to go back through
34  // the appended swaps. However, THIS class knows that no reordering or
35  // reduction occurs.
36  const size_t initial_swap_size = swaps.size();
37  for (;;) {
38  const auto swap_size_before = swaps.size();
39  single_iteration_partial_solution(
40  swaps, vertex_mapping, distances, neighbours);
41  const auto swap_size_after = swaps.size();
42  TKET_ASSERT(swap_size_after >= swap_size_before);
43  if (swap_size_before == swap_size_after) {
44  break;
45  }
46  }
47  const size_t final_swap_size = swaps.size();
48  TKET_ASSERT(initial_swap_size <= final_swap_size);
49  if (initial_swap_size == final_swap_size) {
50  return;
51  }
52  // At least one swap was added.
53  const auto current_back_id_opt = swaps.back_id();
54  TKET_ASSERT(current_back_id_opt);
55  auto current_id = current_back_id_opt.value();
56  for (size_t remaining_swaps = final_swap_size - initial_swap_size;;) {
57  const auto& swap = swaps.at(current_id);
58  path_finder.register_edge(swap.first, swap.second);
59  --remaining_swaps;
60  if (remaining_swaps == 0) {
61  break;
62  }
63  const auto prev_id_opt = swaps.previous(current_id);
64  TKET_ASSERT(prev_id_opt);
65  current_id = prev_id_opt.value();
66  }
67 }

69 void CyclesPartialTsa::single_iteration_partial_solution(
70  SwapList& swaps, VertexMapping& vertex_mapping,
71  DistancesInterface& distances, NeighboursInterface& neighbours) {
72  if (!m_growth_manager.reset(vertex_mapping, distances, neighbours)) {
73  // no solutions.
74  return;
75  }
76 
77  for (auto infinite_loop_guard = m_growth_manager.get_options().max_cycle_size;
78  infinite_loop_guard > 0; --infinite_loop_guard) {
79  if (m_growth_manager.attempt_to_close_cycles(vertex_mapping, distances)) {
80  // Some solutions found.
81  m_candidate_manager.append_partial_solution(
82  m_growth_manager, swaps, vertex_mapping);
83  return;
84  }
85  // No solutions so far, so grow...
86  const auto growth_result =
87  m_growth_manager.attempt_to_grow(vertex_mapping, distances, neighbours);
88  if (growth_result.empty || growth_result.hit_cycle_length_limit) {
89  return;
90  }
91  }
92  TKET_ASSERT(!"growth_manager termination");
93 }
