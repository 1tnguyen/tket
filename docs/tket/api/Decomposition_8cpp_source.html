37 static bool convert_multiqs_CX(Circuit &circ) {
38  bool success = false;
39  VertexList bin;
40  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
41  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
42  OpType optype = op->get_type();
43  if (is_gate_type(optype) && !is_projective_type(optype) &&
44  !is_single_qubit_type(optype) && (optype != OpType::CX)) {
45  Circuit in_circ = CX_circ_from_multiq(op);
46  Subcircuit sub = {
47  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};

50  success = true;

52  }
53  circ.remove_vertices(

65 static bool convert_to_zyz(Circuit &circ) {
66  static const Expr half = SymEngine::div(Expr(1), Expr(2));
67  bool success = Transform::decompose_single_qubits_TK1().apply(circ);
68  VertexList bin;
69  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
70  if (circ.n_in_edges(v) != 1) continue;
71  const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
72  if (op->get_type() == OpType::tk1) {
73  std::vector<Expr> params = op->get_params();
74  Circuit replacement(1);
75  Expr a = params[2] + half;
76  Expr b = params[1];
77  Expr c = params[0] - half;
78  if (!equiv_0(a, 4)) {
79  replacement.add_op<unsigned>(OpType::Rz, a, {0});
80  }
81  if (!equiv_0(b, 4)) {
82  replacement.add_op<unsigned>(OpType::Ry, b, {0});
83  }
84  if (!equiv_0(c, 4)) {
85  replacement.add_op<unsigned>(OpType::Rz, c, {0});
86  }
87  Subcircuit sub = {
88  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
89  bin.push_back(v);
90  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
91  success = true;
92  }
93  }
94  circ.remove_vertices(
95  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
96  return success;
97 }

158 Transform Transform::decompose_ZYZ_to_TK1() {
159  return Transform([](Circuit &circ) {
160  bool success = false;
161  static const Expr zero(0);
162  static const Expr half = SymEngine::div(Expr(1), Expr(2));
163  VertexList bin;
164  VertexVec inputs = circ.q_inputs();
165  for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
166  Edge e = circ.get_nth_out_edge(*i, 0);
167  Vertex v = circ.target(e);
168  while (!is_final_q_type(circ.get_OpType_from_Vertex(v))) {
169  if (circ.get_OpType_from_Vertex(v) == OpType::Rz) {
170  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
171  Expr angle_1 = v_g->get_params()[0];
172  Edge e1 = circ.get_next_edge(v, e);
173  Vertex v2 = circ.target(e1);
174  if (circ.get_OpType_from_Vertex(v2) == OpType::Ry) {
175  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
176  Expr angle_2 = v2_g->get_params()[0];
177  Edge e2 = circ.get_next_edge(v2, e1);
178  Vertex v3 = circ.target(e2);
179  bin.push_back(v2);
180  circ.remove_vertex(
181  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
182  Expr angle_3 = zero;
183  if (circ.get_OpType_from_Vertex(v3) == OpType::Rz) {
184  const Op_ptr v3_g = circ.get_Op_ptr_from_Vertex(v3);
185  angle_3 = v3_g->get_params()[0];
186  circ.remove_vertex(
187  v3, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
188  bin.push_back(v3);
189  }
190  std::vector<Expr> new_params = {
191  angle_3 + half, angle_2, angle_1 - half};
192  circ.dag[v] = {get_op_ptr(OpType::tk1, new_params)};
193  } else {
194  circ.dag[v] = {get_op_ptr(OpType::tk1, {zero, zero, angle_1})};
195  }
196  } else if (circ.get_OpType_from_Vertex(v) == OpType::Ry) {
197  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
198  Expr angle_2 = v_g->get_params()[0];
199  Expr angle_3 = zero;
200  Edge e1 = circ.get_next_edge(v, e);
201  Vertex v2 = circ.target(e1);
202  if (circ.get_OpType_from_Vertex(v2) == OpType::Rz) {
203  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
204  angle_3 = v2_g->get_params()[0];
205  circ.remove_vertex(
206  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
207  bin.push_back(v2);
208  }
209  std::vector<Expr> new_params = {angle_3 + half, angle_2, -half};
210  circ.dag[v] = {get_op_ptr(OpType::tk1, new_params)};
211  }
212  e = circ.get_next_edge(v, e);
213  v = circ.target(e);
214  }
215  }
216  circ.remove_vertices(
217  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
218  return success;
219  });
220 }

285 /* --Rz(a)--Rx(b)--R(c)-- => --Rz(a+c)--PhasedX(b,c)-- */
286 Transform Transform::decompose_ZX_to_HQS1() {
287  return Transform([](Circuit &circ) {
288  bool success = false;
289  VertexList to_bin;
290  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
291  if (circ.get_OpType_from_Vertex(v) == OpType::Rx) {
292  success = true;
293  const Op_ptr g = circ.get_Op_ptr_from_Vertex(v);
294  Expr theta = g->get_params()[0];
295  Vertex prev_vert = *circ.get_predecessors(v).begin();
296  Vertex next_vert = *circ.get_successors(v).begin();
297  if (circ.get_OpType_from_Vertex(prev_vert) == OpType::Rz &&
298  circ.get_OpType_from_Vertex(next_vert) == OpType::Rz) {
299  const Op_ptr prev_g = circ.get_Op_ptr_from_Vertex(prev_vert);
300  const Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next_vert);
301  Expr phi = next_g->get_params()[0];
302  std::vector<Expr> params{theta, phi};
303  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
304  circ.remove_vertex(
305  next_vert, Circuit::GraphRewiring::Yes,
306  Circuit::VertexDeletion::No);
307  to_bin.push_back(next_vert);
308  Expr new_param = prev_g->get_params()[0] + phi;
309  circ.dag[prev_vert].op = get_op_ptr(OpType::Rz, new_param);
310  } else {
311  // if no Rz, initialise a PhasedX op with theta=Rx.params[0],phi=0
312  Expr phi(0);
313  std::vector<Expr> params{theta, phi};
314  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
315  }
316  }
317  }
318  circ.remove_vertices(
319  to_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
320  Transform::remove_redundancies().apply(circ);
321  return success;
322  });
323 }

325 // Decompose CX into MolmerSorensen as:
326 // ---C--- -V-S-|-H-
327 // | --> XX(pi/4)
328 // ---X--- -----|-Vdg-
329 Transform Transform::decompose_MolmerSorensen() {
330  return Transform([](Circuit &circ) {
331  bool success = false;
332  VertexList bin;
333  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
334  if (circ.get_OpType_from_Vertex(v) != OpType::CX) continue;
335  EdgeVec outs = circ.get_all_out_edges(v);
336  if (outs.size() == 2) {
337  Vertex next = circ.target(outs[0]);
338  // Is the next operation equivalent to an Rx, up to phase?
339  Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next);
340  OpType next_type = next_g->get_type();
341  if (is_single_qubit_type(next_type) && !is_projective_type(next_type)) {
342  std::vector<Expr> angles = as_gate_ptr(next_g)->get_tk1_angles();
343  if (equiv_0(angles[0], 2) && equiv_0(angles[2], 2)) {
344  Expr angle = angles[1];
345  Expr phase = angles[3];
346  if (!equiv_0(angles[0], 4)) phase += 1;
347  if (!equiv_0(angles[2], 4)) phase += 1;
348  Edge next_e = circ.get_nth_out_edge(next, 0);
349  Vertex last = circ.target(next_e);
350  if (circ.get_OpType_from_Vertex(last) == OpType::CX &&
351  circ.get_nth_in_edge(last, 1) == outs[1]) {
352  // Recognise exp(-i XX * angle * pi/2)
353  const Op_ptr op_ptr = get_op_ptr(OpType::XXPhase, angle);
354  circ.dag[v] = {op_ptr};
355  bin.push_back(next);
356  circ.remove_vertex(
357  next, Circuit::GraphRewiring::Yes,
358  Circuit::VertexDeletion::No);
359  bin.push_back(last);
360  circ.remove_vertex(
361  last, Circuit::GraphRewiring::Yes,
362  Circuit::VertexDeletion::No);
363  circ.add_phase(phase);
364  success = true;
365  continue;
366  }
367  }
368  }
369  // Replace remaining CX gates
370  Subcircuit sub = {{circ.get_in_edges(v)}, {outs}, {v}};
371  bin.push_back(v);
372  circ.substitute(
373  CircPool::CX_using_XXPhase_1(), sub, Circuit::VertexDeletion::No);
374  success = true;
375  }
376  }
377  circ.remove_vertices(
378  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
379  return success;
380  });
381 }

529 static Circuit clifford_from_tk1(int i, int j, int k) {
530  std_cliff_spec_t spec = tk1_table[i % 4][j % 4][k % 4];
531  if (i >= 4) spec.p += 1;
532  if (j >= 4) spec.p += 1;
533  if (k >= 4) spec.p += 1;
534 
535  Circuit c(1);
536  if (spec.Z0) {
537  c.add_op<unsigned>(OpType::Z, {0});
538  }
539  if (spec.X1) {
540  c.add_op<unsigned>(OpType::X, {0});
541  }
542  if (spec.S2) {
543  c.add_op<unsigned>(OpType::S, {0});
544  }
545  if (spec.V3) {
546  c.add_op<unsigned>(OpType::V, {0});
547  }
548  if (spec.S4) {
549  c.add_op<unsigned>(OpType::S, {0});
550  }
551  c.add_phase(spec.p);
552 
553  return c;
554 }

556 Transform Transform::decompose_cliffords_std() {
557  return Transform([](Circuit &circ) {
558  bool success = false;
559  VertexList bin;
560  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
561  OpType opT = circ.get_OpType_from_Vertex(v);
562  if (opT == OpType::tk1 || opT == OpType::U3 || opT == OpType::U2 ||
563  opT == OpType::U1 || opT == OpType::Rx || opT == OpType::Ry ||
564  opT == OpType::Rz || opT == OpType::PhasedX) {
565  const Op_ptr g = circ.get_Op_ptr_from_Vertex(v);
566  std::vector<Expr> tk1_param_exprs = as_gate_ptr(g)->get_tk1_angles();
567  bool all_reduced = true;
568  bool all_roundable = true;
569  std::vector<int> iangles(3);
570  for (int i = 0; i < 3; i++) {
571  std::optional<double> reduced = eval_expr_mod(tk1_param_exprs[i], 4);
572  if (!reduced)
573  all_reduced = false;
574  else {
575  double angle = 2 * reduced.value(); // > 0
576  iangles[i] = int(angle + 0.5); // nearest integer
577  if (std::abs(angle - iangles[i]) >= EPS) {
578  all_roundable = false;
579  }
580  iangles[i] %= 8; // 8 --> 0
581  }
582  }
583  if (!(all_reduced && all_roundable)) continue;
584  Circuit replacement =
585  clifford_from_tk1(iangles[0], iangles[1], iangles[2]);
586  Subcircuit sub = {
587  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
588  bin.push_back(v);
589  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
590  circ.add_phase(tk1_param_exprs[3]);
591  success = true;
592  }
593  }
594  circ.remove_vertices(
595  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
596  return success;
597  });

600 Transform Transform::decompose_ZX_to_cliffords() {
601  return Transform([](Circuit &circ) {
602  bool success = false;
603  VertexList bin;
604  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
605  const Op_ptr op_ptr = circ.get_Op_ptr_from_Vertex(v);
606  if (op_ptr->get_type() == OpType::Rz ||
607  op_ptr->get_type() == OpType::Rx) {

609  std::optional<double> reduced = eval_expr_mod(param, 4);
610  bool roundable = false;
611  int iangle = 0;
612  if (reduced) {
613  double angle = 2 * reduced.value(); // >= 0
614  iangle = int(angle + 0.5); // nearest integer
615  iangle %= 8; // {0,1,2,3,4,5,6,7}
616  roundable = (std::abs(angle - iangle) < EPS);
617  }
618  if (roundable) {
619  bool is_rz = op_ptr->get_type() == OpType::Rz;
620  switch (iangle % 4) {
621  case 0: {
622  bin.push_back(v);
623  circ.remove_vertex(
624  v, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
625  break;
626  }
627  case 1: {
628  if (is_rz) {
629  circ.dag[v] = {get_op_ptr(OpType::S)};
630  circ.add_phase(-0.25);
631  } else {
632  circ.dag[v] = {get_op_ptr(OpType::V)};
633  }
634  break;
635  }
636  case 2: {
637  if (is_rz) {
638  circ.dag[v] = {get_op_ptr(OpType::Z)};
639  } else {
640  circ.dag[v] = {get_op_ptr(OpType::X)};
641  }
642  circ.add_phase(-0.5);
643  break;
644  }
645  case 3: {
646  if (is_rz) {
647  circ.dag[v] = {get_op_ptr(OpType::Sdg)};
648  circ.add_phase(-0.75);
649  } else {
650  circ.dag[v] = {get_op_ptr(OpType::Vdg)};
651  circ.add_phase(1);
652  }
653  break;
654  }
655  }
656  if (iangle >= 4) circ.add_phase(1);
657  success = true;
658  }
659  }
660  }
661  circ.remove_vertices(
662  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
663  return success;
664  });

667 Transform Transform::decompose_PhaseGadgets() {
668  return Transform([](Circuit &circ) {
669  bool success = false;
670  VertexList big_bin;
671  auto [it, end] = boost::vertices(circ.dag);
672  for (auto next = it; it != end; it = next) {
673  ++next;
674  if (circ.get_OpType_from_Vertex(*it) == OpType::CX &&
675  circ.n_out_edges(*it) == 2) {
676  EdgeVec outs = circ.get_all_out_edges(*it);
677  Vertex next_v = circ.target(outs[1]);
678  Op_ptr g = circ.get_Op_ptr_from_Vertex(next_v);
679  OpType type = g->get_type();
680  if (type == OpType::Rz || type == OpType::U1 ||
681  (type == OpType::tk1 && equiv_0(g->get_params()[1]))) {
682  Vertex last_v = circ.get_next_pair(next_v, outs[1]).first;
683  if (circ.get_OpType_from_Vertex(last_v) == OpType::CX &&
684  circ.get_nth_in_edge(last_v, 0) == outs[0]) {
685  VertexList bin = {next_v, last_v};
686  big_bin.push_back(next_v);
687  big_bin.push_back(last_v);
688  circ.remove_vertices(
689  bin, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
690  Expr t = g->get_params()[0];
691  if (type == OpType::tk1) {
692  t += g->get_params()[2];
693  }
694  circ.dag[*it] = {get_op_ptr(OpType::PhaseGadget, {t}, 2)};
695  if (type == OpType::U1) {
696  circ.add_phase(t / 2);
697  } else if (
698  type == OpType::tk1 && equiv_val(g->get_params()[1], 2, 4)) {
699  circ.add_phase(1);
700  }
701  success = true;
702  }
703  }
704  if (type == OpType::CX) {
705  if (circ.get_target_port(outs[1]) == 1) {
706  Vertex rx = circ.source(circ.get_nth_in_edge(next_v, 0));
707  if (circ.get_OpType_from_Vertex(rx) == OpType::Rx) {
708  if (rx == circ.target(outs[0])) {
709  const Op_ptr rx_g = (circ.get_Op_ptr_from_Vertex(rx));
710  VertexList bin = {rx, next_v};
711  big_bin.push_back(next_v);
712  big_bin.push_back(rx);
713  Circuit replacement(2);
714  circ.remove_vertices(
715  bin, Circuit::GraphRewiring::Yes,
716  Circuit::VertexDeletion::No);
717  replacement.add_op<unsigned>(OpType::H, {0});
718  replacement.add_op<unsigned>(OpType::H, {1});
719  replacement.add_op<unsigned>(
720  OpType::PhaseGadget, rx_g->get_params()[0], {0, 1});
721  replacement.add_op<unsigned>(OpType::H, {0});
722  replacement.add_op<unsigned>(OpType::H, {1});
723  Subcircuit sub = {
724  circ.get_in_edges(*it), circ.get_all_out_edges(*it), {*it}};
725  circ.substitute(replacement, sub, Circuit::VertexDeletion::Yes);
726  success = true;
727  }
728  }
729  }
730  }
731  }
732  }
733  circ.remove_vertices(
734  big_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
735  return success;
736  });
737 }

759 static void swap_sub(
760  Circuit &circ, const Circuit &swap_circ_1, const Circuit &swap_circ_2,
761  Subcircuit &sub, const std::pair<port_t, port_t> &port_comp) {
762  std::pair<port_t, port_t> comp = {0, 1};
763  // Ports only come in 2 cases, {0,1} or {1,0}. if {0,1} (first case),
764  // swap_circ_1 leaves a CX{0,1} next to current CX{0,1}, if not we can assume
765  // second case.
766  if (port_comp == comp)
767  circ.substitute(swap_circ_1, sub, Circuit::VertexDeletion::Yes);
768  else
769  circ.substitute(swap_circ_2, sub, Circuit::VertexDeletion::Yes);
770 }

772 Transform Transform::decompose_SWAP_to_CX(const Architecture &arc) {
773  // Note that the default argument will be out of scope at call-time!
774  // => we replace the default empty Architecture with nullptr
775  // we need to keep arc as a pointer as there is no such thing as
776  // optional references in std
777  const Architecture *arc_ptr = arc.n_uids() ? &arc : nullptr;
778  return Transform([arc_ptr](Circuit &circ) {
779  bool success = false;
780  std::vector<std::pair<Vertex, bool>> bin;
781  for (Circuit::CommandIterator it = circ.begin(); it != circ.end(); ++it) {
782  if (it->get_op_ptr()->get_type() == OpType::SWAP) {
783  unit_vector_t qbs = it->get_args();
784  node_vector_t nodes = {qbs.begin(), qbs.end()};
785  if (arc_ptr != nullptr && arc_ptr->uid_exists(nodes[0]) &&
786  arc_ptr->uid_exists(nodes[1]) &&
787  arc_ptr->connection_exists(nodes[1], nodes[0])) {
788  bin.push_back({it.get_vertex(), true});
789  } else {
790  bin.push_back({it.get_vertex(), false});
791  }
792  }
793  }
794 
795  for (std::pair<Vertex, bool> v : bin) {
796  success = true;
797  // Get predecessor vertices and successor vertices and find subcircuit for
798  // replacement
799  VertexVec preds = circ.get_predecessors(v.first);
800  VertexVec succs = circ.get_successors(v.first);
801  EdgeVec in_edges = circ.get_in_edges(v.first);
802  EdgeVec out_edges = circ.get_all_out_edges(v.first);
803  Subcircuit sub = {in_edges, out_edges, {v.first}};
804 
805  if (preds.size() == 1 &&
806  circ.get_OpType_from_Vertex(preds[0]) == OpType::CX) {
807  // if conditions requires that there is a CX gate before SWAP
808  swap_sub(
809  circ, CircPool::SWAP_using_CX_0(), CircPool::SWAP_using_CX_1(), sub,
810  {circ.get_source_port(in_edges[0]),
811  circ.get_source_port(in_edges[1])});
812  } else if (
813  succs.size() == 1 &&
814  circ.get_OpType_from_Vertex(succs[0]) == OpType::CX) {
815  // if no CX gate before SWAP, check after.
816  swap_sub(
817  circ, CircPool::SWAP_using_CX_0(), CircPool::SWAP_using_CX_1(), sub,
818  {circ.get_target_port(out_edges[0]),
819  circ.get_target_port(out_edges[1])});
820  } else if (v.second) {
821  // We assume in general that a CX gate saving is desired over H gate
822  // savings. If SWAP doesn't lend itself to annihlation though, the SWAP
823  // is inserted to reduce number of H gates added in a 'directed' CX
824  // decomposition.
825  // SWAP_using_CX_1 is added if the backwards direction is available on
826  // the architecture
827  circ.substitute(
828  CircPool::SWAP_using_CX_1(), sub, Circuit::VertexDeletion::Yes);
829  } else {
830  // SWAP_using_CX_0 is added as a default option
831  circ.substitute(
832  CircPool::SWAP_using_CX_0(), sub, Circuit::VertexDeletion::Yes);
833  }
834  }
835  return success;
836  });
837 }

839 Transform Transform::decompose_BRIDGE_to_CX() {
840  return Transform([](Circuit &circ) {
841  bool success = false;
842  // Collect all BRIDGE type vertices
843  std::vector<std::pair<Vertex, bool>> bin;
844  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
845  if (circ.get_OpType_from_Vertex(v) == OpType::BRIDGE) {
846  bin.push_back({v, false});
847  }
848  if (circ.get_OpType_from_Vertex(v) == OpType::Conditional) {
849  const Conditional &b =
850  static_cast<const Conditional &>(*circ.get_Op_ptr_from_Vertex(v));
851  if (b.get_op()->get_type() == OpType::BRIDGE) {
852  bin.push_back({v, true});
853  }
854  }
855  }
856 
857  auto BRIDGE_sub =
858  [&circ](std::pair<Vertex, bool> &candidate, Circuit BRIDGE_circ) {
859  if (candidate.second)
860  circ.substitute_conditional(
861  BRIDGE_circ, candidate.first, Circuit::VertexDeletion::Yes);
862  else
863  circ.substitute(
864  BRIDGE_circ, candidate.first, Circuit::VertexDeletion::Yes);
865  };
866 
867  for (std::pair<Vertex, bool> v : bin) {
868  // Get predecessor vertices and successor vertices and find subcircuit for
869  // replacement
870  success = true;
871  VertexVec preds = circ.get_predecessors(v.first);
872  VertexVec succs = circ.get_successors(v.first);
873  EdgeVec in_edges = circ.get_in_edges(v.first);
874  EdgeVec out_edges = circ.get_all_out_edges(v.first);
875  Subcircuit sub = {in_edges, out_edges, {v.first}};
876 
877  bool done = false;
878  if (preds.size() < 3) { // Implies some predecessor vertices are in a
879  // multi-qubit op together
880  VertexVec comps = {
881  circ.source(in_edges[0]), circ.source(in_edges[1]),
882  circ.source(in_edges[2])};
883  if (comps[0] ==
884  comps[1]) { // First two qubits in BRIDGE in multi-qubit op
885  BRIDGE_sub(v, CircPool::BRIDGE_using_CX_0());
886  done = true;
887  } else if (comps[2] == comps[1]) { // Second two qubits in BRIDGE in
888  // multi-qubit op together before
889  // BRIDGE
890  BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
891  done = true;
892  }
893  }
894  if (succs.size() < 3 && !done) { // Implies some successor vertices are
895  // in a multiq-ubit op together
896  VertexVec comps = {
897  circ.target(out_edges[0]), circ.target(out_edges[1]),
898  circ.target(out_edges[2])};
899  if (comps[0] == comps[1]) { // First two qubits in BRIDGE in
900  // multi-qubit op together before BRIDGE
901  BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
902  done = true;
903  } else if (comps[2] == comps[1]) { // Second two qubits in BRIDGE in
904  // multi-qubit op together before
905  // BRIDGE
906  BRIDGE_sub(v, CircPool::BRIDGE_using_CX_0());
907  done = true;
908  }
909  }
910  if (!done) { // default decomposition
911  BRIDGE_sub(v, CircPool::BRIDGE_using_CX_1());
912  }
913  }
914  return success;
915  });
916 }

918 Transform Transform::decompose_CX_directed(const Architecture &arc) {
919  return Transform([arc](Circuit &circ) {
920  bool success = false;
921  // Collect all CX type vertices
922  std::vector<std::pair<Vertex, bool>> bin;
923  for (Circuit::CommandIterator it = circ.begin(); it != circ.end(); ++it) {
924  if (it->get_op_ptr()->get_type() == OpType::CX) {
925  unit_vector_t qbs = it->get_args();
926  node_vector_t nodes = {qbs.begin(), qbs.end()};
927  if (!arc.connection_exists(nodes[0], nodes[1]) &&
928  arc.connection_exists(nodes[1], nodes[0])) {
929  // Implies CX gate is valid, and needs flipping to respect
930  // Architecture
931  bin.push_back({it.get_vertex(), false});
932  }
933  }
934  if (it->get_op_ptr()->get_type() == OpType::Conditional) {
935  const Conditional &b =
936  static_cast<const Conditional &>(*it->get_op_ptr());
937  if (b.get_op()->get_type() == OpType::CX) {
938  qubit_vector_t qbs = it->get_qubits();
939  node_vector_t nodes = {qbs.begin(), qbs.end()};
940  if (!arc.connection_exists(nodes[0], nodes[1]) &&
941  arc.connection_exists(nodes[1], nodes[0])) {
942  // Implies CX gate is valid, and needs flipping to respect
943  // Architecture
944  bin.push_back({it.get_vertex(), true});
945  }
946  }
947  if (b.get_op()->get_type() == OpType::CircBox) {
948  qubit_vector_t qbs = it->get_qubits();
949  std::shared_ptr<const Box> box_ptr =
950  std::dynamic_pointer_cast<const Box>(b.get_op());
951  qubit_vector_t all_qubits = box_ptr->to_circuit().get()->all_qubits();
952  if (all_qubits.size() != 3)
953  throw std::logic_error("Box being opened not a BRIDGE gate.");
954  std::map<Qubit, Qubit> rmap = {
955  {all_qubits[0], qbs[0]},
956  {all_qubits[1], qbs[1]},
957  {all_qubits[2], qbs[2]}};
958  box_ptr->to_circuit()->rename_units(rmap);
959  Transform::decompose_CX_directed(arc).apply(*box_ptr->to_circuit());
960  }
961  }
962  }
963  for (std::pair<Vertex, bool> v : bin) {
964  if (!v.second) {
965  Subcircuit sub = {
966  circ.get_in_edges(v.first),
967  circ.get_all_out_edges(v.first),
968  {v.first}};
969  circ.substitute(
970  CircPool::CX_using_flipped_CX(), sub, Circuit::VertexDeletion::Yes);
971  }
972  if (v.second) {
973  circ.substitute_conditional(
974  CircPool::CX_using_flipped_CX(), v.first,
975  Circuit::VertexDeletion::Yes);
976  }
977  success = true;
978  }
979  return success;
980  });
