44 bool DynamicTokenTracker::tokens_here_have_equal_locations_in_the_other_object(
45  const DynamicTokenTracker& other) const {
46  for (const auto& vertex_token_pair : m_vertex_to_token) {
47  const auto vertex = vertex_token_pair.first;
48  const auto token = vertex_token_pair.second;
49  const auto citer = other.m_vertex_to_token.find(vertex);
50 
51  if (citer == other.m_vertex_to_token.cend()) {
52  // If it's unmentioned by the other, then the vertex MUST be fixed
53  // to give the same permutation.
54  // Otherwise, the other object doesn't know where the token moved to.
55  if (vertex != token) {
56  return false;
57  }
58  } else {
59  if (token != citer->second) {
60  return false;
61  }
62  }
63  }
64  return true;
65 }

67 size_t DynamicTokenTracker::get_token_at_vertex(size_t vertex) {
68  const auto iter = m_vertex_to_token.find(vertex);
69  if (iter == m_vertex_to_token.end()) {
70  m_vertex_to_token[vertex] = vertex;
71  return vertex;
72  }
73  return iter->second;
74 }
