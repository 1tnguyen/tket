42 const ExactMappingLookup::Result&
43 ExactMappingLookup::improve_upon_existing_result(
44  const VertexMapping& desired_mapping, const vector<Swap>& edges,
45  unsigned max_number_of_swaps) {
46  max_number_of_swaps = std::min(max_number_of_swaps, 16u);
47  const auto& relabelling = m_relabeller(desired_mapping);
48 
49  if (relabelling.identity) {
50  // This beats whatever was there before,
51  // whether or not it was successful.
52  m_result.success = true;
53  m_result.too_many_vertices = false;
54  m_result.swaps.clear();
55  return m_result;
56  }
57  if (relabelling.too_many_vertices) {
58  // We cannot get a new result, so just return the existing one, whether or
59  // not it succeeded.
60  if (!m_result.success) {
61  m_result.too_many_vertices = true;
62  }
63  return m_result;
64  }
65  TKET_ASSERT(relabelling.permutation_hash != 0);
66  {
67  const bool size_match = relabelling.new_to_old_vertices.size() ==
68  relabelling.old_to_new_vertices.size();
69  TKET_ASSERT(size_match);
70  }
71  TKET_ASSERT(relabelling.new_to_old_vertices.size() >= 2);
72 
73  fill_result_from_table(relabelling, edges, max_number_of_swaps);
74  return m_result;
75 }

77 void ExactMappingLookup::fill_result_from_table(
78  const CanonicalRelabelling::Result& relabelling_result,
79  const vector<Swap>& old_edges, unsigned max_number_of_swaps) {
80  if (m_result.success) {
81  if (m_result.swaps.empty()) {
82  return;
83  }
84  max_number_of_swaps =
85  std::min<unsigned>(max_number_of_swaps, m_result.swaps.size() - 1);
86  if (max_number_of_swaps == 0) {
87  return;
88  }
89  } else {
90  m_result.swaps.clear();
91  }
92  SwapConversion::EdgesBitset new_edges_bitset = 0;
93 
94  for (auto old_edge : old_edges) {
95  const auto new_v1_opt = get_optional_value(
96  relabelling_result.old_to_new_vertices, old_edge.first);
97  if (!new_v1_opt) {
98  continue;
99  }
100  const auto new_v2_opt = get_optional_value(
101  relabelling_result.old_to_new_vertices, old_edge.second);
102  if (!new_v2_opt) {
103  continue;
104  }
105  const auto new_v1 = new_v1_opt.value();
106  const auto new_v2 = new_v2_opt.value();
107  TKET_ASSERT(new_v1 <= 5);
108  TKET_ASSERT(new_v2 <= 5);
109  new_edges_bitset |= SwapConversion::get_edges_bitset(
110  SwapConversion::get_hash_from_swap(get_swap(new_v1, new_v2)));
111  }
112 
113  const FilteredSwapSequences::SingleSequenceData table_result(
114  relabelling_result.permutation_hash, new_edges_bitset,
115  max_number_of_swaps);
116 
117  TKET_ASSERT(table_result.number_of_swaps > 0);
118  if (table_result.number_of_swaps > max_number_of_swaps) {
119  // No result in the table.
120  return;
121  }
122  TKET_ASSERT(table_result.edges_bitset != 0);
123  TKET_ASSERT(table_result.swaps_code > 0);
124 
125  m_result.success = true;
126  m_result.swaps.clear();
127  auto swaps_code_copy = table_result.swaps_code;
128  while (swaps_code_copy != 0) {
129  const auto& new_swap =
130  SwapConversion::get_swap_from_hash(swaps_code_copy & 0xF);
131  swaps_code_copy >>= 4;
132  m_result.swaps.push_back(get_swap(
133  relabelling_result.new_to_old_vertices.at(new_swap.first),
134  relabelling_result.new_to_old_vertices.at(new_swap.second)));
135  }
136  TKET_ASSERT(m_result.swaps.size() <= 16);
137 }
