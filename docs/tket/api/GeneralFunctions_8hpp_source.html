66 template <class UINT>
67 static UINT get_rightmost_bit(UINT& x) {
68  // Standard bit hack: decrementing 10000... gives 01111...
69  // E.g., consider:
70  // x = 001101011010000
71  // x-1 = 001101011001111
72  // ~(x-1) = 110010100110000
73  // Therefore, AND x with ~(x-1).
74 
75  // No "if" statements; unsigned int wraparound is allowed.
76  UINT y = x;
77  --y;
78  y = ~y;
79  const UINT bit = (x & y);
80  x ^= bit;
81  return bit;
82 }
