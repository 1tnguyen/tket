32  DistancesInterface& distances, NeighboursInterface& neighbours,
33  RiverFlowPathFinder& path_finder) {
34  const auto initial_L = get_total_home_distances(vertex_mapping, distances);
35  for (size_t counter = initial_L + 1; counter > 0; --counter) {
36  const auto swaps_before = swaps.size();
37  m_cycles_tsa.append_partial_solution(
38  swaps, vertex_mapping, distances, neighbours, path_finder);
39 
40  m_trivial_tsa.append_partial_solution(
41  swaps, vertex_mapping, distances, neighbours, path_finder);
42 
43  if (swaps_before == swaps.size()) {
44  TKET_ASSERT(all_tokens_home(vertex_mapping));
45  return;
46  }
47  }
48  TKET_ASSERT(!"hybrid TSA termination");
49 }
