31 bool MappingManager::route_circuit_with_maps(
32  Circuit& circuit, const std::vector<RoutingMethodPtr>& routing_methods,
33  std::shared_ptr<unit_bimaps_t> maps) const {
34  if (circuit.n_qubits() > this->architecture_->n_nodes()) {
35  std::string error_string =
36  "Circuit has" + std::to_string(circuit.n_qubits()) +
37  " logical qubits. Architecture has " +
38  std::to_string(this->architecture_->n_nodes()) +
39  " physical qubits. Circuit to be routed can not have more "
40  "qubits than the Architecture.";
41  throw MappingManagerError(error_string);
42  }
43 
44  // mapping_frontier tracks boundary between routed & un-routed in circuit
45  // when initialised, boundary is over output edges of input vertices
46  std::shared_ptr<MappingFrontier> mapping_frontier;
47  if (maps) {
48  mapping_frontier = std::make_shared<MappingFrontier>(circuit, maps);
49  } else {
50  mapping_frontier = std::make_shared<MappingFrontier>(circuit);
51  }
52  // updates routed/un-routed boundary
53 
54  mapping_frontier->advance_frontier_boundary(this->architecture_);
55 
56  auto check_finish = [&mapping_frontier]() {
57  for (const std::pair<UnitID, VertPort>& pair :
58  mapping_frontier->quantum_boundary->get<TagKey>()) {
59  Edge e = mapping_frontier->circuit_.get_nth_out_edge(
60  pair.second.first, pair.second.second);
61  Vertex v = mapping_frontier->circuit_.target(e);
62 
63  if (!is_final_q_type(
64  mapping_frontier->circuit_.get_OpType_from_Vertex(v))) {
65  return false;
66  }
67  }
68  return true;
69  };
70 
71  bool circuit_modified = !check_finish();
72  while (!check_finish()) {
73  // The order methods are passed in std::vector<RoutingMethod> is
74  // the order they are run
75  // If a method performs better but only on specific subcircuits,
76  // rank it earlier in the passed vector
77  bool valid_methods = false;
78  for (const auto& rm : routing_methods) {
79  // true => can use held routing method
80  std::pair<bool, unit_map_t> bool_map =
81  rm->routing_method(mapping_frontier, this->architecture_);
82  if (bool_map.first) {
83  valid_methods = true;
84  if (bool_map.second.size() > 0) {
85  std::map<Node, Node> node_map;
86  for (const auto& x : bool_map.second) {
87  node_map.insert({Node(x.first), Node(x.second)});
88  }
89  for (const std::pair<Node, Node>& swap :
90  BestTsaWithArch::get_swaps(*this->architecture_, node_map)) {
91  mapping_frontier->add_swap(swap.first, swap.second);
92  }
93  }
94  break;
95  }
96  }
97  if (!valid_methods) {
98  throw MappingManagerError(
99  "No RoutingMethod suitable to map given subcircuit.");
100  }
101  // find next routed/unrouted boundary given updates
102  mapping_frontier->advance_frontier_boundary(this->architecture_);
103  }
104  return circuit_modified;
105 }
