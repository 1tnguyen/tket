34 NeighbourPlacements::ResultVec NeighbourPlacements::get(
35  unsigned dist, unsigned n, bool optimise, unsigned seed,
36  unsigned max_tries) {
37  rng_.set_seed(seed);
38 
39  // define a comparison function for placements
40  std::vector<Qubit> keys;
41  for (auto [k, v] : init_map_) {
42  keys.push_back(k);
43  }
44  auto map_compare = [&keys](
45  const qubit_mapping_t& a, const qubit_mapping_t& b) {
46  for (auto k : keys) {
47  if (a.at(k) < b.at(k)) {
48  return true;
49  } else if (a.at(k) > b.at(k)) {
50  return false;
51  }
52  }
53  return false;
54  };
55  // set of all generated placement maps
56  std::set<qubit_mapping_t, decltype(map_compare)> placements(map_compare);
57 
58  ResultVec resvec;
59  for (unsigned i = 0; i < n; ++i) {
60  unsigned n_unsuccessful = 0;
61  while (n_unsuccessful < max_tries) {
62  Result res = gen_result(dist, optimise, max_tries);
63  if (!placements.contains(res.map)) {
64  resvec.push_back(res);
65  placements.insert(res.map);
66  break;
67  }
68  ++n_unsuccessful;
69  }
70  if (n_unsuccessful == max_tries) {
71  tket_log()->warn(
72  "Could not generate " + std::to_string(n) + " distinct placements");
73  }
74  }
75  return resvec;
76 }

78 NeighbourPlacements::Result NeighbourPlacements::gen_result(
79  unsigned dist, bool optimise, unsigned max_tries) {
80  SwapList swaps;
81  tsa_internal::SwapListOptimiser optimiser;
82 
83  // it might be impossible to find `dist` non-trivial swaps
84  unsigned n_unsuccessful = 0;
85 
86  while (swaps.size() < dist && n_unsuccessful < max_tries) {
87  Swap new_swap = gen_swap();
88 
89  if (optimise) {
90  SwapList swaps_candidate = swaps;
91  swaps_candidate.push_back(new_swap);
92  optimiser.full_optimise(swaps_candidate);
93  if (swaps_candidate.size() > swaps.size()) {
94  swaps = std::move(swaps_candidate);
95  n_unsuccessful = 0;
96  } else {
97  ++n_unsuccessful;
98  }
99  } else {
100  swaps.push_back(new_swap);
101  }
102  }
103 
104  if (n_unsuccessful == max_tries) {
105  tket_log()->warn(
106  "Unable to generate " + std::to_string(dist) +
107  " swaps for given architecture");
108  }
109 
110  return convert_to_res(swaps.to_vector());
111 }

113 Swap NeighbourPlacements::gen_swap() {
114  auto edges = arc_.get_all_edges_vec();
115  unsigned m = edges.size();
116  auto [n1, n2] = edges[rng_.get_size_t(m - 1)];
117  Swap new_swap{u_to_node_.right.at(n1), u_to_node_.right.at(n2)};
118  return new_swap;
119 }

121 NeighbourPlacements::Result NeighbourPlacements::convert_to_res(
122  const SwapVec& swaps) {
123  NodeSwapVec node_swaps;
124  for (auto [u1, u2] : swaps) {
125  node_swaps.push_back({u_to_node_.left.at(u1), u_to_node_.left.at(u2)});
126  }
127 
128  qubit_bimap_t qubit_to_node;
129  qubit_to_node.left.insert(init_map_.begin(), init_map_.end());
130  for (auto [n1, n2] : node_swaps) {
131  const Qubit q1 = qubit_to_node.right.at(n1);
132  const Qubit q2 = qubit_to_node.right.at(n2);
133  qubit_to_node.left.erase(q1);
134  qubit_to_node.left.erase(q2);
135  qubit_to_node.left.insert({q1, n2});
136  qubit_to_node.left.insert({q2, n1});
137  }
138  qubit_mapping_t map;
139  for (auto [k, v] : qubit_to_node.left) {
140  map.insert({k, v});
141  }
142  return {map, node_swaps};
143 }
