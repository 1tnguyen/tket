50 Transform Transform::synthesise_tket() {
51  Transform seq =
52  Transform::commute_through_multis() >> Transform::remove_redundancies();
53  Transform repeat = Transform::repeat(seq);
54  Transform synth = Transform::decompose_multi_qubits_CX() >>
55  Transform::remove_redundancies() >> repeat >>
56  Transform::squash_1qb_to_tk1();
57  Transform small_part = Transform::remove_redundancies() >> repeat >>
58  Transform::squash_1qb_to_tk1();
59  Transform repeat_synth = Transform::repeat_with_metric(
60  small_part, [](const Circuit &circ) { return circ.n_vertices(); });
61  return synth >> repeat_synth;
62 }

64 static Transform CXs_from_phase_gadgets(CXConfigType cx_config) {
65  return Transform([=](Circuit &circ) {
66  bool success = false;
67  VertexList bin;
68  auto [i, end] = boost::vertices(circ.dag);
69  for (auto next = i; i != end; i = next) {
70  ++next;
71  Vertex a = *i;
72  Op_ptr op = circ.get_Op_ptr_from_Vertex(a);
73  if (op->get_type() == OpType::PhaseGadget) {
74  unsigned n_qubits = op->n_qubits();
75  Circuit replacement =
76  phase_gadget(n_qubits, op->get_params()[0], cx_config);
77  Subcircuit sub = {
78  {circ.get_in_edges(a)}, {circ.get_all_out_edges(a)}, {a}};
79  circ.substitute(replacement, sub, Circuit::VertexDeletion::Yes);
80  success = true;
81  }
82  }
83  return success;
84  });
85 }

104 /* Returns a Circuit with only HQS allowed Ops (Rz, PhasedX, ZZMax) */
105 Transform Transform::synthesise_HQS() {
106  return Transform([](Circuit &circ) {
107  Transform single_loop =
108  remove_redundancies() >> commute_through_multis() >> reduce_XZ_chains();
109  Transform hqs_loop = remove_redundancies() >> commute_and_combine_HQS2() >>
110  reduce_XZ_chains();
111  Transform main_seq = decompose_multi_qubits_CX() >> clifford_simp() >>
112  decompose_ZX() >> repeat(single_loop) >>
113  decompose_CX_to_HQS2() >> repeat(hqs_loop) >>
114  decompose_ZX_to_HQS1();
115  return main_seq.apply(circ);
116  });
