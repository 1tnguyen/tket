26 const ExactMappingLookup::Result& PartialMappingLookup::operator()(
27  const VertexMapping& desired_mapping, const vector<Swap>& edges,
28  const std::set<size_t>& vertices_with_tokens_at_start,
29  unsigned max_number_of_swaps) {
30  const auto& exact_mapping_result =
31  m_exact_mapping_lookup(desired_mapping, edges, max_number_of_swaps);
32 
33  if (exact_mapping_result.success && exact_mapping_result.swaps.empty()) {
34  return exact_mapping_result;
35  }
36 
37  // Are there any empty vertices?
38  m_empty_source_vertices.clear();
39  m_empty_target_vertices.clear();
40  for (const auto& entry : desired_mapping) {
41  if (vertices_with_tokens_at_start.count(entry.first) == 0) {
42  m_empty_source_vertices.push_back(entry.first);
43  m_empty_target_vertices.push_back(entry.second);
44  }
45  }
46  if (m_empty_source_vertices.size() <= 1) {
47  // Only an exact lookup is needed (or possible).
48  return exact_mapping_result;
49  }
50 
51  // There are some empty vertices at the start.
52  // These END UP at empty target vertices
53  // (which, of course, might be completely different!)
54  // For next_permutation, let's permute the empty SOURCE vertices.
55  // They are already sorted, thus already at the first permutation
56  // in the ordering, because they came from the keys of desired_mapping.
57  {
58  const bool next_permutation = std::next_permutation(
59  m_empty_source_vertices.begin(), m_empty_source_vertices.end());
60  TKET_ASSERT(next_permutation);
61  }
62  m_altered_mapping = desired_mapping;
63 
64  for (unsigned perm_count = 0;;) {
65  for (unsigned ii = 0; ii < m_empty_source_vertices.size(); ++ii) {
66  m_altered_mapping[m_empty_source_vertices[ii]] =
67  m_empty_target_vertices[ii];
68  }
69  const auto& exact_map_result_for_permuted_vertices =
70  m_exact_mapping_lookup.improve_upon_existing_result(
71  m_altered_mapping, edges, max_number_of_swaps);
72 
73  if (exact_map_result_for_permuted_vertices.success &&
74  exact_map_result_for_permuted_vertices.swaps.empty()) {
75  return exact_map_result_for_permuted_vertices;
76  }
77  ++perm_count;
78  if (perm_count >= m_parameters.max_number_of_empty_vertex_permutations ||
79  !std::next_permutation(
80  m_empty_source_vertices.begin(), m_empty_source_vertices.end())) {
81  return exact_map_result_for_permuted_vertices;
82  }
83  }
84 }
