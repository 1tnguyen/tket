26 template <typename T>
27 static PassPtr gate_translation_pass(
28  const T &transform, OpTypeSet after_set, bool respect_connectivity,
29  const std::string &name) {
30  after_set.insert(OpType::Measure);
31  after_set.insert(OpType::Collapse);
32  after_set.insert(OpType::Reset);
33  PredicatePtrMap precons;
34  std::type_index ti = typeid(ConnectivityPredicate);
35  PredicatePtr out_gateset = std::make_shared<GateSetPredicate>(after_set);
36  PredicatePtr max2qb = std::make_shared<MaxTwoQubitGatesPredicate>();
37  PredicatePtrMap postcon_spec = {
38  CompilationUnit::make_type_pair(out_gateset),
39  CompilationUnit::make_type_pair(max2qb)};
40  PredicateClassGuarantees g_postcons;
41  if (!respect_connectivity)
42  g_postcons.insert({ti, Guarantee::Clear}); // synthesis passes do not in
43  // general preserve connectivity
44  PostConditions postcon{postcon_spec, g_postcons, Guarantee::Preserve};
45  // record pass config
46  nlohmann::json j;
47  j["name"] = name;
48  PassPtr ptr = std::make_shared<StandardPass>(precons, transform, postcon, j);
49  return ptr;
50 }

371  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
