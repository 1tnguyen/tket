39 FGVertVec Program::get_successors(const FGVert &vert) const {
40  FGEdgeVec outs = get_out_edges(vert);
41  if (outs.size() == 1) {
42  return {get_target(outs.front())};
43  } else if (outs.size() == 2) {
44  FGVertVec children(2);
45  for (const Edge &e : outs) {
46  if (get_branch(e))
47  children[1] = get_target(e);
48  else
49  children[0] = get_target(e);
50  }
51  return children;
52  } else
53  throw ProgramError("Block does not have one or two successsors");
54 }

56 FGVertVec Program::get_predecessors(const FGVert &vert) const {
57  FGEdgeVec ins = get_in_edges(vert);
58  FGVertVec parents;
59  std::unordered_set<FGVert> lookup;
60  for (const Edge &e : ins) {
61  FGVert pred = get_source(e);
62  if (lookup.find(pred) == lookup.end()) {
63  parents.push_back(pred);
64  lookup.insert(pred);
65  }
66  }
67  return parents;
68 }
