19 bool Program::check_valid() const {
20  bool valid = true;
21  valid &= get_in_edges(entry_).empty();
22  valid &= get_out_edges(entry_).size() == 1;
23  valid &= get_out_edges(exit_).empty();
24  valid &= entry_ != exit_;
25  BGL_FORALL_VERTICES(block, flow_, FlowGraph) {
26  FGEdgeVec outs = get_out_edges(block);
27  if (block != exit_) {
28  if (flow_[block].branch_condition) {
29  valid &= outs.size() == 2;
30  unsigned n_true = 0;
31  unsigned n_false = 0;
32  for (const FGEdge &e : outs) {
33  if (get_branch(e))
34  ++n_true;
35  else
36  ++n_false;
37  }
38  valid &= n_true == 1;
39  valid &= n_false == 1;
40  } else {
41  valid &= outs.size() == 1;
42  valid &= get_branch(outs.front()) == false;
43  }
44  }
45  for (const Qubit &qb : flow_[block].circ.all_qubits()) {
46  unit_lookup_t::iterator found = units_.get<TagID>().find(qb);
47  valid &= found != units_.get<TagID>().end();
48  valid &= found->type() == UnitType::Qubit;
49  }
50  for (const Bit &b : flow_[block].circ.all_bits()) {
51  unit_lookup_t::iterator found = units_.get<TagID>().find(b);
52  valid &= found != units_.get<TagID>().end();
53  valid &= found->type() == UnitType::Bit;
54  }
55  }
56  return valid;
57 }
