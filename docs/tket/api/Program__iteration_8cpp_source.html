42 Program::BlockIterator &Program::BlockIterator::operator++() {
43  while (!stack_.empty()) {
44  FGVert last_branch = stack_.back();
45  FGVertVec succs = prog_->get_successors(last_branch);
46  for (const FGVert &s : succs) {
47  if (visited_.find(s) == visited_.end() && s != prog_->exit_) {
48  current_vert_ = s;

53  }
54  stack_.pop_back();
55  }

80 Program::CommandIterator &Program::CommandIterator::operator++() {
94  if (stage_ == ComItStage::Final) {
95  // Just output STOP; move to end()
96  *this = CommandIterator();
97  return *this;
98  }
99  while (current_block_ != prog_->block_end()) {
100  switch (stage_) {
101  case ComItStage::Label: {
102  // Just reached a new block; possibly add a label
103  stage_ = ComItStage::FirstCommand;
104  FGVert block = *current_block_;
105  FGEdgeVec ins = prog_->get_in_edges(block);
106  if (ins.size() != 1 || prog_->get_source(ins.front()) != prev_block_ ||
107  prog_->get_branch(ins.front())) {
108  // Add LABEL
109  std::string label = get_label(block);
110  Op_ptr op = std::make_shared<FlowOp>(OpType::Label, label);
111  current_command_ = Command(op, {});
112  return *this;
113  } else
114  continue;
115  }
116  case ComItStage::FirstCommand: {
117  // Just output a label; start iterating through commands of block
118  current_com_iterator_ = current_block_.get_circuit_ref().begin();
119  if (current_com_iterator_ == current_block_.get_circuit_ref().end()) {
120  stage_ = ComItStage::Branch;
121  continue;
122  } else {
123  stage_ = ComItStage::Command;
124  current_command_ = *current_com_iterator_;
125  return *this;
126  }
127  }
128  case ComItStage::Command: {
129  // Just output a command; continue
130  ++current_com_iterator_;
131  if (current_com_iterator_ == current_block_.get_circuit_ref().end()) {
132  stage_ = ComItStage::Branch;
133  continue;
134  } else {
135  current_command_ = *current_com_iterator_;
136  return *this;
137  }
138  }
139  case ComItStage::Branch: {
140  // Finished commands for block; possibly add a branch
141  FGVert block = *current_block_;
142  std::optional<Bit> condition = prog_->get_condition(block);
143  stage_ = ComItStage::Goto;
144  if (condition) {
145  // Add BRANCH
146  FGVert target = prog_->get_branch_successor(block, true);
147  std::string label = get_label(target);
148  Op_ptr op = std::make_shared<FlowOp>(OpType::Branch, label);
149  current_command_ = Command(op, {*condition});
150  return *this;
151  } else
152  continue;
153  }
154  case ComItStage::Goto: {
155  // Reached end of block; possibly add a Goto before moving to next block
156  prev_block_ = *current_block_;
157  ++current_block_;
158  stage_ = ComItStage::Label;
159  FGVert target = prog_->get_branch_successor(prev_block_, false);
160  if (current_block_ == prog_->block_end()) {
161  if (target == prog_->exit_) continue;
162  } else if (target == *current_block_)
163  continue;
164  // Add Goto
165  std::string label = get_label(target);
166  Op_ptr op = std::make_shared<FlowOp>(OpType::Goto, label);
167  current_command_ = Command(op, {});
168  return *this;
169  }
170  default: {
171  // None of Stop, Final, or End should be hit before
172  // the block iterator reaches end
173  throw ProgramError(
174  "Error in command iteration: hit final stages before "
175  "reaching exit block");
176  }
177  }
178  }
179  // Label to STOP
180  if (stage_ == ComItStage::Label) {
181  std::map<FGVert, std::string>::iterator found = labels_.find(prog_->exit_);
182  if (found != labels_.end()) {
183  Op_ptr op = std::make_shared<FlowOp>(OpType::Label, found->second);
184  current_command_ = Command(op, {});
185  stage_ = ComItStage::Stop;
186  return *this;
187  }
188  }
189  // STOP command
190  Op_ptr op = std::make_shared<FlowOp>(OpType::Stop);
191  current_command_ = Command(op, {});
192  stage_ = ComItStage::Final;
193  return *this;
194 }
