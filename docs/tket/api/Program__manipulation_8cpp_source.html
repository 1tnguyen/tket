68 std::map<FGVert, FGVert> Program::copy_graph(const Program &to_copy) {
69  std::map<FGVert, FGVert> isomap;
70  if (&to_copy == this) {
71  throw ProgramError("Cannot copy a program into itself");
72  }
73  for (const Qubit &qb : to_copy.all_qubits()) {
74  add_qubit(qb, false);
75  }
76  for (const Bit &b : to_copy.all_bits()) {
77  add_bit(b, false);
78  }
79  BGL_FORALL_VERTICES(v, to_copy.flow_, FlowGraph) {
80  FGVert new_v = boost::add_vertex(this->flow_);
81  this->flow_[new_v] = to_copy.flow_[v];
82  isomap.insert({v, new_v});
83  }
84  BGL_FORALL_EDGES(e, to_copy.flow_, FlowGraph) {
85  FGVert source = isomap.at(to_copy.get_source(e));
86  FGVert target = isomap.at(to_copy.get_target(e));
87  bool branch = to_copy.get_branch(e);
88  add_edge(source, target, branch);
89  }
90  return isomap;
91 }

93 FGVert Program::add_block(const Circuit &circ) {
94  FGVert block = add_vertex(circ);
95  FGEdgeVec ins = get_in_edges(exit_);
96  for (const FGEdge &e : ins) {
97  add_edge(get_source(e), block, get_branch(e));

100  add_edge(block, exit_);

107  for (unsigned i = 0; i < args.size(); ++i) {

113  }

117 void Program::add_op(const Op_ptr &op, const unit_vector_t &args) {
118  FGVertVec lasts = get_predecessors(exit_);
119  FGVert block;
120  if (lasts.size() == 1 && lasts.front() != entry_ &&
121  !get_condition(lasts.front())) {
122  block = lasts.front();
123  } else {
124  block = add_block({});
125  }
126  Circuit &circ = flow_[block].circ;
127  op_signature_t sig = op->get_signature();
128  for (unsigned i = 0; i < args.size(); ++i) {
129  if (sig.at(i) == EdgeType::Quantum) {
130  circ.add_qubit(Qubit(args[i]), false);
131  } else {
132  circ.add_bit(Bit(args[i]), false);
133  }
134  }
135  circ.add_op(op, args);
136 }

138 void Program::append(const Program &to_append) {
139  std::map<FGVert, FGVert> isomap = copy_graph(to_append);
140  FGEdgeVec ins = get_in_edges(exit_);
141  FGVert added_entry = isomap.at(to_append.entry_);
142  FGVert target = get_branch_successor(added_entry, false);
143  for (const FGEdge &e : ins) {
144  FGVert source = get_source(e);
145  bool branch = get_branch(e);
146  add_edge(source, target, branch);
147  }
148  remove_vertex(added_entry);
149  remove_vertex(exit_);
150  exit_ = isomap.at(to_append.exit_);
151 }

153 void Program::append_if(const Bit &condition_bit, const Program &body) {
154  std::map<FGVert, FGVert> isomap = copy_graph(body);
155  FGVert added_entry = isomap.at(body.entry_);
156  FGVert added_exit = isomap.at(body.exit_);

185 void Program::append_while(const Bit &condition_bit, const Program &body) {
186  std::map<FGVert, FGVert> isomap = copy_graph(body);
187  FGVert added_entry = isomap.at(body.entry_);
188  FGVert added_exit = isomap.at(body.exit_);
189  FGVert target = get_branch_successor(added_entry, false);
190  FGVert new_exit = add_vertex(Circuit());
191  flow_[added_exit].branch_condition = condition_bit;
192  add_edge(added_exit, target, true);
193  add_edge(added_exit, new_exit, false);
194  add_edge(exit_, added_exit, false);
195  remove_vertex(added_entry);
196  exit_ = new_exit;
197 }
