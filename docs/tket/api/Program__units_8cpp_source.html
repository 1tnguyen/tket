45 std::map<Bit, unsigned> Program::bit_readout() const {
46  std::map<Bit, unsigned> res;
47 
48  // Order bits to generate indices
49  bit_vector_t all_bs = all_bits();
50  std::sort(all_bs.begin(), all_bs.end());
51  unsigned i = 0;
52  for (const Bit& b : all_bs) {

54  i++;
55  }

60 std::map<Qubit, unsigned> Program::qubit_readout() const {
61  std::map<Bit, unsigned> bit_ro = bit_readout();
62  FGVertVec finals = get_predecessors(exit_);
63  if (finals.size() == 1) {
64  // Circuit may not contain every unit from the full program, so need to map
65  // indices
66  const Circuit& circ = get_circuit_ref(finals.front());
67  std::map<Qubit, unsigned> circ_ro = circ.qubit_readout();
68  bit_vector_t circ_bits = circ.all_bits();
69  std::map<Qubit, unsigned> result;
70  for (const std::pair<const Qubit, unsigned>& pair : circ_ro) {
71  result.insert({pair.first, bit_ro.at(circ_bits.at(pair.second))});
72  }
73  return result;
74  } else
75  return {};
76 }

98 void Program::add_qubit(const Qubit& id, bool reject_dups) {
99  unit_lookup_t::index<TagID>::type::iterator found =
100  units_.get<TagID>().find(id);
101  if (found != units_.get<TagID>().end()) {
102  if (reject_dups) {
103  throw CircuitInvalidity(
104  "A unit with ID \"" + id.repr() + "\" already exists");
105  } else if (found->type() == UnitType::Qubit) {
106  return;
107  } else {
108  throw CircuitInvalidity(
109  "A bit with ID \"" + id.repr() + "\" already exists");
110  }
111  }
112  opt_reg_info_t reg_info = get_reg_info(id.reg_name());
113  register_info_t correct_info = {UnitType::Qubit, id.reg_dim()};
114  if (reg_info && !(reg_info.value() == correct_info))
115  throw CircuitInvalidity(
116  "Cannot add qubit with ID \"" + id.repr() +
117  "\" as register is not compatible");
118  units_.insert(id);
119 }
