21 size_t RNG::get_size_t(size_t max_value) {
22  if (max_value == 0) {
23  return 0;
24  }
25  // Raw data; now must convert to a value to return!
26  const std::uint64_t random_int = m_engine();
27 
28  if (max_value > m_engine.max() / 4) {
29  // If choosing such a large potential number of values,
30  // the bias will unavoidably be very bad,
31  // if only generating a single random int.
32  // Surely no deterministic function
33  // f : {0,1,...,N} -> {0,1,...,M}
34  // can be close to giving a uniform distribution,
35  // if N != M are both large and nearly equal.
36  // (Should be a theorem in here somewhere!)
37  if (max_value >= m_engine.max()) {
38  // Care! Maybe max_value+1 == 0 by wraparound,
39  // so we cannot do division by max_value+1 !
40  return random_int;
41  }
42  return random_int % (max_value + 1);
43  }
44 
45  // NOW we know that max_value+1 won't overflow.
46 
47  // Mathematical note on the below: let:
48  // m = maximum possible value of "random_int"
49  // w = interval_width
50  // v = max possible value to return.
51  //
52  // Thus, random_int could be one of {0,1,2,...,m},
53  // and we must return one of {0,1,2,...,v}.
54  //
55  // With int arithmetic, we get w = int((m+1)/(v+1)).
56  //
57  // e.g., if m=5, v=2, then w = int(6/3) = 2,
58  // the possible random_int values are {0,1,2,3,4,5},
59  // and this is partitioned into 3 sets:
60  // {0,1}, {2,3}, {4,5}.
61  //
62  // [Since, with int arithmetic,
63  // int(0/2) = int(1/2) = 0, int(2/2) = int(3/2) = 1, ...]
64  //
65  // Because these sets have equal size 2, each of the values 0,1,2
66  // has equal probability 1/3 of being returned.
67  // BUT, what if (m+1)/(v+1) is not an integer?
68  //
69  // e.g., m=5, v=3.
70  // Now, we must partition the set {0,1,2,3,4,5} into 4 subsets.
71  // With the below algorithm, w=int((5+1)/(3+1)) = 1, so the partition is
72  // {0}, {1}, {2}, {3,4,5}.
73  // Notice that 0,1,2 have probabilities 1/6 of being returned,
74  // but v=3 has probability 3/6 of being returned, quite a large bias.
75  //
76  // How bad can it be? In general:
77  //
78  // (m+1)/(v+1) - 1 < w <= (m+1)/(v+1).
79  // Thus
80  // m-v+1 <= w(v+1) <= m+1.
81  //
82  // Now, the random_int sets causing the values 0,1,...,v to be returned are
83  //
84  // { 0, 1, ..., w-1} --> returns 0
85  // { w, w+1, ..., 2w-1} --> returns 1
86  // {2w, 2w+1, ..., 3w-1} --> returns 2
87  // ....
88  // {vw, vw+1, ..., (v+1)w - 1, ... , m } --> returns v
89  //
90  // Notice that all sets except the last have size w.
91  // The last set has size m-vw+1. So, the final value v has
92  // more ways than the other values 0,1,... to be returned, by a factor of
93  //
94  // U = (m-vw+1)/w = (m+1)/w - v.
95  //
96  // U is the "bias factor" which we want to be as close to 1 as possible.
97  // Always, U >= (m+1)/[(m+1)/(v+1)] - v = v+1-v = 1,
98  // as we already know. Also,
99  //
100  // U <= (m+1)(v+1)/(m-v+1) - v.
101  //
102  // Let's assume that v << m.
103  // Then we can expand with a geometric series:
104  // (m+1)(v+1)/(m-v+1) = (v+1).[1-v/(m+1)]^{-1}
105  // = (v+1).[1 + v/(m+1) + A]
106  // = v+1 + v(v+1)/(m+1) + (v+1)A,
107  //
108  // where A ~ (v/m)^2, with ~ here meaning
109  // "roughly equal size, up to constant factors".
110  // Thus, U <= 1 + v(v+1)/(m+1) + (v+1)A.
111  //
112  // So, finally, assume also that v(v+1) << m+1.
113  // [This is the same as saying v^2 << m, since m = 2^64-1 is very large,
114  // and thus m+1~m, sqrt(m+1)~sqrt(m)].
115  //
116  // ...then: A ~ (v^2/m) / m << 1/m, (v+1)A << v/m << 1,
117  // and so U = 1 + C where C << 1.
118  //
119  // Thus, the bias towards the max value v is negligible, as required.
120 
121  // Divide range into approximately equal widths.
122  // Notice, we can't do m_engine.max()+1 because it overflows to 0.
123  // But the chance of getting m_engine.max() is negligibly small anyway.
124  const std::uint64_t interval_width =
125  m_engine.max() /
126  // Doesn't overflow, because of the above checks.
127  (static_cast<std::uint64_t>(max_value) + 1);
128 
129  // interval_width cannot be zero, because we ensured above that
130  // max_value + 1 <= m_engine.max().
131  const size_t result = random_int / interval_width;
132 
133  // Modulo arithmetic shouldn't be necessary, but be paranoid,
134  // in case there are mistakes in the above analysis (very likely!)
135  return result % (max_value + 1);
136 }

138 size_t RNG::get_size_t(size_t min_value, size_t max_value) {
139  if (min_value > max_value) {
140  std::swap(min_value, max_value);
141  }
142  return min_value + get_size_t(max_value - min_value);
143 }

145 vector<size_t> RNG::get_permutation(size_t size) {
146  vector<size_t> numbers(size);
147  for (size_t i = 0; i < size; ++i) {
148  numbers[i] = i;
149  }
150  do_shuffle(numbers);
151  return numbers;
152 }

156 bool RNG::check_percentage(size_t percentage) {
157  // e.g. the numbers {0,1,2,3,4} are 5%
158  // of the numbers {0,1,...,99}.
