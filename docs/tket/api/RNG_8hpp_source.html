24 // Something like this is needed for proper random test data generation
25 // if you want to be platform-independent, as the C++ standard is stupid.
26 // (A major scandal, in my opinion).
27 // The random engines are mostly guaranteed by the standard,
28 // but the DISTRIBUTIONS, e.g. uniform_distribution, are NOT
29 // (i.e., the actual algorithm used to convert a string of bits to a number
30 // in the range {0,1,2,...,N} is not specified at all by the C++ standard).
31 // Thus, we are NOT guaranteed to get the same results, even with the same
32 // (1) engine; (2) initial seed; (3) distribution,
33 // if we use different platforms (or even different compilers
34 // on the SAME platform), or even different compiler VERSIONS!!!
35 //
36 // The C++ standard as far as I know does not specify ANY distribution
37 // implementations, not even optionally, so you HAVE to do this yourself,
38 // even for something as simple as a uniform distribution.
39 // The same applies to, e.g., std::random_shuffle.

105  template <class T>
106  void do_shuffle(std::vector<T>& elements) {
107  if (elements.size() < 2) {
108  return;
109  }
110  m_shuffling_data.resize(elements.size());
111  for (size_t i = 0; i < m_shuffling_data.size(); ++i) {
112  m_shuffling_data[i].first = m_engine();
113  // Tricky subtle point: without this extra entry to break ties,
114  // std::sort could give DIFFERENT results across platforms and compilers,
115  // if the object T allows unequal elements comparing equal.
116  m_shuffling_data[i].second = i;
117  }
118  std::sort(
119  m_shuffling_data.begin(), m_shuffling_data.end(),
120  [](const std::pair<std::uintmax_t, size_t>& lhs,
121  const std::pair<std::uintmax_t, size_t>& rhs) {
122  return lhs.first < rhs.first ||
123  (lhs.first == rhs.first && lhs.second < rhs.second);
124  });
125  // Don't need to make a copy of "elements"! Just do repeated swaps...
126  for (size_t i = 0; i < m_shuffling_data.size(); ++i) {
127  const size_t& j = m_shuffling_data[i].second;
128  if (i != j) {
129  std::swap(elements[i], elements[j]);
130  }
131  }
132  }

160  }
161  size_t index = get_size_t(elements.size() - 1);
162  const T copy = elements[index];
163  elements[index] = elements.back();
164  elements.pop_back();
165  return copy;
166  }
