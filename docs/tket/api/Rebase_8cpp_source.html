41 static bool standard_rebase(
42  Circuit& circ, const OpTypeSet& multiqs, const Circuit& cx_replacement,
43  const OpTypeSet& singleqs,
44  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
45  tk1_replacement) {
46  bool success = false;
47  VertexList bin;
48  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
49  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
50  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
51  if (n_qubits <= 1) continue;
52  bool conditional = op->get_type() == OpType::Conditional;
53  if (conditional) {
54  const Conditional& cond = static_cast<const Conditional&>(*op);
55  op = cond.get_op();
56  }
57  OpType type = op->get_type();
58  if (multiqs.find(type) != multiqs.end() || type == OpType::CX ||
59  type == OpType::Barrier)
60  continue;
61  // need to convert
62  Circuit replacement = CX_circ_from_multiq(op);
63  if (conditional) {
64  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
65  } else {
66  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
67  }
68  bin.push_back(v);
69  success = true;
70  }
71  if (multiqs.find(OpType::CX) == multiqs.end()) {
72  const Op_ptr cx_op = get_op_ptr(OpType::CX);
73  success = circ.substitute_all(cx_replacement, cx_op) | success;
74  }
75  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
76  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) != 1 ||
77  circ.n_in_edges_of_type(v, EdgeType::Quantum) != 1)
78  continue;
79  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
80  bool conditional = op->get_type() == OpType::Conditional;
81  if (conditional) {
82  const Conditional& cond = static_cast<const Conditional&>(*op);
83  op = cond.get_op();
84  }
85  OpType type = op->get_type();
86  if (!is_gate_type(type) || is_projective_type(type) ||
87  singleqs.find(type) != singleqs.end())
88  continue;
89  // need to convert
90  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
91  Circuit replacement =
92  tk1_replacement(tk1_angles[0], tk1_angles[1], tk1_angles[2]);
93  if (conditional) {
94  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
95  } else {
96  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
97  }
98  circ.add_phase(tk1_angles[3]);
99  bin.push_back(v);
100  success = true;
101  }
102  circ.remove_vertices(
103  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
104  return success;
105 }

146 Circuit Transform::tk1_to_rzh(
147  const Expr& alpha, const Expr& beta, const Expr& gamma) {
148  Circuit c(1);
149  std::optional<unsigned> cliff = equiv_Clifford(beta, 4);
150  if (cliff) {
151  switch (*cliff % 4) {
152  case 0: {
153  c.add_op<unsigned>(OpType::Rz, gamma + alpha, {0});
154  break;
155  }
156  case 1: {
157  c.add_op<unsigned>(OpType::Rz, gamma - 0.5, {0});
158  c.add_op<unsigned>(OpType::H, {0});
159  c.add_op<unsigned>(OpType::Rz, alpha - 0.5, {0});
160  c.add_phase(-0.5);
161  break;
162  }
163  case 2: {
164  c.add_op<unsigned>(OpType::Rz, gamma - alpha, {0});
165  c.add_op<unsigned>(OpType::H, {0});
166  c.add_op<unsigned>(OpType::Rz, 1., {0});
167  c.add_op<unsigned>(OpType::H, {0});
168  break;
169  }
170  case 3: {
171  c.add_op<unsigned>(OpType::Rz, gamma + 0.5, {0});
172  c.add_op<unsigned>(OpType::H, {0});
173  c.add_op<unsigned>(OpType::Rz, alpha + 0.5, {0});
174  c.add_phase(-0.5);
175  break;
176  }
177  }
178  if (cliff >= 4u) c.add_phase(1.);
179  } else {
180  c.add_op<unsigned>(OpType::Rz, gamma, {0});
181  c.add_op<unsigned>(OpType::H, {0});
182  c.add_op<unsigned>(OpType::Rz, beta, {0});
183  c.add_op<unsigned>(OpType::H, {0});
184  c.add_op<unsigned>(OpType::Rz, alpha, {0});
185  }
186  Transform::remove_redundancies().apply(c);
187  return c;
188 }

196 Circuit Transform::tk1_to_rzsx(
197  const Expr& alpha, const Expr& beta, const Expr& gamma) {
198  Circuit c(1);
199  Expr correction_phase = 0;
200  if (equiv_0(beta)) {
201  // b = 2k, if k is odd, then Rx(b) = -I
202  c.add_op<unsigned>(OpType::Rz, alpha + gamma, {0});
203  correction_phase = int_half(beta);
204  } else if (equiv_0(beta + 1)) {
205  // Use Rx(2k-1) = i(-1)^{k}SxSx
206  correction_phase = -0.5 + int_half(beta - 1);
207  if (equiv_0(alpha - gamma)) {
208  // a - c = 2m
209  // overall operation is (-1)^{m}Rx(2k -1)
210  c.add_op<unsigned>(OpType::SX, {0});
211  c.add_op<unsigned>(OpType::SX, {0});
212  correction_phase += int_half(alpha - gamma);
213  } else {
214  c.add_op<unsigned>(OpType::Rz, gamma, {0});
215  c.add_op<unsigned>(OpType::SX, {0});
216  c.add_op<unsigned>(OpType::SX, {0});
217  c.add_op<unsigned>(OpType::Rz, alpha, {0});
218  }
219  } else if (equiv_0(beta - 0.5) && equiv_0(alpha) && equiv_0(gamma)) {
220  // a = 2k, b = 2m+0.5, c = 2n
221  // Rz(2k)Rx(2m + 0.5)Rz(2n) = (-1)^{k+m+n}e^{-i \pi /4} SX
222  c.add_op<unsigned>(OpType::SX, {0});
223  correction_phase =
224  int_half(beta - 0.5) + int_half(alpha) + int_half(gamma) - 0.25;
225  } else if (equiv_0(alpha - 0.5) && equiv_0(gamma - 0.5)) {
226  // Rz(2k + 0.5)Rx(b)Rz(2m + 0.5) = -i(-1)^{k+m}SX.Rz(1-b).SX
227  c.add_op<unsigned>(OpType::SX, {0});
228  c.add_op<unsigned>(OpType::Rz, 1 - beta, {0});
229  c.add_op<unsigned>(OpType::SX, {0});
230  correction_phase = int_half(alpha - 0.5) + int_half(gamma - 0.5) - 0.5;
231  } else {
232  c.add_op<unsigned>(OpType::Rz, gamma + 0.5, {0});
233  c.add_op<unsigned>(OpType::SX, {0});
234  c.add_op<unsigned>(OpType::Rz, beta - 1, {0});
235  c.add_op<unsigned>(OpType::SX, {0});
236  c.add_op<unsigned>(OpType::Rz, alpha + 0.5, {0});
237  correction_phase = -0.5;
238  }
239  c.add_phase(correction_phase);
240  Transform::remove_redundancies().apply(c);
241  return c;
242 }
