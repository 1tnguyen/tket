82 void RiverFlowPathFinder::Impl::grow_path(
83  size_t target_vertex, size_t required_path_size) {
84  TKET_ASSERT(path.size() < required_path_size);
85  TKET_ASSERT(!path.empty());
86 
87  // We don't yet know how to move on, so we must choose a neighbour.
88  // All candidates will have the same edge count.
89  candidate_moves.clear();
90 
91  const auto remaining_distance = required_path_size - path.size();
92  const auto& neighbours = neighbours_calculator(path.back());
93  distances_calculator.register_neighbours(path.back(), neighbours);
94 
95  for (size_t neighbour : neighbours) {
96  const auto neighbour_distance_to_target =
97  distances_calculator(neighbour, target_vertex);
98 
99  if (neighbour_distance_to_target == remaining_distance - 1) {
100  // Notice that nonexistent entries will be automatically set
101  // to have count 0, by the C++ standard.
102  const auto edge_count = edge_counts[get_swap(path.back(), neighbour)];
103  if (!candidate_moves.empty()) {
104  // We'll only add candidates with the same count or higher.
105  if (candidate_moves[0].count > edge_count) {
106  continue;
107  }
108  if (candidate_moves[0].count < edge_count) {
109  candidate_moves.clear();
110  }
111  }
112  candidate_moves.emplace_back();
113  candidate_moves.back().end_vertex = neighbour;
114  candidate_moves.back().count = edge_count;
115  continue;
116  }
117  // GCOVR_EXCL_START
118  TKET_ASSERT(
119  neighbour_distance_to_target == remaining_distance ||
120  neighbour_distance_to_target == remaining_distance + 1 ||
121  AssertMessage() << "d(v_" << path.back() << ", v_" << target_vertex
122  << ")=" << remaining_distance << ". But v_"
123  << path.back() << " has neighbour v_" << neighbour
124  << ", at distance " << neighbour_distance_to_target
125  << " to the target v_" << target_vertex);
126  // GCOVR_EXCL_STOP
127  }
128  // GCOVR_EXCL_START
129  TKET_ASSERT(
130  !candidate_moves.empty() ||
131  AssertMessage() << "No neighbours of v_" << path.back()
132  << " at correct distance " << remaining_distance - 1
133  << " to target vertex v_" << target_vertex);
134  // GCOVR_EXCL_STOP
135 
136  const auto& choice = rng.get_element(candidate_moves);
137  path.push_back(choice.end_vertex);
138 }

158 const vector<size_t>& RiverFlowPathFinder::operator()(
159  size_t vertex1, size_t vertex2) {
160  m_pimpl->path.clear();
161  m_pimpl->path.push_back(vertex1);
162  if (vertex1 == vertex2) {
163  return m_pimpl->path;
164  }
165 
166  // We must build up the path.
167  // The number of vertices including the source and target.
168  const size_t final_path_size =
169  1 + m_pimpl->distances_calculator(vertex1, vertex2);
170 
171  for (size_t infinite_loop_guard = 10 * final_path_size;
172  infinite_loop_guard != 0; --infinite_loop_guard) {
173  m_pimpl->grow_path(vertex2, final_path_size);
174  if (m_pimpl->path.size() == final_path_size) {
175  TKET_ASSERT(m_pimpl->path.back() == vertex2);
176  m_pimpl->update_data_with_path();
177  return m_pimpl->path;
178  }
179  }
180  throw std::runtime_error("get path - dropped out of loop");
181 }
