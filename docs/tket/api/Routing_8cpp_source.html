103 bool subgraph_remove_if_connected(
104  Architecture& arc, const Architecture& subarc, const Node& node) {
105  // do not remove if node is in subarc
106  if (subarc.uid_exists(node)) {
107  return false;
108  }
109  if (subarc.n_uids() > 0) {
110  node_set_t ap = arc.get_articulation_points(subarc);
111 
112  if (ap.find(node) != ap.end()) {
113  return false;
114  }
115  }
116 
117  arc.remove_uid(node);
118  return true;
119 }

121 void remove_unmapped_nodes(
122  Architecture& arc, qubit_bimap_t& map, Circuit& circ) {
123  std::vector<Node> unmapped_nodes;
124  std::vector<Node> mapped_nodes;
125 
126  r_const_iterator_t iend = map.right.end();
127  for (const UnitID& uid : arc.get_all_uids()) {
128  Node n(uid);
129  r_const_iterator_t find_node = map.right.find(n);
130  if (find_node == iend) {
131  unmapped_nodes.push_back(n);
132  } else {
133  mapped_nodes.push_back(n);
134  }
135  }
136  Architecture subarc = arc.create_subarch(mapped_nodes);
137 
138  // sort mapped nodes from least connected to most (remove least connected
139  // first)
140  std::sort(
141  unmapped_nodes.begin(), unmapped_nodes.end(), [&arc](Node x, Node y) {
142  return (arc.get_out_degree(x) < arc.get_out_degree(y));
143  });
144 
145  qubit_vector_t available;
146  for (const Qubit& q : circ.all_qubits()) {
147  if (map.left.find(q) == map.left.end()) {
148  available.push_back(q);
149  }
150  }
151 
152  for (const Node& node : unmapped_nodes) {
153  if (!subgraph_remove_if_connected(arc, subarc, node)) {
154  // if node can't be removed, map to first unmapped qubit
155  if (available.empty())
156  throw CircuitInvalidity(
157  "Routing is unable to construct connected placement from partial "
158  "placement using unplaced logical qubits. Please update the "
159  "circuit placement to a set of connected physical qubits.");
160  map.insert({available.front(), node});
161  available.erase(available.begin());
162  }
163  }
164 }

166 qubit_mapping_t get_qmap_from_circuit(Architecture& arc, Circuit& circ) {
167  qubit_vector_t all_qbs = circ.all_qubits();
168  node_set_t all_nodes = arc.get_all_uids_set();
169 
170  qubit_mapping_t qubit_map;
171  for (Qubit q : all_qbs) {
172  Node n(q);
173  if (all_nodes.find(n) != all_nodes.end()) {
174  qubit_map.insert({q, n});

176  }

180 std::pair<Circuit, bool> Routing::solve(const RoutingConfig& config) {
181  config_ = config;
182  qubit_mapping_t qubit_map = get_qmap_from_circuit(current_arc_, circ_);
183  slice_frontier_.init();
184  if (slice_frontier_.slice->empty()) {
185  organise_registers_and_maps();
186  } else {
187  // Some nodes are permanently unused due to difference in architecture nodes
188  // and number of used wires in circuit To account for this, place highest
189  // numbered wires (i.e. unused) into set bad nodes of architecture
190 
191  // Placement method attempts to find a good initial allocation of qubits to
192  // nodes, aiming to reduce overall circuit depth. The method aims to put
193  // intreacting qubits in the first few circuit timesteps on adjacent nodes
194  // If no placement, qubits placed sequentially on nodes i.e. qubit 0 -> node
195  // 0 etc.
196 
197  if (qubit_map.size() != 0) {
198  init_map.left.insert(qubit_map.begin(), qubit_map.end());
199  }
200  remove_unmapped_nodes(current_arc_, init_map, circ_);
201  final_map = remap(init_map);
202  organise_registers_and_maps();
203  }
204  bool modified = circuit_modified();
205  return {circ_, modified};
206 }

208 // Tidying up of qregisters and initial and final maps after SWAP adding.
209 void Routing::organise_registers_and_maps() {
210  // Given all the new empty wires with no home, if a qubit isnt in the initial
211  // map, find it an unassigned node and shove it there.
212  auto all_nodes = original_arc_.get_all_uids_vec();
213  unsigned next_ind = 0;
214  Node next_node = all_nodes[next_ind];
215 
216  for (const Qubit& qb : circ_.all_qubits()) {
217  if (init_map.left.find(qb) == init_map.left.end()) {
218  // find next free node
219  while (init_map.right.count(next_node)) {
220  next_node = all_nodes[++next_ind];
221  if (next_ind == all_nodes.size()) {
222  throw ArchitectureMismatch(circ_.n_qubits(), current_arc_.n_uids());
223  }
224  }
225  init_map.left.insert({qb, next_node});
226  final_map.left.insert({qb, next_node});
227  }
228  }
229 
230  // Due to the addition of SWAP gates, a qubit path may change, and so it's
231  // output boundary ordering may not match the input boundary ordering. The
232  // following updates the output boundary to match the ordering of the final
233  // slice frontier Make the input boundary match up to node numbering of
234  // architecture.
235  boundary_t new_boundary;
236  qubit_mapping_t reorder_map = bimap_to_map(init_map.left);
237  for (const std::pair<const Qubit, Node>& map : reorder_map) {
238  Qubit target = final_map.right.at(map.second);
239  new_boundary.insert(
240  {map.second, circ_.get_in(map.first), circ_.get_out(target)});
241  // Which makes it all nicer
242  }
243  // add classical bits to new_boundary
244  for (auto [it, end] =
245  circ_.boundary.get<TagType>().equal_range(UnitType::Bit);
246  it != end; it++) {
247  new_boundary.insert(*it);
248  }
249 
250  circ_.boundary = new_boundary;
251  circ_.update_initial_map(reorder_map);
252  circ_.update_final_map(bimap_to_map(final_map.left));
253 }

255 // Remap completes the routing algorithm
256 // slices passed as copy as 3 pass placement needs original preserved
257 qubit_bimap_t Routing::remap(const qubit_bimap_t& init) {
258  qmap = init;
259 
260  advance_frontier();
261  // The routing algorithm:
262  // 1) Slices of circuit are parallelised/packed/whatever into 'timesteps'
263  // 2) Swaps are 'proposed' on edges connected to any nodes housing an
264  // 'interacting' qubit (interacting -> qubit is in some two qubit interaction
265  // in timestep 0) 3) A distance heuristic is used to determine whether the
266  // swap proposed will bring interacting qubits closer 4) If a swap is bring
267  // interacting qubits together it is compared to a held 'best swap'. The
268  // comparison is achieved by applying the same distance heuristic over future
269  // timesteps, until one is deemed strictly better. 5) If a succesful swap is
270  // found (from 3)), the swap gate is added to the circuit, information on
271  // which nodes home which qubits is updated and 1)->4) is repeated. 6) If no
272  // succesful swap is found, Dijkstra's algorithm is used to find a path in
273  // the graph between two interacting qubits, which the qubits are then swapped
274  // along.
275  // ... The pair of interacting qubits in the first timestep with greatest path
276  // distance between them is chosen. Algorithm then repeats 1)->4).
277  // for(unsigned count=0;slice_frontier_.slice.size()!=0 && count<2;count++){
278  while (!slice_frontier_.slice->empty()) {
279  SwapResults single_swap = try_all_swaps(current_arc_.get_connections_vec());
280  if (single_swap.success) {
281  route_stats.n_try_all_swaps++;
282  perform_action(single_swap.swap);
283  } else {
284  route_stats.n_solve_furthest++;
285  if (!solve_furthest()) {
286  throw RoutingFailure();
287  }
288  }
289  advance_frontier();
290  }
291 
292  qubit_bimap_t final_qmap;
293  for (l_const_iterator_t it = qmap.left.begin(); it != qmap.left.end(); ++it) {
294  Edge e = slice_frontier_.quantum_out_edges->get<TagKey>()
295  .find(it->first)
296  ->second;
297  Vertex v = circ_.target(e);
298  while (!circ_.detect_final_Op(v)) {
299  e = circ_.get_next_edge(v, e);
300  v = circ_.target(e);
301  }
302  Qubit out_q(circ_.get_id_from_out(v));
303  final_qmap.insert({out_q, it->second});
304  }
305 
306  return final_qmap;
307 }
