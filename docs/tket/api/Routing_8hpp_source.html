119 // structure of configuration parameters for routing
120 struct RoutingConfig {
121  // circuit look ahead limit for SWAP picking
122  unsigned depth_limit;
123  // circuit look ahead limit for Distributed CX gate checking
124  unsigned distrib_limit;
125  // number of interactions considered in Distributed CX gate checking
126  unsigned interactions_limit;
127  // Whether to use a Distributed CX gate instead of a SWAP and a CX is
128  // determined by comparing the distance between some interacting pairs of
129  // qubits with and without the permutation. Changing distrib_exponent changes
130  // how much later interactions are considered. distrib_exponent < 0 => less
131  // effect from later interactoins, distrib_exponent > 0 => greater effect,
132  // distrib_exponent = 0 => no effect
133  double distrib_exponent;
134  // Constructors
135  RoutingConfig(
136  unsigned _depth_limit, unsigned _distrib_limit,
137  unsigned _interactions_limit, const double &_distrib_exponent)
138  : depth_limit(_depth_limit),
139  distrib_limit(_distrib_limit),
140  interactions_limit(_interactions_limit),
141  distrib_exponent(_distrib_exponent) {}
142 
143  RoutingConfig() : RoutingConfig(50, 75, 10, 0) {}
144 
145  bool operator==(const RoutingConfig &other) const;
146 };
