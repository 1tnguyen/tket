36 class StandardSquasher : public AbstractSquasher {
37  private:
38  using Func = std::function<Circuit(const Expr &, const Expr &, const Expr &)>;
39 
40  public:
41  StandardSquasher(const OpTypeSet &singleqs, const Func &tk1_replacement)
42  : singleqs(singleqs), squash_fn(tk1_replacement), combined() {
43  for (OpType ot : singleqs) {
44  if (!is_single_qubit_type(ot))
45  throw NotValid(
46  "OpType given to standard_squash is not a single qubit gate");
47  }
48  }
49 
50  bool accepts(OpType type) const override {
51  return (singleqs.find(type) != singleqs.end()) && !is_projective_type(type);
52  }
53 
54  void append(Gate_ptr gate) override {
55  std::vector<Expr> angs = gate->get_tk1_angles();
56  combined.apply(Rotation(OpType::Rz, angs.at(2)));
57  combined.apply(Rotation(OpType::Rx, angs.at(1)));
58  combined.apply(Rotation(OpType::Rz, angs.at(0)));
59  }
60 
61  std::pair<Circuit, Gate_ptr> flush(
62  std::optional<Pauli> = std::nullopt) const override {
63  auto [a, b, c] = combined.to_pqp(OpType::Rz, OpType::Rx);
64  Circuit replacement = squash_fn(c, b, a);
65  BGL_FORALL_VERTICES(rv, replacement.dag, DAG) {
66  OpType v_type = replacement.get_OpType_from_Vertex(rv);
67  if (!is_boundary_q_type(v_type) &&
68  singleqs.find(v_type) == singleqs.end()) {
69  throw NotValid(
70  "tk1_replacement given to standard_squash "
71  "does not preserve gate set");
72  }
73  }
74  return {replacement, nullptr};
75  }
76 
77  void clear() override { combined = Rotation(); }
78 
79  private:
80  const OpTypeSet &singleqs;
81  const Func &squash_fn;
82  Rotation combined;
83 };

85 static bool standard_squash(
86  Circuit &circ, const OpTypeSet &singleqs,
87  const std::function<Circuit(const Expr &, const Expr &, const Expr &)>
88  &tk1_replacement) {
89  auto squasher = std::make_unique<StandardSquasher>(singleqs, tk1_replacement);
90  return SingleQubitSquash(std::move(squasher), false).squash(circ);
91 }

93 Transform squash_factory(
94  const OpTypeSet &singleqs,
95  const std::function<Circuit(const Expr &, const Expr &, const Expr &)>
96  &tk1_replacement) {
97  return Transform([=](Circuit &circ) {
98  return standard_squash(circ, singleqs, tk1_replacement);
99  });
