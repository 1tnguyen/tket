64 unsigned SwapConversion::get_number_of_swaps(
65  SwapConversion::SwapHash swaps_code) {
66  unsigned num_swaps = 0;
67  while (swaps_code != 0) {
68  ++num_swaps;
69  const auto swap_hash = swaps_code & 0xF;
70  swaps_code >>= 4;
71  TKET_ASSERT(swap_hash > 0);
72  TKET_ASSERT(swap_hash <= 15);
73  }
74  return num_swaps;
75 }
