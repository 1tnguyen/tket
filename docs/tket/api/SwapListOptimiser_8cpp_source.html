39 void SwapListOptimiser::optimise_pass_remove_empty_swaps(
40  SwapList& list, VertexMapping vertex_mapping) {
41  auto id_opt = list.front_id();
42  while (id_opt) {
43  const auto id = id_opt.value();
44  id_opt = list.next(id);
45  const VertexSwapResult result(list.at(id), vertex_mapping);
46  if (result.tokens_moved == 0) {
47  list.erase(id);
48  }
49  }

52 std::optional<SwapListOptimiser::ID>
53 SwapListOptimiser::get_id_of_previous_blocker(SwapList& list, SwapID id) {
54  const auto& initial_swap = list.at(id);
55 
56  // This is the first non-disjoint swap it hits when it moves back.
57  // Guaranteed to be valid if we drop out of the loop.
58  SwapID current_id = id;
59 
60  bool terminated_correctly = false;
61  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
62  --infinite_loop_guard) {
63  const auto prev_id = list.previous(current_id);
64  if (!prev_id) {
65  // Right at the front!
66  return {};
67  }
68  current_id = prev_id.value();
69  const auto& new_swap = list.at(current_id);
70  if (!disjoint(initial_swap, new_swap)) {
71  // Blocks, OR identical
72  if (new_swap != initial_swap) {
73  // It blocks
74  return current_id;
75  }
76  terminated_correctly = true;
77  break;
78  }
79  }
80  TKET_ASSERT(terminated_correctly);
81  // It's hit a copy of itself
82  list.erase(id);
83  list.erase(current_id);
84  return {};
85 }

87 bool SwapListOptimiser::move_swap_towards_front(SwapList& list, SwapID id) {
88  TKET_ASSERT(list.front_id());
89  if (id == list.front_id().value()) {
90  return false;
91  }
92  const auto old_size = list.size();
93  const auto previous_blocker_opt = get_id_of_previous_blocker(list, id);
94  if (old_size != list.size()) {
95  // The swap was erased!
96  return true;
97  }
98  if (previous_blocker_opt) {
99  // It can't move all the way to the front.
100  const ID blocker = previous_blocker_opt.value();
101 
102  // Must be non-null.
103  const ID previous_id = list.previous(id).value();
104  if (blocker != previous_id) {
105  // Do the move...erase before insert to minimise possible sizes...
106  const auto swap = list.at(id);
107  list.erase(id);
108  const auto new_id = list.insert_after(blocker);
109  list.at(new_id) = swap;
110  }
111  return false;
112  }
113  // There was no blocker, so we CAN move all the way to the front
114  // (and we checked before that we're not already at the front).
115  const auto swap = list.at(id);
116  list.erase(id);
117  list.push_front(swap);
118  return false;
119 }

121 void SwapListOptimiser::optimise_pass_with_zero_travel(SwapList& list) {
122  if (list.size() <= 1) {
123  return;
124  }
125  ID current_id = list.front_id().value();
126 
127  // This moves swaps to cancel with previous identical swaps,
128  // if there is nothing blocking the move.
129  // However, only worth doing if previous identical swaps do exist;
130  // leaves them unchanged otherwise.
131  // We can be sneaky: rather than storing all previous IDs
132  // for each swap, we store the NUMBER of them; we don't need
133  // to know the previous location, since the move back
134  // will check for that anyway.
135  //
136  // This probably could be cleverly optimised further
137  // but would require more thought.
138  for (auto& entry : m_data) {
139  // This is quicker than clearing and reinserting;
140  // no tree rebalancing.
141  entry.second = 0;
142  }
143  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
144  --infinite_loop_guard) {
145  const auto next_id_opt = list.next(current_id);
146 
147  // C++ guarantees nonexistent values will be set to 0.
148  auto& swap_count = m_data[list.at(current_id)];
149  if (swap_count == 0) {
150  swap_count = 1;
151  } else {
152  // There's a possibility of cancellation.
153  const auto old_size = list.size();
154  (void)get_id_of_previous_blocker(list, current_id);
155  if (old_size == list.size()) {
156  // No cancellation.
157  ++swap_count;
158  } else {
159  // Cancellation occurred; "get_id_of_previous_blocker" already erased
160  // both vertex swaps, but didn't update the counts.
161  --swap_count;
162  }
163  }
164  if (!next_id_opt) {
165  return;
166  }
167  current_id = next_id_opt.value();
168  }
169  TKET_ASSERT(!"optimise_pass_with_zero_travel termination");
170 }

172 void SwapListOptimiser::optimise_pass_with_frontward_travel(SwapList& list) {
173  if (list.size() <= 1) {
174  return;
175  }
176  // Start one past the front.
177  ID current_id = list.front_id().value();
178  current_id = list.next(current_id).value();
179 
180  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
181  --infinite_loop_guard) {
182  const auto next_id_opt = list.next(current_id);
183  move_swap_towards_front(list, current_id);
184  if (!next_id_opt) {
185  return;
186  }
187  current_id = next_id_opt.value();
188  }
189  TKET_ASSERT(!"optimise_pass_with_frontward_travel termination");
190 }

200 void SwapListOptimiser::
201  optimise_pass_with_token_tracking_without_clearing_tracker(SwapList& list) {
202  if (list.size() <= 1) {
203  return;
204  }
205  // Put a different token at each vertex, and start swapping.
206  // Now, if a TOKEN swap (rather than vertex swap)
207  // repeats, then removing this vertex swap together with the preceding one
208  // in which those two tokens were exchanged gives the same final result.
209  // This is because, if we don't actually carry out the first swap,
210  // everything proceeds as before, and all tokens except those two
211  // are in the same place. When we reach the time of the second swap,
212  // everything is as before EXCEPT that the two tokens have changed places;
213  // thus the effect of the second swap
214  // was merely to interchange those two tokens again.
215  //
216  // Now, m_data will store the previous LOCATIONS of vertex swaps.
217  //
218  // The actual values of the tokens are irrelevant,
219  // as long as they are distinct.
220 
221  const auto invalid_index = VectorListHybridSkeleton::get_invalid_index();
222 
223  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
224  --infinite_loop_guard) {
225  // Keep looping until we stop changing.
226  // The size is always decreasing or unchanged;
227  // we never insert, only erase.
228  const auto old_size = list.size();
229  if (old_size == 0) {
230  return;
231  }
232  for (auto& entry : m_data) {
233  entry.second = invalid_index;
234  }
235  ID current_id = list.front_id().value();
236  bool terminated_correctly = false;
237  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
238  --infinite_loop_guard) {
239  const auto& vertex_swap = list.at(current_id);
240  const auto token_swap = m_token_tracker.do_vertex_swap(vertex_swap);
241  const auto citer = m_data.find(token_swap);
242  if (citer != m_data.cend() && citer->second != invalid_index) {
243  // The swap occurred before, the entry tells us the ID.
244  // Erase both swaps.
245  list.erase(citer->second);
246  list.erase(current_id);
247  // We have to start at the beginning.
248  // Changing the labels for these tokens
249  // messes up other entries between the two swaps.
250  // A warm restart from the middle of the swap list
251  // would be a lot of extra complication, not worth it for now.
252  terminated_correctly = true;
253  break;
254  }
255  // Swap hasn't occurred before, now advance.
256  m_data[token_swap] = current_id;
257  const auto next_id_opt = list.next(current_id);
258  if (!next_id_opt) {
259  terminated_correctly = true;
260  break;
261  }
262  current_id = next_id_opt.value();
263  }
264  TKET_ASSERT(terminated_correctly);
265  const auto new_size = list.size();
266  if (old_size == new_size) {
267  return;
268  }
269  TKET_ASSERT(new_size < old_size);
270  }
271  TKET_ASSERT(!"optimise_pass_with_token_tracking termination");
272 }

274 void SwapListOptimiser::full_optimise(SwapList& list) {
275  // More experimentation needed to find the best combination.
276  optimise_pass_with_zero_travel(list);
277  m_token_tracker.reset();
278  optimise_pass_with_token_tracking_without_clearing_tracker(list);
279 }

281 void SwapListOptimiser::full_optimise(
282  SwapList& list, const VertexMapping& vertex_mapping) {
283  for (auto infinite_loop_guard = 1 + list.size(); infinite_loop_guard > 0;
284  --infinite_loop_guard) {
285  const auto old_size = list.size();
286  full_optimise(list);
287  optimise_pass_remove_empty_swaps(list, vertex_mapping);
288  if (old_size == list.size() || list.size() == 0) {
289  return;
290  }
291  TKET_ASSERT(list.size() < old_size);
292  }
293  TKET_ASSERT(!"full_optimise termination");
294 }
