28 const SwapListSegmentOptimiser::Output&
29 SwapListSegmentOptimiser::optimise_segment(
30  SwapID initial_id, const std::set<size_t>& vertices_with_tokens_at_start,
31  VertexMapResizing& map_resizing, SwapList& swap_list) {
32  m_best_optimised_swaps.clear();
33 
34  // Nonzero if and only if a valid sequence of swaps was stored.
35  m_output.initial_segment_size = 0;
36 
37  // If the mapping has too many vertices, it MAY happen that
38  // adding more swaps REDUCES the number of vertices
39  // (since, some vertices may move back to their original positions,
40  // and hence be "ignored"). Thus, we ALLOW the lookup to fail a few times
41  // due to too many vertices before we give up.
42  const int max_consecutive_too_many_vertices = 5;
43  int too_many_vertices_count = max_consecutive_too_many_vertices;
44 
45  VertexMapping current_map;
46  {
47  const auto& initial_swap = swap_list.at(initial_id);
48  current_map[initial_swap.first] = initial_swap.second;
49  current_map[initial_swap.second] = initial_swap.first;
50  }
51  size_t current_number_of_swaps = 1;
52  VertexMapping current_map_copy;
53  for (auto next_id_opt = swap_list.next(initial_id);;) {
54  bool too_many_vertices = false;
55 
56  // As we keep adding swaps to a sequence and updating the resultant
57  // target->source vertex mapping, should we look up EVERY mapping in the
58  // table, or is it enough to do so only when the map increases in size, etc.
59  // etc.? Desperately need some theory here! We look up almost EVERYTHING, so
60  // table lookup is one possible slowdown; reducing unnecessary lookups is
61  // worthwhile.
62  // TODO: think of some theory, and experiment!
63  bool attempt_to_optimise = current_map.size() >= 3;
64  if (!attempt_to_optimise && !next_id_opt) {
65  // Because it's the FINAL segment, optimise it whatever we do.
66  attempt_to_optimise = true;
67  }
68  if (attempt_to_optimise) {
69  // We're going to attempt to optimise.
70  current_map_copy = current_map;
71  const auto& resize_result = map_resizing.resize_mapping(current_map);
72  if (resize_result.success) {
73  const auto& lookup_result = m_mapping_lookup(
74  current_map, resize_result.edges, vertices_with_tokens_at_start,
75  current_number_of_swaps);
76 
77  if (lookup_result.success) {
78  // We've got a new result from the table; do we store it?
79  bool should_store = m_output.initial_segment_size == 0;
80  if (!should_store) {
81  // Something IS stored, but is our new solution better?
82  // GCOVR_EXCL_START
83  TKET_ASSERT(
84  m_output.initial_segment_size >= m_best_optimised_swaps.size());
85  // GCOVR_EXCL_STOP
86  const size_t current_decrease =
87  m_output.initial_segment_size - m_best_optimised_swaps.size();
88  TKET_ASSERT(current_number_of_swaps >= lookup_result.swaps.size());
89  const size_t new_decrease =
90  current_number_of_swaps - lookup_result.swaps.size();
91  should_store = new_decrease > current_decrease;
92  }
93  if (should_store) {
94  m_output.initial_segment_size = current_number_of_swaps;
95  m_best_optimised_swaps = lookup_result.swaps;
96  }
97  } else {
98  if (lookup_result.too_many_vertices) {
99  too_many_vertices = true;
100  }
101  }
102  } else {
103  // We couldn't resize the mapping, so there must be too many vertices.
104  too_many_vertices = true;
105  // Also, the vertex mapping may be corrupted, so restore it
106  current_map = current_map_copy;
107  }
108  }
109 
110  if (too_many_vertices) {
111  --too_many_vertices_count;
112  if (too_many_vertices_count <= 0) {
113  break;
114  }
115  } else {
116  too_many_vertices_count = max_consecutive_too_many_vertices;
117  }
118 
119  // Now add a swap.
120  if (next_id_opt) {
121  const auto id = next_id_opt.value();
122  const Swap swap = swap_list.at(id);
123  add_swap(current_map, swap);
124  ++current_number_of_swaps;
125  next_id_opt = swap_list.next(id);
126  } else {
127  // We've reached the end!
128  break;
129  }
130  }
131  fill_final_output_and_swaplist(initial_id, swap_list);
132  return m_output;
133 }

135 void SwapListSegmentOptimiser::fill_final_output_and_swaplist(
136  SwapID initial_id, SwapList& swap_list) {
137  if (m_output.initial_segment_size == 0) {
138  // No improvement was found.
139  m_output.final_segment_size = 0;
140  m_output.new_segment_last_id = {};
141  return;
142  }
143  m_output.final_segment_size = m_best_optimised_swaps.size();
144  TKET_ASSERT(m_output.final_segment_size <= m_output.initial_segment_size);
145  const auto initial_size = swap_list.size();
146 
147  if (m_best_optimised_swaps.empty()) {
148  swap_list.erase_interval(initial_id, m_output.initial_segment_size);
149  m_output.new_segment_last_id = {};
150  } else {
151  const auto overwrite_result = swap_list.overwrite_interval(
152  initial_id, m_best_optimised_swaps.cbegin(),
153  m_best_optimised_swaps.cend());
154 
155  // GCOVR_EXCL_START
156  TKET_ASSERT(
157  overwrite_result.number_of_overwritten_elements ==
158  m_best_optimised_swaps.size());
159  // GCOVR_EXCL_STOP
160  m_output.new_segment_last_id =
161  overwrite_result.final_overwritten_element_id;
162 
163  const size_t remaining_elements_to_erase =
164  m_output.initial_segment_size - m_output.final_segment_size;
165 
166  const auto next_id_opt =
167  swap_list.next(overwrite_result.final_overwritten_element_id);
168  if (next_id_opt) {
169  swap_list.erase_interval(
170  next_id_opt.value(), remaining_elements_to_erase);
171  }
172  }
173  // GCOVR_EXCL_START
174  TKET_ASSERT(
175  swap_list.size() + m_output.initial_segment_size ==
176  initial_size + m_output.final_segment_size);
177  // GCOVR_EXCL_STOP
178 }
