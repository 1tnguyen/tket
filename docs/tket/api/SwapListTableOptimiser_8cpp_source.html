26 enum class EmptySwapCheckResult {
27  NOT_EMPTY,
28  CONTINUE_AFTER_ERASURE,
29  TERMINATE_AFTER_ERASURE
30 };
31 
32 // current_id is KNOWN to be valid.
33 // vertices_with_tokens is correct just BEFORE performing the swap.
34 // If the swap is empty, erase it and update current_id (to the next swap).
35 static EmptySwapCheckResult check_for_empty_swap(
36  const std::set<size_t>& vertices_with_tokens, SwapID& current_id,
37  SwapList& swap_list) {
38  const auto swap = swap_list.at(current_id);
39  if (vertices_with_tokens.count(swap.first) != 0 ||
40  vertices_with_tokens.count(swap.second) != 0) {
41  return EmptySwapCheckResult::NOT_EMPTY;
42  }
43  const auto next_id_opt = swap_list.next(current_id);
44  swap_list.erase(current_id);
45  if (!next_id_opt) {
46  return EmptySwapCheckResult::TERMINATE_AFTER_ERASURE;
47  }
48  current_id = next_id_opt.value();
49  return EmptySwapCheckResult::CONTINUE_AFTER_ERASURE;
50 }
51 
52 // current_id is KNOWN to be valid.
53 // vertices_with_tokens is correct just BEFORE performing the swap.
54 // Keep erasing empty swaps and updating current_id
55 // until EITHER we hit a nonempty swap, OR we run out of swaps,
56 // and thus return false.
57 static bool erase_empty_swaps_interval(
58  const std::set<size_t>& vertices_with_tokens, SwapID& current_id,
59  SwapList& swap_list) {
60  for (auto infinite_loop_guard = 1 + swap_list.size(); infinite_loop_guard > 0;
61  --infinite_loop_guard) {
62  switch (check_for_empty_swap(vertices_with_tokens, current_id, swap_list)) {
63  case EmptySwapCheckResult::CONTINUE_AFTER_ERASURE:
64  // Maybe more to erase!
65  break;
66  case EmptySwapCheckResult::NOT_EMPTY:
67  return true;
68  case EmptySwapCheckResult::TERMINATE_AFTER_ERASURE:
69  return false;
70  default:
71  TKET_ASSERT(!"unknown EmptySwapCheckResult enum");
72  break;
73  }
74  }
75  // Should never get here!
76  TKET_ASSERT(!"erase_empty_swaps_interval failed to terminate");
77  return false;
78 }
79 
80 // current_id is KNOWN to be valid and nonempty.
81 // vertices_with_tokens is correct just BEFORE we perform the current swap.
82 // Perform the swap (i.e., updating vertices_with_tokens),
83 // and advance current_id to the next swap.
84 static bool perform_current_nonempty_swap(
85  std::set<size_t>& vertices_with_tokens, SwapID& current_id,
86  const SwapList& swap_list) {
87  const auto swap = swap_list.at(current_id);
88 
89  if (vertices_with_tokens.count(swap.first) == 0) {
90  // No empty swaps!
91  TKET_ASSERT(vertices_with_tokens.count(swap.second) != 0);
92  // Second has a token, first doesn't.
93  TKET_ASSERT(vertices_with_tokens.insert(swap.first).second);
94  TKET_ASSERT(vertices_with_tokens.erase(swap.second) == 1);
95  } else {
96  // First has a token.
97  if (vertices_with_tokens.count(swap.second) == 0) {
98  // Second has no token.
99  TKET_ASSERT(vertices_with_tokens.erase(swap.first) == 1);
100  TKET_ASSERT(vertices_with_tokens.insert(swap.second).second);
101  }
102  }
103 
104  const auto next_id_opt = swap_list.next(current_id);
105  if (!next_id_opt) {
106  return false;
107  }
108  current_id = next_id_opt.value();
109  return true;
110 }

112 void SwapListTableOptimiser::optimise(
113  const std::set<size_t>& vertices_with_tokens_at_start,
114  VertexMapResizing& map_resizing, SwapList& swap_list,
115  SwapListOptimiser& swap_list_optimiser) {
116  if (vertices_with_tokens_at_start.empty()) {
117  swap_list.clear();
118  return;
119  }
120  if (swap_list.empty()) {
121  return;
122  }
123 
124  // Because we'll go in both directions, we need to know
125  // which tokens exist at the END of the mapping.
126  auto vertices_with_tokens_at_end = vertices_with_tokens_at_start;
127  {
128  // Already checked to be nonempty.
129  auto current_id = swap_list.front_id().value();
130  bool terminated_correctly = false;
131  for (auto infinite_loop_guard = 1 + swap_list.size();
132  infinite_loop_guard > 0; --infinite_loop_guard) {
133  if (!erase_empty_swaps_interval(
134  vertices_with_tokens_at_end, current_id, swap_list)) {
135  terminated_correctly = true;
136  break;
137  }
138  if (!perform_current_nonempty_swap(
139  vertices_with_tokens_at_end, current_id, swap_list)) {
140  terminated_correctly = true;
141  break;
142  }
143  }
144  TKET_ASSERT(terminated_correctly);
145  if (swap_list.size() <= 1) {
146  return;
147  }
148  }
149  // Now begin the forward/backward loop.
150  for (auto infinite_loop_guard = 1 + swap_list.size(); infinite_loop_guard > 0;
151  --infinite_loop_guard) {
152  const auto old_size = swap_list.size();
153  optimise_in_forward_direction(
154  vertices_with_tokens_at_start, map_resizing, swap_list,
155  swap_list_optimiser);
156 
157  swap_list.reverse();
158  optimise_in_forward_direction(
159  vertices_with_tokens_at_end, map_resizing, swap_list,
160  swap_list_optimiser);
161 
162  // Must reverse again to get back to start!
163  swap_list.reverse();
164  const auto new_size = swap_list.size();
165  TKET_ASSERT(new_size <= old_size);
166  if (new_size == old_size) {
167  return;
168  }
169  }
170  TKET_ASSERT(!"SwapListTableOptimiser::optimise");
171 }

173 void SwapListTableOptimiser::optimise_in_forward_direction(
174  const std::set<size_t>& vertices_with_tokens_at_start,
175  VertexMapResizing& map_resizing, SwapList& swap_list,
176  SwapListOptimiser& swap_list_optimiser) {
177  swap_list_optimiser.optimise_pass_with_frontward_travel(swap_list);
178 
179  m_segment_optimiser.optimise_segment(
180  swap_list.front_id().value(), vertices_with_tokens_at_start, map_resizing,
181  swap_list);
182 
183  if (swap_list.size() <= 1) {
184  return;
185  }
186  // Will always remain valid. We perform this swap and then optimise
187  // starting from the next one.
188  auto current_id = swap_list.front_id().value();
189  auto vertices_with_tokens = vertices_with_tokens_at_start;
190 
191  for (size_t infinite_loop_guard = swap_list.size(); infinite_loop_guard != 0;
192  --infinite_loop_guard) {
193  if (!erase_empty_swaps_interval(
194  vertices_with_tokens, current_id, swap_list)) {
195  return;
196  }
197  // We now have a valid ID with nonempty swap.
198  if (!perform_current_nonempty_swap(
199  vertices_with_tokens, current_id, swap_list)) {
200  return;
201  }
202 
203  // NOW we want to optimise from this ID.
204  // However, we must be careful; maybe it will be erased, so we have
205  // to get the PREVIOUS and recover from there.
206  const auto previous_id_opt = swap_list.previous(current_id);
207 
208  m_segment_optimiser.optimise_segment(
209  current_id, vertices_with_tokens, map_resizing, swap_list);
210 
211  // We now want to set "current_id" to the first swap of
212  // the newly optimised segment (if any) - which may of course
213  // be unchanged, changed, or empty.
214 
215  // If there was no previous ID, we must have been at the front
216  // just before we optimised.
217  auto current_id_opt = swap_list.front_id();
218  if (previous_id_opt) {
219  // There WAS a previous ID, so we CAN move onto the next.
220  current_id_opt = swap_list.next(previous_id_opt.value());
221  }
222  if (!current_id_opt) {
223  return;
224  }
225  current_id = current_id_opt.value();
226  }
227 }
