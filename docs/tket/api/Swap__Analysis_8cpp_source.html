27 // Generates distance vector from input interaction vector
28 std::vector<std::size_t> Routing::generate_distance_vector(
29  const Interactions &inter) const {
30  const unsigned n = current_arc_.get_diameter();
31  // const unsigned n = active_distance_matrix.maxCoeff();
32  if (n < 1) {
33  throw ArchitectureInvalidity("Architecture has diameter 0.");
34  }
35  std::vector<std::size_t> dv(n - 1);
36  for (auto [n1, n2] : inter) {
37  unsigned dist = current_arc_.get_distance(n1, n2);
38  if (dist > 1) {
39  ++dv[n - dist];
40  }
41  }
42  return dv;
43 }

45 // Returns the distance between n1 and p1 and the distance between n2 and p2,
46 // distance ordered (greatest first)
47 const std::pair<std::size_t, std::size_t> Routing::pair_dists(
48  const Node &n1, const Node &p1, const Node &n2, const Node &p2) const {
49  std::size_t curr_dist1 = current_arc_.get_distance(n1, p1);
50  std::size_t curr_dist2 = current_arc_.get_distance(n2, p2);
51  return (curr_dist1 > curr_dist2) ? std::make_pair(curr_dist1, curr_dist2)
52  : std::make_pair(curr_dist2, curr_dist1);
53 }

55 // Determines if a proposed swap brings interacting qubits closer, improving
56 // board state.
57 bool Routing::swap_decreases(
58  const Swap &nodes, const Interactions &inte) const {
59  Node node1 = nodes.first;
60  Node pair1 = inte.at(node1);
61  Node node2 = nodes.second;
62  Node pair2 = inte.at(node2);
63 
64  if (pair1 == node2 || (node1 == pair1 && node2 == pair2)) {
65  return false;
66  }
67  const std::pair<std::size_t, std::size_t> &curr_dists =
68  pair_dists(node1, pair1, node2, pair2);
69  const std::pair<std::size_t, std::size_t> &news_dists =
70  pair_dists(node2, pair1, node1, pair2);
71 
72  return news_dists < curr_dists;
73 }

77 void Routing::increment_distance(
78  graph::dist_vec &new_dist_vector, const Swap &pair, int increment) const {
79  const unsigned n = current_arc_.get_diameter();
80  const unsigned dis_index =
81  n - current_arc_.get_distance(pair.first, pair.second);
82  if (dis_index < new_dist_vector.size()) {
83  new_dist_vector[dis_index] += increment;

92 // Updates distance vector from presented interaction vector
93 graph::dist_vec Routing::update_distance_vector(
94  const Swap &nodes, std::vector<std::size_t> new_dist_vector,
95  const Interactions &inte) const {
96  increment_distance(new_dist_vector, {nodes.first, inte.at(nodes.first)}, -2);
97  increment_distance(
98  new_dist_vector, {nodes.second, inte.at(nodes.second)}, -2);
99  increment_distance(new_dist_vector, {nodes.second, inte.at(nodes.first)}, 2);
100  increment_distance(new_dist_vector, {nodes.first, inte.at(nodes.second)}, 2);
101  return new_dist_vector;
102 }

104 // Updates qmap to reflect performed swap
105 void Routing::update_qmap(qubit_bimap_t &map, const Swap &swap) {
106  const Qubit qb1 = map.right.at(swap.first);
107  const Qubit qb2 = map.right.at(swap.second);
108  map.right.erase(swap.first);
109  map.right.erase(swap.second);
110  map.left.insert({qb1, swap.second});
111  map.left.insert({qb2, swap.first});
112 }

114 std::vector<Swap> Routing::candidate_swaps(
115  const std::vector<Architecture::Connection> &trial_edges,
116  const Interactions &inte) const {
117  std::vector<Swap> potential_swaps;
118  for (auto [node, adjacent_node] : trial_edges) {
119  if (inte.at(node) != node || inte.at(adjacent_node) != adjacent_node) {
120  Swap proposed = {node, adjacent_node};
121  if (swap_decreases(proposed, inte)) {
122  potential_swaps.push_back(proposed);
123  }
124  }
125  }
126  return potential_swaps;
127 }

129 // Move heuristic in try_all_swaps loop outside, for testing help and easy
130 // changing?
131 std::vector<Swap> Routing::cowtan_et_al_heuristic(
132  std::vector<Swap> &candidate_swaps,
133  const std::vector<std::size_t> &base_dists,
134  const Interactions &interac) const {
135  const Swap winner = candidate_swaps.back();
136  candidate_swaps.pop_back();
137  std::vector<std::size_t> winner_distances =
138  update_distance_vector(winner, base_dists, interac);
139  std::vector<Swap> smaller_set;
140  smaller_set.push_back(winner);
141  for (const Swap &proposed_swap : candidate_swaps) {
142  const std::vector<std::size_t> proposed_distances =
143  update_distance_vector(proposed_swap, base_dists, interac);
144  const int comp =
145  tri_lexicographical_comparison(proposed_distances, winner_distances);
146  if (comp == -1) {
147  smaller_set.push_back(proposed_swap);
148  } else if (comp == 1) {
149  smaller_set = {proposed_swap};
150  winner_distances = proposed_distances;
151  }
152  }
153  return smaller_set;
154 }

156 SwapResults Routing::try_all_swaps(const std::vector<Architecture::Connection>
157  &trial_edges) { // don't need to change
158  std::vector<Swap> potential_swaps = candidate_swaps(trial_edges, interaction);
159 
160  if (potential_swaps.empty()) return {false, {Node(0), Node(0)}};
161 
162  RoutingFrontier high_sf = slice_frontier_;
163 
164  for (unsigned i = 0; i < config_.depth_limit && !high_sf.slice->empty() &&
165  potential_swaps.size() > 1;
166  i++) {
167  Interactions interac =
168  (i == 0) ? interaction : generate_interaction_frontier(high_sf);
169  std::vector<std::size_t> base_dists =
170  (i == 0) ? dist_vector : generate_distance_vector(interac);
171 
172  potential_swaps =
173  cowtan_et_al_heuristic(potential_swaps, base_dists, interac);
174 
175  high_sf.next_slicefrontier();
176  }
177 
178  return {1, potential_swaps.back()};
179 }

181 std::vector<Swap> Routing::path_to_swaps(const std::vector<Node> &path) {
182  const unsigned len = path.size();
183  std::vector<Swap> output_swaps;
184  if (len > 2) {
185  unsigned halfway = len / 2;
186  for (unsigned i = 0; (i < halfway) || ((halfway + 2 + i) < len); i++) {
187  if (i < halfway) {
188  Swap sw1 = {path[i], path[i + 1]};
189  output_swaps.push_back(sw1);
190  }
191  if ((halfway + 2 + i) < len) {
192  Swap sw2 = {path[len - i - 2], path[len - i - 1]};
193  output_swaps.push_back(sw2);
194  }
195  }
196  }
197  return output_swaps;
198 }

200 // If heuristic can't settle on a suitable single swap or pair of swaps, find a
201 // path between the two interacting qubits at greatest distance and swap along
202 // it.
203 bool Routing::solve_furthest() {
204  bool success = false;
205  std::optional<Node> max_node = std::nullopt;
206  unsigned max_dist = 0;
207  for (auto [q1, q2] : interaction) {
208  unsigned dist = current_arc_.get_distance(q1, q2);
209  if (dist > max_dist) {
210  max_dist = dist;
211  max_node = q1;
212  }
213  }
214  if (!max_node.has_value()) {
215  throw ArchitectureInvalidity("Architecture is disconnected");
216  }
217  Node root = *max_node;
218  if (max_dist > 1) {
219  Node target = interaction.at(root);
220  const std::vector<Node> path = current_arc_.get_path(root, target);
221  const std::vector<Swap> swaps_to_perform = path_to_swaps(path);
222  for (const Swap &swap : swaps_to_perform) {
223  success = true;
224  add_swap(swap);
225  }
226  }
227  return success;
228 }

254 // Difference in distance between interacting qubits between nodes in SWAP can
255 // only differ by 1 Compares the difference in distance between all given
256 // interactions, scales them dependent on how timesteps to interaction, and
257 // returns whether a distributed cx is desired.
258 void Routing::compare_distributed_cx_distances(
259  distributed_cx_info &candidate_distributed_cx,
260  const std::pair<std::vector<Node>, std::vector<Node>> &inter_node) {
261  std::pair<int, int> distance_check = {0, 0};
262  for (unsigned i = 1; i < inter_node.first.size(); i++) {
263  distance_check.first += pow(i, config_.distrib_exponent) *
264  (int(current_arc_.get_distance(
265  inter_node.second[0], inter_node.first[i])) -
266  int(current_arc_.get_distance(
267  inter_node.first[0], inter_node.first[i])));
268  }
269  for (unsigned i = 1; i < inter_node.second.size(); i++) {
270  distance_check.second += pow(i, config_.distrib_exponent) *
271  (int(current_arc_.get_distance(
272  inter_node.first[0], inter_node.second[i])) -
273  int(current_arc_.get_distance(
274  inter_node.second[0], inter_node.second[i])));
275  }
276  if (distance_check.first < 0) {
277  candidate_distributed_cx.first.first = false;
278  }
279  if (distance_check.second < 0) {
280  candidate_distributed_cx.second.first = false;
281  }
282 }

284 bool check_vertex_is_CX(const Circuit &circ_, const Vertex &v) {
285  OpType ot = circ_.get_OpType_from_Vertex(v);
286  if (ot != OpType::CX) {
287  if (ot == OpType::Conditional) {
288  const Conditional &b =
289  static_cast<const Conditional &>(*circ_.get_Op_ptr_from_Vertex(v));
290  if (b.get_op()->get_type() != OpType::CX) {
291  return false;
292  }
293  } else {
294  return false;
295  }
296  }
297  return true;
298 }
299 // Method is supplied with a pair of nods with the intention of being swapped.
300 // Before this SWAP gate is added, this method considers whether a distributed
301 // CX gate between interacting qubits distance 2 away from eachother is a better
302 // option The returned bool pair instructs perform_action whether to add a
303 // distributed CX gate between nodes.first and its partner node and nodes.second
304 // and its partner node respectively
305 distributed_cx_info Routing::check_distributed_cx(const Swap &nodes) {
306  // 1) Determine which nodes in SWAP gate could complete their CX with a
307  // distributed CX gate instead
308  distributed_cx_info candidate_distributed_cx = {
309  {current_arc_.get_distance(nodes.first, interaction[nodes.first]) == 2,
310  Node(0)},
311  {current_arc_.get_distance(nodes.second, interaction[nodes.second]) == 2,
312  Node(0)}};
313  // 1 pt2) Is the vertex a CX gate or Conditioned CX gate?
314  auto cx_check = [&](bool candidate, const Qubit &qb) {
315  if (candidate)
316  return check_vertex_is_CX(
317  circ_,
318  circ_.target(slice_frontier_.quantum_in_edges->find(qb)->second));
319  return true;
320  };
321  if (!cx_check(
322  candidate_distributed_cx.first.first, qmap.right.at(nodes.first)))
323  return {{false, Node(0)}, {false, Node(0)}};
324  if (!cx_check(
325  candidate_distributed_cx.second.first, qmap.right.at(nodes.second)))
326  return {{false, Node(0)}, {false, Node(0)}};
327 
328  if (candidate_distributed_cx.first.first ||
329  candidate_distributed_cx.second.first) {
330  // 2) Find number of next interactions for node in SWAP equivalent to
331  // config, or reached within depth limit
332  std::pair<std::vector<Node>, std::vector<Node>> inter_node = {
333  {nodes.first}, {nodes.second}};
334  std::pair<unsigned, unsigned> ni_limit = {0, 0};
335 
336  RoutingFrontier high_sf = slice_frontier_;
337 
338  for (unsigned i = 0; i < config_.distrib_limit && !high_sf.slice->empty() &&
339  (ni_limit.first < config_.interactions_limit ||
340  ni_limit.second < config_.interactions_limit);
341  i++) {
342  // Find interaction frontier for current slice and find the interacting
343  // pairs of nodes for incident SWAP gate.
344  Interactions interac =
345  (i == 0) ? interaction : generate_interaction_frontier(high_sf);
346 
347  if (nodes.first != interac[nodes.first] &&
348  ni_limit.first < config_.interactions_limit) {
349  inter_node.first.push_back(interac[nodes.first]);
350  ni_limit.first++;
351  }
352  if (nodes.second != interac[nodes.second] &&
353  ni_limit.second < config_.interactions_limit) {
354  inter_node.second.push_back(interac[nodes.second]);
355  ni_limit.second++;
356  }
357  high_sf.next_slicefrontier();
358  }
359  if (ni_limit.first > 0 && ni_limit.second > 0) {
360  // 3) Compare difference in distances between interacting qubits given the
361  // permutation of qubits from added SWAP gate, or not.
362  compare_distributed_cx_distances(candidate_distributed_cx, inter_node);
363  if (candidate_distributed_cx.first.first ||
364  candidate_distributed_cx.second.first) {
365  // 4) If desirable, find the central node of the bridge.
366  update_central_nodes(nodes, interaction, candidate_distributed_cx);
367  return candidate_distributed_cx;
368  }
369  }
370  }
371  return {{false, Node(0)}, {false, Node(0)}};
372 }

374 // Give a node with a control qubit on it, finds its respective target node and
375 // node between them, and replaces the CX gate between the control and target
376 // with a distributed CX
377 void Routing::add_distributed_cx(
378  const Node &cx_node_0, const Node &cx_node_1, const Node &central_node) {
379  // Find interacting node for starting_node, find node between them. Also swap
380  // control and target node if necessary.
381 
382  if (current_arc_.get_distance(cx_node_0, cx_node_1) != 2) {
383  throw BridgeInvalid("Bridge Nodes are not distance 2 apart.");
384  }
385  if (current_arc_.get_distance(cx_node_0, central_node) != 1 ||
386  current_arc_.get_distance(cx_node_1, central_node) != 1) {
387  throw BridgeInvalid(
388  "Central BRIDGE node not adjacent to Control and Target "
389  "nodes.");
390  }
391 
392  route_stats.bridge_count++;
393  Edge edge_0 =
394  slice_frontier_.quantum_in_edges->find(qmap.right.at(cx_node_0))->second;
395  Edge edge_1 =
396  slice_frontier_.quantum_in_edges->find(qmap.right.at(cx_node_1))->second;
397 
398  // Assign control and target nodes from cx_node_0 and cx_node_1
399  // Depends on the port ordering of the cx_node_0 and cx_node_1 corresponding
400  // edges attached to the CX vertex
401  Node control_node, target_node;
402  if (circ_.get_ports(edge_1).second < circ_.get_ports(edge_0).second) {
403  control_node = cx_node_1;
404  target_node = cx_node_0;
405  } else {
406  control_node = cx_node_0;
407  target_node = cx_node_1;
408  }
409 
410  // Find qubits associated to each node
411  const Qubit control_qb = qmap.right.at(control_node);
412  const Qubit central_qb = qmap.right.at(central_node);
413  const Qubit target_qb = qmap.right.at(target_node);
414 
415  // Initialize variables appropriate for substituting Conditionals with CX
416  // gates to Conditionals with BRIDGE gates.
417  Op_ptr new_bridge_ptr;
418  EdgeVec b_in_edges = {};
419  OpType gate_op;
420  std::vector<std::tuple<Vertex, port_t, port_t>> classical_edge_info = {};
421 
422  Vertex to_be_replaced = slice_frontier_.circ.target(
423  slice_frontier_.quantum_in_edges->find(control_qb)->second);
424  // If OpType is a Conditional{CX}, replace with Conditional{BRIDGE} instead
425  if (circ_.get_OpType_from_Vertex(to_be_replaced) == OpType::Conditional) {
426  Op_ptr pt = circ_.get_Op_ptr_from_Vertex(to_be_replaced);
427  const Conditional &b = static_cast<const Conditional &>(
428  *circ_.get_Op_ptr_from_Vertex(to_be_replaced));
429  gate_op = b.get_op()->get_type();
430  new_bridge_ptr = std::make_shared<Conditional>(
431  get_op_ptr(OpType::BRIDGE, std::vector<Expr>(), 3), b.get_width(),
432  b.get_value());
433  // Also collect any classical in edges
434  b_in_edges = circ_.get_in_edges_of_type(to_be_replaced, EdgeType::Boolean);
435  for (Edge e : b_in_edges) {
436  classical_edge_info.push_back(
437  {circ_.source(e), circ_.get_source_port(e),
438  circ_.get_target_port(e)});
439  }
440  } else { // else make normal bridge
441  new_bridge_ptr = get_op_ptr(OpType::BRIDGE);
442  gate_op = circ_.get_OpType_from_Vertex(to_be_replaced);
443  }
444 
445  if (gate_op != OpType::CX) {
446  throw BridgeInvalid(
447  "OpType::BRIDGE being substituted for a vertex that isn't "
448  "OpType::CX. Please rebase two-qubit primitive to CX gate.");
449  }
450  // Collect all required Quantum edge information
451 
452  Edge control_in_edge =
453  slice_frontier_.quantum_in_edges->find(control_qb)->second;
454  Edge control_out_edge =
455  slice_frontier_.quantum_out_edges->find(control_qb)->second;
456  Edge central_edge =
457  slice_frontier_.quantum_in_edges->find(central_qb)->second;
458  Edge target_in_edge =
459  slice_frontier_.quantum_in_edges->find(target_qb)->second;
460  Edge target_out_edge =
461  slice_frontier_.quantum_out_edges->find(target_qb)->second;
462 
463  VertPort control_pred = {
464  circ_.source(control_in_edge), circ_.get_source_port(control_in_edge)};
465  VertPort central_pred = {
466  circ_.source(central_edge), circ_.get_source_port(central_edge)};
467  VertPort target_pred = {
468  circ_.source(target_in_edge), circ_.get_source_port(target_in_edge)};
469 
470  VertPort control_succ = {
471  circ_.target(control_out_edge), circ_.get_target_port(control_out_edge)};
472  VertPort central_succ = {
473  circ_.target(central_edge), circ_.get_target_port(central_edge)};
474  VertPort target_succ = {
475  circ_.target(target_out_edge), circ_.get_target_port(target_out_edge)};
476 
477  // remove old vertex, add new vertex
478  circ_.remove_vertex(
479  to_be_replaced, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
480  Vertex bridge_vert = circ_.add_vertex(new_bridge_ptr);
481  // add Boolean edges
482  for (std::tuple<Vertex, port_t, port_t> vpp : classical_edge_info) {
483  circ_.add_edge(
484  {std::get<0>(vpp), std::get<1>(vpp)}, {bridge_vert, std::get<2>(vpp)},
485  EdgeType::Boolean);
486  }
487 
488  unsigned num_classicals = classical_edge_info.size();
489  // add control qubit in edge
490  Edge control_in = circ_.add_edge(
491  control_pred, {bridge_vert, num_classicals}, EdgeType::Quantum);
492  // add control qubit out edge
493  Edge control_out = circ_.add_edge(
494  {bridge_vert, num_classicals}, control_succ, EdgeType::Quantum);
495  // add central qubit in edge
496  Edge central_in = circ_.add_edge(
497  central_pred, {bridge_vert, num_classicals + 1}, EdgeType::Quantum);
498  // add central qubit out edge
499  Edge central_out = circ_.add_edge(
500  {bridge_vert, num_classicals + 1}, central_succ, EdgeType::Quantum);
501  // add target qubit in edge
502  Edge target_in = circ_.add_edge(
503  target_pred, {bridge_vert, num_classicals + 2}, EdgeType::Quantum);
504  // add target qubit out edge
505  Edge target_out = circ_.add_edge(
506  {bridge_vert, num_classicals + 2}, target_succ, EdgeType::Quantum);
507 
508  // Remove central_edge which is now going through BRIDGE vertex
509  circ_.remove_edge(central_edge);
510 
511  unit_frontier_t::iterator control_qb_in_it =
512  slice_frontier_.quantum_in_edges->find(control_qb);
513  unit_frontier_t::iterator central_qb_in_it =
514  slice_frontier_.quantum_in_edges->find(central_qb);
515  unit_frontier_t::iterator target_qb_in_it =
516  slice_frontier_.quantum_in_edges->find(target_qb);
517 
518  slice_frontier_.quantum_in_edges->replace(
519  control_qb_in_it, {control_qb, control_in});
520  slice_frontier_.quantum_in_edges->replace(
521  central_qb_in_it, {central_qb, central_in});
522  slice_frontier_.quantum_in_edges->replace(
523  target_qb_in_it, {target_qb, target_in});
524 
525  // Update slice frontier out edges
526  unit_frontier_t::iterator control_qb_out_it =
527  slice_frontier_.quantum_out_edges->find(control_qb);
528  unit_frontier_t::iterator central_qb_out_it =
529  slice_frontier_.quantum_out_edges->find(central_qb);
530  unit_frontier_t::iterator target_qb_out_it =
531  slice_frontier_.quantum_out_edges->find(target_qb);
532 
533  slice_frontier_.quantum_out_edges->replace(
534  control_qb_out_it, {control_qb, control_out});
535  slice_frontier_.quantum_out_edges->replace(
536  central_qb_out_it, {central_qb, central_out});
537  slice_frontier_.quantum_out_edges->replace(
538  target_qb_out_it, {target_qb, target_out});
539 
540  // Remove CX vertex from Slice (i.e. VertexVec) in slice_frontier-
541  slice_frontier_.slice->erase(
542  std::remove(
543  slice_frontier_.slice->begin(), slice_frontier_.slice->end(),
544  to_be_replaced),
545  slice_frontier_.slice->end());
546  slice_frontier_.slice->push_back(bridge_vert);
547 }

549 // Suitable swap found, amend all global constructs

558  // = | X
559  // --- --X--/ \--

562 
563  // find edges using qubits
564  EdgeVec preds = {
565  slice_frontier_.quantum_in_edges->find(qb1)->second,
566  slice_frontier_.quantum_in_edges->find(qb2)->second};
567 
568  Vertex swap_v = circ_.add_vertex(OpType::SWAP);
569  circ_.rewire(swap_v, preds, {EdgeType::Quantum, EdgeType::Quantum});

578  slice_frontier_.quantum_in_edges->find(qb2);
579  slice_frontier_.quantum_in_edges->replace(qb2_in_it, {qb2, swap_outs[1]});

581  slice_frontier_.quantum_out_edges->find(qb1);

583  slice_frontier_.quantum_out_edges->find(qb2);
