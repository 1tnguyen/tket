38 /* Helpers `run_bfs` and `run_dfs` return instances of TreeSearchBase
39  * subclasses. These provide a host of useful member functions to obtain the
40  * relevant results from the tree search.
41  *
42  * Note that run() must have been called before
43  * using the member functions to retrieve results.
44  * This is done automatically by the utility functions in `TreeSearch.hpp`
45  *
46  * First template argument is Graph type, second is the boost property map type
47  * that should be used as a vertex index.
48  */
49 template <typename Graph, typename BoostPMap>
50 class TreeSearchBase {
51  protected:
52  using vertex = utils::vertex<Graph>;
53  using size_t = std::size_t;
54  using parent_vec = std::vector<vertex>;
55  using color_vec = std::vector<boost::default_color_type>;
56  using index_pmap_t = BoostPMap;
57  using dist_pmap_t =
58  boost::iterator_property_map<typename dist_vec::iterator, index_pmap_t>;
59  using parent_pmap_t =
60  boost::iterator_property_map<typename parent_vec::iterator, index_pmap_t>;
61  using color_pmap_t =
62  boost::iterator_property_map<typename color_vec::iterator, index_pmap_t>;
63  using visitor_t = std::pair<
64  boost::distance_recorder<dist_pmap_t, boost::on_tree_edge>,
65  boost::predecessor_recorder<parent_pmap_t, boost::on_tree_edge>>;
66 
67  public:
68  /* Constructor: search through Graph g with root v, using index map pmap */
69  TreeSearchBase(vertex v, Graph&& g, BoostPMap pmap)
70  : root(v),
71  to_index(pmap),
72  graph(g),
73  dists(boost::num_vertices(g)),
74  parents(boost::num_vertices(g)),
75  color_map(boost::num_vertices(g)),
76  visitor{
77  boost::record_distances(
78  boost::make_iterator_property_map(dists.begin(), to_index),
79  boost::on_tree_edge()),
80  boost::record_predecessors(
81  boost::make_iterator_property_map(parents.begin(), to_index),
82  boost::on_tree_edge())} {
83  // for every vertex, assign self as parent
84  for (auto v : boost::make_iterator_range(boost::vertices(graph))) {
85  parents[to_index[v]] = v;
86  }
87  }
88 
89  /* Get results from tree search */
90  /* vector of vertex predecessor in search */
91  const parent_vec& get_parents() const& { return parents; }
92  const parent_vec&& get_parents() const&& { return std::move(parents); }
93  /* vector of distances from the root */
94  const dist_vec& get_dists() const& { return dists; }
95  const dist_vec&& get_dists() const&& { return std::move(dists); }
96  size_t get_dist(const vertex& v) const { return dists[to_index[v]]; }
97 
98  /* Rebase tree search from some new root */
99  void change_root(vertex v) {
100  // naive implementation
101  // This might be made more efficient in the subclasses
102  vertex old_root = root;
103  root = v;
104  if (root != old_root) {
105  std::fill(dists.begin(), dists.end(), 0);
106  std::iota(parents.begin(), parents.end(), 0);
107  run();
108  }
109  }
110 
111  /* Vector from target to root (following reverse edges) */
112  parent_vec path_to_root(vertex target) const {
113  std::vector<vertex> path;
114 
115  vertex current = target;
116  path.push_back(target);
117  while (current != root) {
118  vertex parent = parents[to_index[current]];
119  if (parent == current) {
120  // graph is disconnected and there is no path to root
121  return {};
122  }
123  current = parent;
124  path.push_back(current);
125  }
126  return path;
127  }
128 
129  /* Vector from root to target along graph edges */
130  parent_vec path_from_root(vertex target) const {
131  parent_vec out = path_to_root(target);
132  std::reverse(out.begin(), out.end());
133  return out;
134  }
135 
136  /* Depth of tree search */
137  size_t max_depth() const {
138  auto it = std::max_element(dists.cbegin(), dists.cend());
139  if (it == dists.cend()) {
140  throw std::invalid_argument(
141  "TreeSearch::max_depth: There is no entry in distance "
142  "vector");
143  }
144  return *it;
145  }
146 
147  /* Vertex of max depth
148  * if more than one exist, returns first found */
149  vertex max_depth_vertex() const {
150  const size_t target_depth = max_depth();
151  for (vertex v : boost::make_iterator_range(boost::vertices(this->graph))) {
152  if (get_dist(v) == target_depth) {
153  return v;
154  }
155  }
156  throw std::runtime_error("max_depth_vertex: No vertex has maximal depth");
157  }
158 
159  /* Path from root to deepest vertex */
160  parent_vec longest_path() const { return path_from_root(max_depth_vertex()); }
161 
162  /* Run tree search */
163  virtual void run() = 0;
164 
165  protected:
166  template <typename Visitor>
167  auto get_default_named_params(Visitor& vis) {
168  return boost::visitor(vis).color_map(
169  boost::make_iterator_property_map(color_map.begin(), to_index));
170  }
171 
172  vertex root; // root of tree search
173  index_pmap_t to_index; // vertex index pmap
174  Graph graph; // graph to be searched
175  dist_vec dists; // distance from root
176  parent_vec parents; // parent towards root
177  color_vec color_map; // color used by tree search algorithm
178  visitor_t visitor; // visitor passed to boost tree search alg
179 };

181 /* A vertex index map, mapping boost vertex descriptors to unsigned
182  * is required for tree searches (to index the parents and dists vectors).
183  *
184  * The TreeSearch class accepts a std::map-like PMap argument to define such a
185  * mapping. This is converted to a boost property map using
186  * boost::associative_property_map. If none is provided, the default
187  * boost::vertex_index is used (by default, this is available on boost graphs
188  * using the vecS underlying EdgeOutList type
189  */
190 
191 /* Note on the implementation:
192  * The third template argument (a bool) should always be left to its
193  * default value. It is used to flag whether the Graph type considered
194  * has an implicit integer vertex index.
195  * This flag is used in the template specialisation below to change
196  * the template definition when `HasImplicitIndex == True`,
197  * i.e. the first definition of `TreeSearch` is for the case `HasImplicitIndex
198  * == False` while the template partial specialisation below is for the case
199  * `HasImplicitIndex == True`.
200  */
201 template <
202  typename Graph, typename PMap = void,
203  bool HasImplicitIndex = utils::detail::has_integral_index<Graph>>
204 class TreeSearch
205  : public TreeSearchBase<Graph, boost::associative_property_map<PMap>> {
206  private:
207  using Base = TreeSearchBase<Graph, boost::associative_property_map<PMap>>;
208  using vertex = typename Base::vertex;
209 
210  public:
211  TreeSearch(vertex v, Graph&& g, PMap& pmap)
212  : Base(v, std::forward<Graph>(g), boost::make_assoc_property_map(pmap)) {}
213 };

232 /* Return type of `run_bfs`.
233  *
234  * For each vertex, computes the BFS distance to the root
235  * as well as the next node on the path towards root (parent).
236  * These are stored in the base class TreeSearchBase and can
237  * be retrieved through one of the member functions defined therein
238  */
239 template <typename... Args>
240 class BFS : public detail::TreeSearch<Args...> {
241  private:
242  using Base = detail::TreeSearch<Args...>;
243 
244  public:
245  // inherit constructors
246  using Base::Base;
247 
248  using parent_vec = typename Base::parent_vec;
249 
250  /* Run bfs */
251  void run() override {
252  auto vis = boost::make_bfs_visitor(this->visitor);
253  boost::breadth_first_search(
254  this->graph, this->root, this->get_default_named_params(vis));
255  }
256 };
257 
258 /* Return type of `run_dfs`.
259  *
260  * For each vertex, computes the DFS distance to the root
261  * as well as the next node on the path towards root (parent).
262  * These are stored in the base class TreeSearchBase and can
263  * be retrieved through one of the member functions defined therein
264  */
265 template <typename... Args>
266 class DFS : public detail::TreeSearch<Args...> {
267  private:
268  using Base = detail::TreeSearch<Args...>;
269 
270  public:
271  // inherit constructors
272  using Base::Base;
273 
274  using parent_vec = typename Base::parent_vec;
275 
276  /* Run dfs */
277  void run() override {
278  auto vis = boost::make_dfs_visitor(this->visitor);
279  boost::depth_first_search(
280  this->graph,
281  this->get_default_named_params(vis).root_vertex(this->root));
282  }
283 };
