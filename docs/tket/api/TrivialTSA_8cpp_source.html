31 // Make an arrow from each nonempty vertex to its target;
32 // what are the connected components of the resulting directed graph?
33 // Two different arrows cannot point INTO the same vertex.
34 // So, EITHER a cycle (so, a, abstract cyclic shift on tokens is performed),
35 // OR a path, with all except the final vertex being nonempty.
36 // In either case, we enact a cyclic shift.
37 
38 // To find a component, we might have to go backwards along arrows
39 // as well as forwards.

47 bool TrivialTSA::grow_cycle_forwards(
48  const VertexMapping& vertex_mapping, Endpoints& endpoints) {
49  auto current_id = endpoints.first;
50  const auto start_vertex = m_abstract_cycles_vertices.at(current_id);
51 
52  // If valid, a single cycle contains at most one empty vertex.
53  // Thus there are at most N+1 vertices.
54  for (size_t infin_loop_guard = vertex_mapping.size() + 1;
55  infin_loop_guard != 0; --infin_loop_guard) {
56  const auto v1 = m_abstract_cycles_vertices.at(current_id);
57  const auto citer = vertex_mapping.find(v1);
58  if (citer == vertex_mapping.cend()) {
59  // We end at an empty vertex.
60  endpoints.second = current_id;
61  return false;
62  }
63  if (citer->second == start_vertex) {
64  // We've hit the start.
65  endpoints.second = current_id;
66  return true;
67  }
68  current_id = m_abstract_cycles_vertices.insert_after(current_id);
69  m_abstract_cycles_vertices.at(current_id) = citer->second;
70  }
71  TKET_ASSERT(!"TrivialTSA::grow_cycle_forwards: "
72  "hit vertex count limit; invalid vertex mapping");
73  return false;
74 }

76 void TrivialTSA::grow_cycle_backwards(Endpoints& endpoints) {
77  auto current_id = endpoints.first;
78 
79  // In a valid cycle, every vertex but one (the empty vertex)
80  // is the target of something, and therefore there are <= N+1 vertices.
81  for (size_t infin_loop_guard = m_reversed_vertex_mapping.size() + 1;
82  infin_loop_guard != 0; --infin_loop_guard) {
83  const auto v1 = m_abstract_cycles_vertices.at(current_id);
84  const auto citer = m_reversed_vertex_mapping.find(v1);
85  if (citer == m_reversed_vertex_mapping.cend()) {
86  // Our vertex is not the target of anything.
87  // So, it's the START.
88  endpoints.first = current_id;
89  return;
90  }
91  // Remember the reverse order!
92  current_id = m_abstract_cycles_vertices.insert_before(current_id);
93  m_abstract_cycles_vertices.at(current_id) = citer->second;
94  }
95  TKET_ASSERT(!"TrivialTSA::grow_cycle_backwards: "
96  "hit vertex count limit; invalid vertex mapping");
97 }

99 void TrivialTSA::do_final_checks() const {
100  m_vertices_seen.clear();
101  for (const auto& entry : m_reversed_vertex_mapping) {
102  m_vertices_seen.insert(entry.first);
103  m_vertices_seen.insert(entry.second);
104  }
105  TKET_ASSERT(m_vertices_seen.size() == m_abstract_cycles_vertices.size());
106 
107  // Erase them again...!
108  for (const auto& endpoints : m_cycle_endpoints) {
109  for (auto id = endpoints.first;;
110  id = m_abstract_cycles_vertices.next(id).value()) {
111  // GCOVR_EXCL_START
112  TKET_ASSERT(
113  m_vertices_seen.erase(m_abstract_cycles_vertices.at(id)) == 1);
114  // GCOVR_EXCL_STOP
115  if (id == endpoints.second) {
116  break;
117  }
118  }
119  }
120  TKET_ASSERT(m_vertices_seen.empty());
121 }

123 void TrivialTSA::fill_disjoint_abstract_cycles(
124  const VertexMapping& vertex_mapping) {
125  m_vertices_seen.clear();
126  m_abstract_cycles_vertices.clear();
127  m_cycle_endpoints.clear();
128  m_reversed_vertex_mapping = get_reversed_map(vertex_mapping);
129  Endpoints endpoints;
130 
131  // Get the disjoint abstract cycles.
132  for (const auto& entry : vertex_mapping) {
133  if (m_vertices_seen.count(entry.first) != 0) {
134  continue;
135  }
136  m_abstract_cycles_vertices.push_back(entry.first);
137  endpoints.first = m_abstract_cycles_vertices.back_id().value();
138  if (!grow_cycle_forwards(vertex_mapping, endpoints)) {
139  grow_cycle_backwards(endpoints);
140  }
141  m_cycle_endpoints.push_back(endpoints);
142 
143  // Now, add the vertices to vertices seen...
144  for (auto id = endpoints.first;;
145  id = m_abstract_cycles_vertices.next(id).value()) {
146  // GCOVR_EXCL_START
147  TKET_ASSERT(
148  m_vertices_seen.insert(m_abstract_cycles_vertices.at(id)).second);
149  // GCOVR_EXCL_STOP
150  if (id == endpoints.second) {
151  break;
152  }
153  }
154  }
155 }

164 void TrivialTSA::append_partial_solution(
165  SwapList& swaps, VertexMapping& vertex_mapping,
166  DistancesInterface& distances, RiverFlowPathFinder& path_finder) {
167  if (all_tokens_home(vertex_mapping)) {
168  return;
169  }
170  fill_disjoint_abstract_cycles(vertex_mapping);
171  do_final_checks();
172 
173  if (m_options == Options::FULL_TSA) {
174  // OK, below, for a single cycle, we use CyclicShiftCostEstimate
175  // to estimate, not ONLY the cheapest single cycle, but ALSO
176  // the start vertex to enact it most cheaply.
177  // We could do that here also and it might save a bit,
178  // BUT the full Trivial TSA is really only used for testing now
179  // so don't bother.
180  append_partial_solution_with_all_cycles(swaps, vertex_mapping, path_finder);
181  return;
182  }
183  TKET_ASSERT(m_options == Options::BREAK_AFTER_PROGRESS);
184  // We're only going to do ONE cycle; so find which cycle
185  // has the shortest estimated number of swaps
186  size_t best_estimated_concrete_swaps = std::numeric_limits<size_t>::max();
187  Endpoints best_endpoints;
188  size_t start_v_index = std::numeric_limits<size_t>::max();
189 
190  for (const auto& endpoints : m_cycle_endpoints) {
191  copy_vertices_to_work_vector(endpoints);
192  if (m_vertices_work_vector.size() < 2) {
193  TKET_ASSERT(m_vertices_work_vector.size() == 1);
194  continue;
195  }
196  const CyclicShiftCostEstimate estimate(m_vertices_work_vector, distances);
197  // GCOVR_EXCL_START
198  TKET_ASSERT(
199  estimate.estimated_concrete_swaps < std::numeric_limits<size_t>::max());
200  TKET_ASSERT(estimate.start_v_index < m_vertices_work_vector.size());
201  // GCOVR_EXCL_STOP
202  if (estimate.estimated_concrete_swaps < best_estimated_concrete_swaps) {
203  best_estimated_concrete_swaps = estimate.estimated_concrete_swaps;
204  start_v_index = estimate.start_v_index;
205  best_endpoints = endpoints;
206  }
207  }
208  // GCOVR_EXCL_START
209  TKET_ASSERT(
210  best_estimated_concrete_swaps < std::numeric_limits<size_t>::max());
211  // GCOVR_EXCL_STOP
212  const auto swap_size_before = swaps.size();
213  const auto decrease = append_partial_solution_with_single_cycle(
214  best_endpoints, start_v_index, swaps, vertex_mapping, distances,
215  path_finder);
216  TKET_ASSERT(swap_size_before < swaps.size());
217  TKET_ASSERT(decrease > 0);
218 }

220 void TrivialTSA::copy_vertices_to_work_vector(const Endpoints& endpoints) {
221  m_vertices_work_vector.clear();
222  for (auto id = endpoints.first;;
223  id = m_abstract_cycles_vertices.next(id).value()) {
224  m_vertices_work_vector.push_back(m_abstract_cycles_vertices.at(id));
225  if (id == endpoints.second) {
226  break;
227  }
228  }

231 void TrivialTSA::append_partial_solution_with_all_cycles(
232  SwapList& swaps, VertexMapping& vertex_mapping,
233  RiverFlowPathFinder& path_finder) {
234  for (const auto& endpoints : m_cycle_endpoints) {
235  copy_vertices_to_work_vector(endpoints);
236  if (m_vertices_work_vector.size() < 2) {
237  continue;
238  }
239  // Break the abstract cycle into abstract swaps...
240  // To shift: [a,b,c,d] -> [d,a,b,c], we do abstract swaps in
241  // opposite order of the shift direction, i.e. cd bc ab
242  for (size_t ii = m_vertices_work_vector.size() - 1; ii > 0; --ii) {
243  // Abstract swap(v1, v2).
244  const auto v1 = m_vertices_work_vector[ii];
245  const auto v2 = m_vertices_work_vector[ii - 1];
246  TKET_ASSERT(v1 != v2);
247  const auto& path = path_finder(v1, v2);
248  TKET_ASSERT(path.size() >= 2);
249  append_swaps_to_interchange_path_ends(path, vertex_mapping, swaps);
250  }
251  }
252 }

254 size_t TrivialTSA::append_partial_solution_with_single_cycle(
255  const Endpoints& endpoints, size_t start_v_index, SwapList& swaps,
256  VertexMapping& vertex_mapping, DistancesInterface& distances,
257  RiverFlowPathFinder& path_finder) {
258  copy_vertices_to_work_vector(endpoints);
259  TKET_ASSERT(m_vertices_work_vector.size() >= 2);
260  TKET_ASSERT(start_v_index < m_vertices_work_vector.size());
261 
262  // Can go negative! But MUST be >= 1 at the end
263  // (otherwise this cycle was useless and should never have occurred).
264  int current_L_decrease = 0;
265 
266  // To shift: [a,b,c,d] -> [d,a,b,c], we do abstract swaps in the opposite
267  // order to the shift direction, i.e. cd bc ab
268  for (size_t ii = m_vertices_work_vector.size() - 1; ii > 0; --ii) {
269  // Abstract swap(v1, v2).
270  const auto v1 = m_vertices_work_vector
271  [(ii + start_v_index) % m_vertices_work_vector.size()];
272 
273  const auto v2 = m_vertices_work_vector
274  [((ii - 1) + start_v_index) % m_vertices_work_vector.size()];
275 
276  TKET_ASSERT(v1 != v2);
277  const auto& path = path_finder(v1, v2);
278  TKET_ASSERT(path.size() >= 2);
279 
280  // e.g., to swap endpoints: [x,a,b,c,y] -> [y,a,b,c,x],
281  // do concrete swaps xa ab bc cy bc ab xa.
282 
283  // xa ab bc cy ...(ascending)
284  for (size_t jj = 1; jj < path.size(); ++jj) {
285  current_L_decrease +=
286  get_swap_decrease(vertex_mapping, path[jj], path[jj - 1], distances);
287 
288  VertexSwapResult(path[jj], path[jj - 1], vertex_mapping, swaps);
289  if (current_L_decrease > 0) {
290  return static_cast<size_t>(current_L_decrease);
291  }
292  }
293  // Now the reverse: bc ab xa
294  for (size_t kk = path.size() - 2; kk > 0; --kk) {
295  current_L_decrease +=
296  get_swap_decrease(vertex_mapping, path[kk], path[kk - 1], distances);
297 
298  VertexSwapResult(path[kk], path[kk - 1], vertex_mapping, swaps);
299  if (current_L_decrease > 0) {
300  return static_cast<size_t>(current_L_decrease);
301  }
302  }
303  }
304  // The cycle MUST have decreased L overall,
305  // otherwise we shouldn't have done it.
306  TKET_ASSERT(!"TrivialTSA::append_partial_solution_with_single_cycle");
307  return 0;
308 }
