65  if (!uid_exists(edge.first) || !uid_exists(edge.second)) {

76  }
77  utils::remove_edge_with_map(

161 // Returns the path between the nodes |root| and |target|.
162 template <typename UID_t, typename OutEdgeListS, typename VertexListS>
163 std::vector<UID_t>
164 UIDConnectivityBase<UID_t, OutEdgeListS, VertexListS>::get_path(
165  const UID_t root, const UID_t target) const {
166  if (!uid_exists(root) || !uid_exists(target)) {
167  throw UIDDoesNotExistError(
168  "Trying to get path between non-existent vertices");
169  }
170  using parent_vec = typename BFS<UndirectedConnGraph>::parent_vec;
171 
172  const UndirectedConnGraph& g = get_undirected_connectivity();
173  auto bfs = run_bfs(to_vertices(root), g);
174 
175  auto to_uid = [&g](UndirectedVertex v) { return g[v].uid; };
176  parent_vec path = bfs.path_to_root(to_vertices(target));
177  std::vector<UID_t> converted_path(path.size());
178  std::transform(path.begin(), path.end(), converted_path.begin(), to_uid);
179  return converted_path;
180 }

219 template <typename UID_t, typename OutEdgeListS, typename VertexListS>
220 std::set<UID_t>
221 UIDConnectivityBase<UID_t, OutEdgeListS, VertexListS>::get_neighbour_uids(
222  const UID_t v) const {
223  if (!uid_exists(v)) {
224  throw UIDDoesNotExistError(
225  "Trying to get neighbours from non-existent vertex");
226  }
227  std::set<UID_t> neighbours;
228  for (auto [it, end] = boost::out_edges(to_vertices(v), graph); it != end;
229  ++it) {
230  neighbours.insert(UID_t(get_uid(boost::target(*it, graph))));
231  }
232  for (auto [it, end] = boost::in_edges(to_vertices(v), graph); it != end;
233  ++it) {
234  neighbours.insert(UID_t(get_uid(boost::source(*it, graph))));
235  }
236  return neighbours;
237 }

291 template <typename UID_t, typename OutEdgeListS, typename VertexListS>
292 bool UIDConnectivityBase<UID_t, OutEdgeListS, VertexListS>::operator==(
293  const UIDConnectivityBase<UID_t, OutEdgeListS, VertexListS>& other) const {
294  std::set<UID_t> uids = this->get_all_uids_set();
295  if (uids != other.get_all_uids_set()) return false;
296  for (const UID_t& u : uids) {
297  for (const UID_t& v : uids) {
298  if (this->connection_exists(u, v)) {
299  if (!other.connection_exists(u, v))
300  return false;
301  else if (
302  this->get_connection_weight(u, v) !=
303  other.get_connection_weight(u, v))
304  return false;
305  } else if (other.connection_exists(u, v))
306  return false;
307  }
308  }
309  return true;
310 }

353 template <typename UID_t, typename OutEdgeListS, typename VertexListS>
354 std::size_t UIDConnectivity<UID_t, OutEdgeListS, VertexListS>::get_distance(
355  const UID_t uid1, const UID_t uid2) const {
356  if (uid1 == uid2) {
357  return 0;
358  }
359  size_t d;
360  if (distance_cache.find(uid1) != distance_cache.end()) {
361  d = distance_cache[uid1][this->to_vertices(uid2)];
362  } else if (distance_cache.find(uid2) != distance_cache.end()) {
363  d = distance_cache[uid2][this->to_vertices(uid1)];
364  } else {
365  distance_cache[uid1] = Base::get_distances(uid1);
366  d = distance_cache[uid1][this->to_vertices(uid2)];
367  }
368  if (d == 0) {
369  throw UIDsNotConnected(uid1, uid2);
370  }
371  return d;
372 }
