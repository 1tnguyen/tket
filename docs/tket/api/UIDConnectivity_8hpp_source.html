111  void remove_uid(const UID_t uid) {
112  if (!uid_exists(uid)) {
113  throw UIDDoesNotExistError(
114  "The UID passed to UIDConnectivity::remove_uid must "
115  "exist!");
116  }
117  Vertex v = to_vertices(uid);
118  boost::clear_vertex(v, graph);
119  utils::remove_vertex_with_map(v, graph, uid_to_vertex.right);
