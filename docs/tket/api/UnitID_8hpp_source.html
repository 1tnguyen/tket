64 class UnitID {
65  public:
66  UnitID() : data_(std::make_shared<UnitData>()) {}
67 
69  std::string repr() const;
70 
72  std::string reg_name() const { return data_->name_; }
73 
75  unsigned reg_dim() const { return data_->index_.size(); }
76 
78  std::vector<unsigned> index() const { return data_->index_; }
79 
81  UnitType type() const { return data_->type_; }
82 
84  register_info_t reg_info() const { return {type(), reg_dim()}; }
85 
86  bool operator<(const UnitID &other) const {
87  int n = data_->name_.compare(other.data_->name_);
88  if (n > 0) return false;
89  if (n < 0) return true;
90  return data_->index_ < other.data_->index_;
91  }
92  bool operator==(const UnitID &other) const {
93  return (this->data_->name_ == other.data_->name_) &&
94  (this->data_->index_ == other.data_->index_);
95  }
96  bool operator!=(const UnitID &other) const { return !(*this == other); }
97 
98  friend std::size_t hash_value(UnitID const &unitid) {
99  size_t seed = 0;
100  boost::hash_combine(seed, unitid.data_->name_);
101  boost::hash_combine(seed, unitid.data_->index_);
102  boost::hash_combine(seed, unitid.data_->type_);
103  return seed;
104  }
105 
106  protected:
107  UnitID(
108  const std::string &name, const std::vector<unsigned> &index,
109  UnitType type)
110  : data_(std::make_shared<UnitData>(name, index, type)) {}
111 
112  private:
113  struct UnitData {
114  std::string name_;
115  std::vector<unsigned> index_;
116  UnitType type_;
117 
118  UnitData() : name_(), index_(), type_(UnitType::Qubit) {}
119  UnitData(
120  const std::string &name, const std::vector<unsigned> &index,
121  UnitType type)
122  : name_(name), index_(index), type_(type) {
123  static const std::string id_regex_str = "[a-z][A-Za-z0-9_]*";
124  static const std::regex id_regex(id_regex_str);
125  if (!name.empty() && !std::regex_match(name, id_regex)) {
126  std::stringstream msg;
127  msg << "UnitID name '" << name << "' does not match '" << id_regex_str
128  << "', as required for QASM conversion.";
129  tket_log()->warn(msg.str());
130  }
131  }
132  };
133  std::shared_ptr<UnitData> data_;
134 };
