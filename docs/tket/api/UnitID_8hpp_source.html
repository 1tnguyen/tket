65 class UnitID {
66  public:
67  UnitID() : data_(std::make_shared<UnitData>()) {}
68 
70  std::string repr() const;
71 
73  std::string reg_name() const { return data_->name_; }
74 
76  unsigned reg_dim() const { return data_->index_.size(); }
77 
79  std::vector<unsigned> index() const { return data_->index_; }
80 
82  UnitType type() const { return data_->type_; }
83 
85  register_info_t reg_info() const { return {type(), reg_dim()}; }
86 
87  bool operator<(const UnitID &other) const {
88  int n = data_->name_.compare(other.data_->name_);
89  if (n > 0) return false;
90  if (n < 0) return true;
91  return data_->index_ < other.data_->index_;
92  }
93  bool operator==(const UnitID &other) const {
94  return (this->data_->name_ == other.data_->name_) &&
95  (this->data_->index_ == other.data_->index_);
96  }
97  bool operator!=(const UnitID &other) const { return !(*this == other); }
98 
99  friend std::size_t hash_value(UnitID const &unitid) {
100  size_t seed = 0;
101  boost::hash_combine(seed, unitid.data_->name_);
102  boost::hash_combine(seed, unitid.data_->index_);
103  boost::hash_combine(seed, unitid.data_->type_);
104  return seed;
105  }
106 
107  protected:
108  UnitID(
109  const std::string &name, const std::vector<unsigned> &index,
110  UnitType type)
111  : data_(std::make_shared<UnitData>(name, index, type)) {}
112 
113  private:
114  struct UnitData {
115  std::string name_;
116  std::vector<unsigned> index_;
117  UnitType type_;
118 
119  UnitData() : name_(), index_(), type_(UnitType::Qubit) {}
120  UnitData(
121  const std::string &name, const std::vector<unsigned> &index,
122  UnitType type)
123  : name_(name), index_(index), type_(type) {
124  static const std::string id_regex_str = "[a-z][A-Za-z0-9_]*";
125  static const std::regex id_regex(id_regex_str);
126  if (!name.empty() && !std::regex_match(name, id_regex)) {
127  std::stringstream msg;
128  msg << "UnitID name '" << name << "' does not match '" << id_regex_str
129  << "', as required for QASM conversion.";
130  tket_log()->warn(msg.str());
131  }
132  }
133  };
134  std::shared_ptr<UnitData> data_;
135 };
