38 void VectorListHybridSkeleton::clear() {
39  if (m_links.empty()) {
40  TKET_ASSERT(m_size == 0);
41  TKET_ASSERT(m_front == INVALID_INDEX);
42  TKET_ASSERT(m_back == INVALID_INDEX);
43  TKET_ASSERT(m_deleted_front == INVALID_INDEX);
44  return;
45  }
46  m_size = 0;
47  m_front = INVALID_INDEX;
48  m_back = INVALID_INDEX;
49  for (Index nn = 1; nn < m_links.size(); ++nn) {
50  // Not strictly necessary, as deleted links are only a forward list;
51  // but make absolutely sure no leakage of prior internal link data can
52  // occur.
53  m_links[nn].previous = nn - 1;
54  m_links[nn - 1].next = nn;
55  }
56  m_links[0].previous = INVALID_INDEX;
57  m_links.back().next = INVALID_INDEX;
58  m_deleted_front = 0;
59 }

61 void VectorListHybridSkeleton::fast_clear() {
62  if (m_back == INVALID_INDEX) {
63  // No elements stored currently; nothing to do.
64  TKET_ASSERT(m_size == 0);
65  TKET_ASSERT(m_front == INVALID_INDEX);
66  return;
67  }
68  TKET_ASSERT(m_size > 0);
69  TKET_ASSERT(m_front != INVALID_INDEX);
70  TKET_ASSERT(m_links[m_back].next == INVALID_INDEX);
71  // There are some existing elements.
72  // Recall that deleted elements are ONLY a forward list,
73  // so we don't need to update "previous".
74  // To combine existing active elements with
75  // existing deleted elements,
76  // the valid elements will be joined to
77  // the start of the deleted list.
78  if (m_deleted_front != INVALID_INDEX) {
79  m_links[m_back].next = m_deleted_front;
80  }
81  // Convert "active" elements into deleted elements.
82  m_deleted_front = m_front;
83  m_front = INVALID_INDEX;
84  m_back = INVALID_INDEX;
85  m_size = 0;
86 }

88 void VectorListHybridSkeleton::reverse() {
89  if (m_size <= 1) {
90  // Nothing to do.
91  return;
92  }
93  TKET_ASSERT(m_front != INVALID_INDEX);
94  TKET_ASSERT(m_back != INVALID_INDEX);
95  TKET_ASSERT(m_front != m_back);
96  // The deleted element links don't need to change.
97  {
98  auto current_index = m_front;
99  bool terminated_correctly = false;
100  for (auto infinite_loop_guard = 1 + m_links.size(); infinite_loop_guard > 0;
101  --infinite_loop_guard) {
102  auto& link = m_links[current_index];
103  const auto next_index = link.next;
104  std::swap(link.next, link.previous);
105  if (next_index >= m_links.size()) {
106  TKET_ASSERT(next_index == INVALID_INDEX);
107  terminated_correctly = true;
108  break;
109  }
110  current_index = next_index;
111  }
112  TKET_ASSERT(terminated_correctly);
113  }
114  std::swap(m_front, m_back);
115 }

131 void VectorListHybridSkeleton::erase(Index index) {
132  --m_size;
133  auto& current_link = m_links[index];
134  if (current_link.previous == INVALID_INDEX) {
135  // We're erasing the front.
136  m_front = current_link.next;
137  } else {
138  m_links[current_link.previous].next = current_link.next;
139  }
140  if (current_link.next == INVALID_INDEX) {
141  // We're erasing the back.
142  m_back = current_link.previous;
143  } else {
144  m_links[current_link.next].previous = current_link.previous;
145  }
146  // Recall: deleted elements are a forward list ONLY.
147  current_link.next = m_deleted_front;
148  m_deleted_front = index;
149 }

151 void VectorListHybridSkeleton::erase_interval(
152  Index index, size_t number_of_elements) {
153  if (number_of_elements == 0) {
154  return;
155  }
156  // First, find the index of the LAST element to be erased.
157  // Notice that this is the only O(N) part; the rest are O(1).
158  // We update only O(1) links in total, not O(N),
159  // so slightly faster than a loop of next/erase calls.
160  Index last_element_index = index;
161  for (size_t nn = 1; nn < number_of_elements; ++nn) {
162  last_element_index = m_links.at(last_element_index).next;
163 
164  // GCOVR_EXCL_START
165  TKET_ASSERT(
166  last_element_index < m_links.size() ||
167  AssertMessage() << "erase_interval with start index " << index
168  << ", number_of_elements=" << number_of_elements
169  << ", size " << m_links.size()
170  << ", runs out of elements at N=" << nn
171  << " (got index " << last_element_index << ")");
172  // GCOVR_EXCL_STOP
173  }
174  TKET_ASSERT(number_of_elements <= m_size);
175  m_size -= number_of_elements;
176 
177  // Now, splice the soon-to-be-logically-erased interval into the deleted
178  // elements. Start the new deleted list at the erased interval.
179  const auto index_of_node_after_interval = m_links[last_element_index].next;
180 
181  // Correct whether or not m_deleted_front equals INVALID_INDEX.
182  m_links[last_element_index].next = m_deleted_front;
183  // No need to update previous, since the deleted nodes are only a forward
184  // list.
185  m_deleted_front = index;
186 
187  // Link the node BEFORE the interval to the new next node.
188  const auto index_of_node_before_interval = m_links[index].previous;
189 
190  if (index_of_node_before_interval < m_links.size()) {
191  // There IS a previous node to be dealt with.
192  auto& next_node_index_ref = m_links[index_of_node_before_interval].next;
193  TKET_ASSERT(next_node_index_ref == index);
194  // This is correct even if index_of_node_after_interval is INVALID_INDEX.
195  next_node_index_ref = index_of_node_after_interval;
196  TKET_ASSERT(m_front != index);
197  } else {
198  // No previous node, we must have been at the start already.
199  TKET_ASSERT(index_of_node_before_interval == INVALID_INDEX);
200  TKET_ASSERT(m_front == index);
201  m_front = index_of_node_after_interval;
202  }
203  // Link the node AFTER the interval to the new previous node.
204  if (index_of_node_after_interval < m_links.size()) {
205  // There are more unerased elements after the interval,
206  // so the first one must be dealt with.
207  auto& prev_node_index = m_links[index_of_node_after_interval].previous;
208  TKET_ASSERT(prev_node_index == last_element_index);
209  // Correct even if there IS no node before the interval.
210  prev_node_index = index_of_node_before_interval;
211  TKET_ASSERT(m_back != last_element_index);
212  } else {
213  // No node after, we have erased up to the back.
214  TKET_ASSERT(index_of_node_after_interval == INVALID_INDEX);
215  TKET_ASSERT(m_back == last_element_index);
216  m_back = index_of_node_before_interval;
217  }
218  if (m_size == 0) {
219  TKET_ASSERT(m_front == INVALID_INDEX);
220  TKET_ASSERT(m_back == INVALID_INDEX);
221  } else {
222  TKET_ASSERT(m_front < m_links.size());
223  TKET_ASSERT(m_back < m_links.size());
224  if (m_size == 1) {
225  TKET_ASSERT(m_front == m_back);
226  }
227  }
228 }

266 Index VectorListHybridSkeleton::get_new_index() {
267  ++m_size;
268  if (m_deleted_front == INVALID_INDEX) {
269  // We need to create a new element, it's full.
270  m_links.emplace_back();
271  return m_links.size() - 1;
272  }
273  // Reuse a deleted element.
274  const auto old_deleted_front = m_deleted_front;
275  m_deleted_front = m_links[old_deleted_front].next;
276  return old_deleted_front;
277 }
