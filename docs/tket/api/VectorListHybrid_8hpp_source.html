465 template <class T>
466 template <class CIter>
467 OverwriteIntervalResult VectorListHybrid<T>::overwrite_interval(
468  typename VectorListHybrid<T>::ID id, const CIter& new_elements_cbegin,
469  const CIter& new_elements_cend) {
470  // The links are unchanged; only the elements need to be changed.
471  OverwriteIntervalResult result;
472  result.final_overwritten_element_id = id;
473  CIter citer = new_elements_cbegin;
474  TKET_ASSERT(citer != new_elements_cend);
475  const auto max_number_of_elements = m_links_data.size();
476  result.number_of_overwritten_elements = 0;
477  for (;;) {
478  m_data.at(result.final_overwritten_element_id) = *citer;
479  ++result.number_of_overwritten_elements;
480  // GCOVR_EXCL_START
481  TKET_ASSERT(
482  result.number_of_overwritten_elements <= max_number_of_elements);
483  // GCOVR_EXCL_STOP
484  ++citer;
485  if (citer == new_elements_cend) {
486  return result;
487  }
488  // There IS another element, where will it be overwritten?
489  result.final_overwritten_element_id =
490  m_links_data.next(result.final_overwritten_element_id);
491  }
492  // Should be impossible to reach here
493  TKET_ASSERT(false);
494 }
