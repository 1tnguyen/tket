30 const vector<size_t>& VertexMapResizing::operator()(size_t vertex) {
31  const auto citer = m_cached_neighbours.find(vertex);
32  if (citer != m_cached_neighbours.cend()) {
33  return citer->second;

35  auto& list = m_cached_neighbours[vertex];
36  list = m_neighbours(vertex);

43 const VertexMapResizing::Result& VertexMapResizing::resize_mapping(
44  VertexMapping& mapping, unsigned desired_size) {
45  m_result.success = false;
46  m_result.edges.clear();
47  if (mapping.size() > desired_size) {
48  for (auto infinite_loop_guard = 1 + mapping.size(); infinite_loop_guard > 0;
49  --infinite_loop_guard) {
50  const auto old_size = mapping.size();
51  remove_vertex(mapping);
52  const auto new_size = mapping.size();
53  if (new_size <= desired_size) {
54  fill_result_edges(mapping);
55  m_result.success = true;
56  return m_result;
57  }
58  if (old_size <= new_size) {
59  return m_result;
60  }
61  }
62  TKET_ASSERT(!"VertexMapResizing::resize_mapping");
63  }
64  TKET_ASSERT(mapping.size() <= desired_size);
65  bool terminated_correctly = false;
66  for (auto infinite_loop_guard = 1 + desired_size; infinite_loop_guard > 0;
67  --infinite_loop_guard) {
68  const auto old_size = mapping.size();
69  if (old_size >= desired_size) {
70  terminated_correctly = true;
71  break;
72  }
73  add_vertex(mapping);
74  const auto new_size = mapping.size();
75  if (old_size == new_size) {
76  // Couldn't add a vertex.
77  terminated_correctly = true;
78  break;
79  }
80  // Must have added exactly one vertex.
81  TKET_ASSERT(old_size + 1 == new_size);
82  }
83  TKET_ASSERT(terminated_correctly);
84  // It's acceptable to have too few vertices,
85  // it can still be looked up in the table.
86  m_result.success = true;
87  fill_result_edges(mapping);
88  return m_result;
89 }

92  const VertexMapping& mapping, size_t vertex) {
93  const auto& neighbours = operator()(vertex);
94  return std::count_if(
95  neighbours.cbegin(), neighbours.cend(),
96  // Note that "neighbours" automatically will not contain "vertex" itself.
97  [&mapping](size_t vertex) { return mapping.count(vertex) != 0; });

100 void VertexMapResizing::add_vertex(VertexMapping& mapping) {
101  std::set<size_t> new_vertices;
102 
103  // Multipass, maybe a bit inefficient, but doesn't matter.
104  // After a few calls, it's just map lookup so not so bad.
105  for (const auto& existing_vertex_pair : mapping) {
106  // A valid mapping should have the same source/target vertices,
107  // so don't need to consider .second.
108  const auto& neighbours = operator()(existing_vertex_pair.first);
109  for (auto vv : neighbours) {
110  if (mapping.count(vv) == 0) {
111  new_vertices.insert(vv);
112  }
113  }
114  }
115 
116  // Now find the new vertex which would add the largest number of new edges.
117  size_t maximum_new_edges = 0;
118  size_t best_new_vertex = std::numeric_limits<size_t>::max();
119 
120  for (auto new_v : new_vertices) {
121  const auto edge_count = get_edge_count(mapping, new_v);
122  if (edge_count > maximum_new_edges) {
123  best_new_vertex = new_v;
124  maximum_new_edges = edge_count;
125  }
126  }
127  if (maximum_new_edges > 0) {
128  mapping[best_new_vertex] = best_new_vertex;
129  }
130 }

132 void VertexMapResizing::remove_vertex(VertexMapping& mapping) {
133  const auto invalid_number_of_edges = std::numeric_limits<size_t>::max();
134 
135  // We want to leave as many edges as possible,
136  // so we remove the minimum number.
137  size_t minimum_edges_removed = invalid_number_of_edges;
138  size_t best_vertex = std::numeric_limits<size_t>::max();
139  for (const auto& existing_vertex_pair : mapping) {
140  if (existing_vertex_pair.first != existing_vertex_pair.second) {
141  // The vertex is not fixed, so we cannot remove it.
142  continue;
143  }
144  const auto edge_count = get_edge_count(mapping, existing_vertex_pair.first);
145  if (edge_count < minimum_edges_removed) {
146  best_vertex = existing_vertex_pair.first;
147  minimum_edges_removed = edge_count;
148  }
149  }
150  if (minimum_edges_removed < invalid_number_of_edges) {
151  TKET_ASSERT(mapping.at(best_vertex) == best_vertex);
152  TKET_ASSERT(mapping.erase(best_vertex) == 1);
153  }
154 }

156 void VertexMapResizing::fill_result_edges(const VertexMapping& mapping) {
157  m_result.edges.clear();
158  for (auto citer1 = mapping.cbegin(); citer1 != mapping.cend(); ++citer1) {
159  auto citer2 = citer1;
160  for (++citer2; citer2 != mapping.cend(); ++citer2) {
161  const auto edge = get_swap(citer1->first, citer2->first);
162  if (m_cached_full_edges.count(edge) != 0) {
163  m_result.edges.push_back(edge);
164  }
165  }
166  }
