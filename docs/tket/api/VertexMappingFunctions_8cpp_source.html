71 size_t get_source_vertex(
72  VertexMapping& source_to_target_map, size_t target_vertex) {
73  if (source_to_target_map.count(target_vertex) == 0) {
74  // If it IS a genuine permutation mapping (which we assume),
75  // then the vertex is as yet unmentioned (and hence unmoved).
76  source_to_target_map[target_vertex] = target_vertex;
77  return target_vertex;
78  }
79  for (const auto& entry : source_to_target_map) {
80  if (entry.second == target_vertex) {
81  return entry.first;
82  }
83  }
84  TKET_ASSERT(!"get_source_vertex");
85  return target_vertex;
86 }
