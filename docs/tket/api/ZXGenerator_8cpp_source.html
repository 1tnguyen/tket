70 ZXGen_ptr ZXGen::create_gen(ZXType type, QuantumType qtype) {
71  ZXGen_ptr op;
72  switch (type) {
73  case ZXType::Input:
74  case ZXType::Output:
75  case ZXType::Open: {
76  op = std::make_shared<const BoundaryGen>(type, qtype);
77  break;
78  }
79  case ZXType::ZSpider:
80  case ZXType::XSpider: {
81  op = std::make_shared<const BasicGen>(type, 0., qtype);
82  break;
83  }
84  case ZXType::Hbox: {
85  op = std::make_shared<const BasicGen>(type, -1., qtype);
86  break;
87  }
88  case ZXType::Triangle: {
89  op = std::make_shared<const DirectedGen>(type, qtype);
90  break;
91  }
92  default:
93  throw ZXError("Cannot instantiate a ZXGen of the required type");
94  }
95  return op;
96 }

300  if (!port) return false;

326  ZXVertVec boundary = diag_->get_boundary();
327  std::vector<QuantumType> sig;
328  for (const ZXVert& b : boundary) {
