70 std::vector<qubit_bimap_t> monomorphism_edge_break(
71  const Architecture& arc, const QubitGraph& q_graph, unsigned max_matches,
72  unsigned timeout) {
73  if (q_graph.n_uids() > arc.n_uids()) {
74  throw ArchitectureInvalidity(
75  "Interaction graph too large for architecture");
76  }
77 
78  using ArchitectureConn = Architecture::UndirectedConnGraph;
79  using InteractionGraph = QubitGraph::UndirectedConnGraph;
80 
81  ArchitectureConn undirected_target = arc.get_undirected_connectivity();
82  InteractionGraph undirected_pattern = q_graph.get_undirected_connectivity();
83 
84  std::vector<qubit_bimap_t> all_maps;
85  std::chrono::time_point<std::chrono::steady_clock> end_time =
86  std::chrono::steady_clock::now() + std::chrono::milliseconds(timeout);
87 
88  while (true) {
89  long search_timeout = std::chrono::duration_cast<std::chrono::milliseconds>(
90  (end_time - std::chrono::steady_clock::now()) / 2)
91  .count();
92  if (search_timeout <= 0) search_timeout = 1;
93  vf2_match_add_callback<InteractionGraph, ArchitectureConn> callback(
94  all_maps, undirected_pattern, undirected_target, max_matches);
95  bool found_monomorphism = boost::vf2_subgraph_mono(
96  undirected_pattern, undirected_target, callback, search_timeout);
97 
98  if (std::chrono::steady_clock::now() >= end_time) {
99  tket_log()->warn(
100  "boost::vf2_subgraph_mono reached {} millisecond timeout before "
101  "reaching set max matches {}, instead finding {} matches. "
102  "Please change PlacementConfig.timeout to allow more matches.",
103  timeout, callback.max, all_maps.size());
104  if (all_maps.empty()) {
105  throw std::runtime_error("No mappings found before timeout.");
106  }
107  return all_maps;
108  } else if (found_monomorphism) {
109  return all_maps;
110  } else if (boost::num_edges(undirected_pattern) != 0) {
111  using edge_t = graphs::utils::edge<InteractionGraph>;
112  auto e_its = boost::edges(undirected_pattern);
113  auto max_e_it = boost::first_max_element(
114  e_its.first, e_its.second,
115  [&undirected_pattern](const edge_t& lhs, const edge_t& rhs) {
116  return undirected_pattern[lhs].weight <
117  undirected_pattern[rhs].weight;
118  });
119  graphs::utils::remove_edge(*max_e_it, undirected_pattern, true);
120  }
121  }
122 }
