45 std::vector<qubit_bimap_t> monomorphism_edge_break(
46  const Architecture& arc, const QubitGraph& q_graph, unsigned max_matches,
47  unsigned timeout) {
48  if (q_graph.n_nodes() > arc.n_nodes()) {
49  throw ArchitectureInvalidity(
50  "Interaction graph too large for architecture");
51  }
52 
53  using ArchitectureConn = Architecture::UndirectedConnGraph;
54  using InteractionGraph = QubitGraph::UndirectedConnGraph;
55 
56  ArchitectureConn undirected_target = arc.get_undirected_connectivity();
57  InteractionGraph undirected_pattern = q_graph.get_undirected_connectivity();
58 
59  std::vector<qubit_bimap_t> all_maps;
60  std::chrono::time_point<std::chrono::steady_clock> end_time =
61  std::chrono::steady_clock::now() + std::chrono::milliseconds(timeout);
62 
63  while (true) {
64  long search_timeout = std::chrono::duration_cast<std::chrono::milliseconds>(
65  (end_time - std::chrono::steady_clock::now()) / 2)
66  .count();
67  if (search_timeout <= 0) search_timeout = 1;
68  vf2_match_add_callback<InteractionGraph, ArchitectureConn> callback(
69  all_maps, undirected_pattern, undirected_target, max_matches);
70  bool found_monomorphism = boost::vf2_subgraph_mono(
71  undirected_pattern, undirected_target, callback, search_timeout);
72 
73  if (std::chrono::steady_clock::now() >= end_time) {
74  std::stringstream ss;
75  ss << "boost::vf2_subgraph_mono reached " << timeout
76  << " millisecond timeout before reaching set max matches "
77  << callback.max << ", instead finding " << all_maps.size()
78  << " matches. "
79  "Please change PlacementConfig.timeout to allow more matches.";
80  tket_log()->warn(ss.str());
81  if (all_maps.empty()) {
82  throw std::runtime_error("No mappings found before timeout.");
83  }
84  std::sort(all_maps.begin(), all_maps.end());
85  return all_maps;
86  } else if (found_monomorphism) {
87  std::sort(all_maps.begin(), all_maps.end());
88  return all_maps;
89  } else if (boost::num_edges(undirected_pattern) != 0) {
90  using edge_t = graphs::utils::edge<InteractionGraph>;
91  auto e_its = boost::edges(undirected_pattern);
92  auto max_e_it = boost::first_max_element(
93  e_its.first, e_its.second,
94  [&undirected_pattern](const edge_t& lhs, const edge_t& rhs) {
95  return undirected_pattern[lhs].weight <
96  undirected_pattern[rhs].weight;
97  });
98  graphs::utils::remove_edge(*max_e_it, undirected_pattern, true);
99  }
100  }
101 }
